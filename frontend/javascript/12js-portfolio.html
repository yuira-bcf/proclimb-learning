/**
 * ローダーモジュール
 * ページ読み込み完了時にローディング画面を非表示にする
 */

const LOADER_ID = 'loader';
const HIDDEN_CLASS = 'hidden';
const MIN_DISPLAY_TIME = 500; // 最低表示時間（ms）

/**
 * ローダーを非表示にする
 */
function hideLoader() {
  const loader = document.getElementById(LOADER_ID);

  if (!loader) {
    console.warn('Loader element not found');
    return;
  }

  loader.classList.add(HIDDEN_CLASS);

  // トランジション完了後にDOMから削除（任意）
  loader.addEventListener('transitionend', () => {
    loader.remove();
  }, { once: true });
}

/**
 * ローダーを初期化
 */
export function initLoader() {
  const startTime = Date.now();

  window.addEventListener('load', () => {
    const elapsedTime = Date.now() - startTime;
    const remainingTime = Math.max(0, MIN_DISPLAY_TIME - elapsedTime);

    // 最低表示時間を確保してから非表示
    setTimeout(hideLoader, remainingTime);
  });
}
実装ポイント
  • 最低表示時間：読み込みが速すぎるとチラつくため、最低500msは表示
  • visibility + opacity：opacityだけだとクリックが効いてしまうので両方使用
  • transitionend：アニメーション完了後にDOMから削除して軽量化

JavaScript 
/**
 * ナビゲーションモジュール
 * ハンバーガーメニューとスムーズスクロール
 */

const SELECTORS = {
  toggle: '.nav-toggle',
  menu: '.nav-menu',
  links: '.nav-link[href^="#"]',
  header: '.header'
};

const ACTIVE_CLASS = 'active';
const SCROLL_OFFSET = 80; // ヘッダーの高さ分オフセット

/**
 * ハンバーガーメニューの開閉
 */
function toggleMenu() {
  const toggle = document.querySelector(SELECTORS.toggle);
  const menu = document.querySelector(SELECTORS.menu);

  toggle.classList.toggle(ACTIVE_CLASS);
  menu.classList.toggle(ACTIVE_CLASS);

  // body のスクロールを制御
  document.body.style.overflow =
    menu.classList.contains(ACTIVE_CLASS) ? 'hidden' : '';
}

/**
 * メニューを閉じる
 */
function closeMenu() {
  const toggle = document.querySelector(SELECTORS.toggle);
  const menu = document.querySelector(SELECTORS.menu);

  toggle.classList.remove(ACTIVE_CLASS);
  menu.classList.remove(ACTIVE_CLASS);
  document.body.style.overflow = '';
}

/**
 * スムーズスクロール
 */
function smoothScroll(e) {
  e.preventDefault();

  const targetId = e.currentTarget.getAttribute('href');
  const targetElement = document.querySelector(targetId);

  if (!targetElement) return;

  const targetPosition =
    targetElement.getBoundingClientRect().top +
    window.scrollY -
    SCROLL_OFFSET;

  window.scrollTo({
    top: targetPosition,
    behavior: 'smooth'
  });

  // メニューを閉じる
  closeMenu();
}

/**
 * スクロール時のヘッダー変化
 */
function handleScroll() {
  const header = document.querySelector(SELECTORS.header);

  if (window.scrollY > 100) {
    header.classList.add('scrolled');
  } else {
    header.classList.remove('scrolled');
  }
}

/**
 * ナビゲーションを初期化
 */
export function initNavigation() {
  const toggle = document.querySelector(SELECTORS.toggle);
  const links = document.querySelectorAll(SELECTORS.links);

  // ハンバーガーメニュー
  if (toggle) {
    toggle.addEventListener('click', toggleMenu);
  }

  // スムーズスクロール
  links.forEach(link => {
    link.addEventListener('click', smoothScroll);
  });

  // スクロールイベント
  window.addEventListener('scroll', handleScroll, { passive: true });

  // Escキーでメニューを閉じる
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') closeMenu();
  });
}

/**
 * ギャラリーモジュール
 * モーダル表示とキーボード操作
 */

let currentIndex = 0;
let galleryItems = [];

/**
 * モーダルを作成
 */
function createModal() {
  const modal = document.createElement('div');
  modal.id = 'gallery-modal';
  modal.className = 'gallery-modal';
  modal.innerHTML = `
    <div class="gallery-modal-overlay"></div>
    <div class="gallery-modal-content">
      <button class="gallery-close" aria-label="閉じる">&times;</button>
      <button class="gallery-prev" aria-label="前へ">&lt;</button>
      <img class="gallery-modal-image" src="" alt="">
      <button class="gallery-next" aria-label="次へ">&gt;</button>
      <div class="gallery-caption"></div>
    </div>
  `;

  document.body.appendChild(modal);
  return modal;
}

/**
 * モーダルを開く
 */
function openModal(index) {
  let modal = document.getElementById('gallery-modal');

  if (!modal) {
    modal = createModal();
    setupModalEvents(modal);
  }

  currentIndex = index;
  updateModalContent(modal);

  modal.classList.add('active');
  document.body.style.overflow = 'hidden';
}

/**
 * モーダルを閉じる
 */
function closeModal() {
  const modal = document.getElementById('gallery-modal');
  if (modal) {
    modal.classList.remove('active');
    document.body.style.overflow = '';
  }
}

/**
 * 前の画像へ
 */
function showPrev() {
  currentIndex = (currentIndex - 1 + galleryItems.length) % galleryItems.length;
  updateModalContent(document.getElementById('gallery-modal'));
}

/**
 * 次の画像へ
 */
function showNext() {
  currentIndex = (currentIndex + 1) % galleryItems.length;
  updateModalContent(document.getElementById('gallery-modal'));
}

/**
 * モーダルの内容を更新
 */
function updateModalContent(modal) {
  const item = galleryItems[currentIndex];
  const img = modal.querySelector('.gallery-modal-image');
  const caption = modal.querySelector('.gallery-caption');

  img.src = item.src;
  img.alt = item.alt;
  caption.textContent = item.alt;
}

/**
 * モーダルのイベント設定
 */
function setupModalEvents(modal) {
  // 閉じるボタン・オーバーレイクリック
  modal.querySelector('.gallery-close').addEventListener('click', closeModal);
  modal.querySelector('.gallery-modal-overlay').addEventListener('click', closeModal);

  // 前後ボタン
  modal.querySelector('.gallery-prev').addEventListener('click', showPrev);
  modal.querySelector('.gallery-next').addEventListener('click', showNext);
}

/**
 * キーボード操作
 */
function handleKeydown(e) {
  const modal = document.getElementById('gallery-modal');
  if (!modal || !modal.classList.contains('active')) return;

  switch (e.key) {
    case 'Escape':
      closeModal();
      break;
    case 'ArrowLeft':
      showPrev();
      break;
    case 'ArrowRight':
      showNext();
      break;
  }
}

/**
 * ギャラリーを初期化
 */
export function initGallery() {
  const items = document.querySelectorAll('.gallery-item');

  galleryItems = Array.from(items).map(item => ({
    src: item.dataset.src || item.querySelector('img').src,
    alt: item.dataset.title || item.querySelector('img').alt
  }));

  items.forEach((item, index) => {
    item.addEventListener('click', () => openModal(index));
  });

  document.addEventListener('keydown', handleKeydown);
}

/**
 * スクロールアニメーションモジュール
 * Intersection Observerでスクロールアニメーションを制御
 */

const ANIMATION_CLASS = 'animate-on-scroll';
const VISIBLE_CLASS = 'visible';

const DEFAULT_OPTIONS = {
  threshold: 0.15,
  rootMargin: '0px 0px -50px 0px'
};

/**
 * アニメーション対象の要素にスタガー（遅延）を設定
 */
function setStaggerDelay(elements, baseDelay = 100) {
  elements.forEach((el, index) => {
    const delay = index * baseDelay;
    el.style.transitionDelay = `${delay}ms`;
  });
}

/**
 * スクロールアニメーションを初期化
 */
export function initScrollAnimations(options = {}) {
  const config = { ...DEFAULT_OPTIONS, ...options };

  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add(VISIBLE_CLASS);

        // 一度だけ実行
        observer.unobserve(entry.target);
      }
    });
  }, config);

  // アニメーション対象を登録
  const elements = document.querySelectorAll(`.${ANIMATION_CLASS}`);
  elements.forEach(el => observer.observe(el));

  // グループ要素にスタガー設定
  const staggerGroups = document.querySelectorAll('[data-stagger]');
  staggerGroups.forEach(group => {
    const children = group.querySelectorAll(`.${ANIMATION_CLASS}`);
    const delay = parseInt(group.dataset.stagger) || 100;
    setStaggerDelay(children, delay);
  });

  return observer;
}

/**
 * フォームバリデーションモジュール
 */

const RULES = {
  required: (value) => value.trim() !== '',
  email: (value) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value),
  minLength: (value, min) => value.length >= min,
  maxLength: (value, max) => value.length <= max
};

const MESSAGES = {
  required: 'この項目は必須です',
  email: '有効なメールアドレスを入力してください',
  minLength: (min) => `${min}文字以上で入力してください`,
  maxLength: (max) => `${max}文字以内で入力してください`
};

/**
 * フィールドをバリデーション
 */
function validateField(field) {
  const value = field.value;
  const rules = field.dataset.validate?.split('|') || [];
  const errors = [];

  rules.forEach(rule => {
    const [ruleName, param] = rule.split(':');

    if (RULES[ruleName]) {
      const isValid = RULES[ruleName](value, parseInt(param));

      if (!isValid) {
        const message = typeof MESSAGES[ruleName] === 'function'
          ? MESSAGES[ruleName](param)
          : MESSAGES[ruleName];
        errors.push(message);
      }
    }
  });

  return errors;
}

/**
 * エラーメッセージを表示
 */
function showError(field, errors) {
  const wrapper = field.closest('.form-group');
  let errorEl = wrapper.querySelector('.form-error');

  if (errors.length > 0) {
    field.classList.add('error');
    field.classList.remove('valid');

    if (!errorEl) {
      errorEl = document.createElement('div');
      errorEl.className = 'form-error';
      wrapper.appendChild(errorEl);
    }

    errorEl.textContent = errors[0];
  } else {
    field.classList.remove('error');
    field.classList.add('valid');

    if (errorEl) {
      errorEl.remove();
    }
  }

  return errors.length === 0;
}

/**
 * フォーム送信処理
 */
async function handleSubmit(e, form) {
  e.preventDefault();

  const fields = form.querySelectorAll('[data-validate]');
  let isValid = true;

  fields.forEach(field => {
    const errors = validateField(field);
    if (!showError(field, errors)) {
      isValid = false;
    }
  });

  if (isValid) {
    const formData = new FormData(form);
    const data = Object.fromEntries(formData);

    console.log('送信データ:', data);
    showSuccess(form);
  }
}

/**
 * 送信成功メッセージ
 */
function showSuccess(form) {
  const message = document.createElement('div');
  message.className = 'form-success';
  message.textContent = '送信が完了しました！';

  form.reset();
  form.insertAdjacentElement('beforebegin', message);

  setTimeout(() => message.remove(), 5000);
}

/**
 * フォームを初期化
 */
export function initForm(formSelector = '#contact-form') {
  const form = document.querySelector(formSelector);
  if (!form) return;

  const fields = form.querySelectorAll('[data-validate]');

  // リアルタイムバリデーション
  fields.forEach(field => {
    field.addEventListener('blur', () => {
      const errors = validateField(field);
      showError(field, errors);
    });

    field.addEventListener('input', () => {
      if (field.classList.contains('error')) {
        const errors = validateField(field);
        showError(field, errors);
      }
    });
  });

  // 送信処理
  form.addEventListener('submit', (e) => handleSubmit(e, form));
}

/**
 * API連携モジュール
 * GitHub APIからリポジトリ情報を取得
 */

const GITHUB_API = 'https://api.github.com';
const CACHE_KEY = 'github_repos';
const CACHE_DURATION = 60 * 60 * 1000; // 1時間

/**
 * キャッシュからデータを取得
 */
function getFromCache(key) {
  const cached = localStorage.getItem(key);
  if (!cached) return null;

  const { data, timestamp } = JSON.parse(cached);

  if (Date.now() - timestamp > CACHE_DURATION) {
    localStorage.removeItem(key);
    return null;
  }

  return data;
}

/**
 * キャッシュにデータを保存
 */
function saveToCache(key, data) {
  localStorage.setItem(key, JSON.stringify({
    data,
    timestamp: Date.now()
  }));
}

/**
 * GitHubリポジトリを取得
 */
export async function fetchGitHubRepos(username, limit = 6) {
  // キャッシュ確認
  const cached = getFromCache(`${CACHE_KEY}_${username}`);
  if (cached) return cached;

  try {
    const response = await fetch(
      `${GITHUB_API}/users/${username}/repos?sort=updated&per_page=${limit}`
    );

    if (!response.ok) {
      throw new Error(`GitHub API Error: ${response.status}`);
    }

    const repos = await response.json();

    // キャッシュに保存
    saveToCache(`${CACHE_KEY}_${username}`, repos);

    return repos;
  } catch (error) {
    console.error('Failed to fetch repos:', error);
    return [];
  }
}

/**
 * リポジトリカードを生成
 */
function createRepoCard(repo) {
  return `
    <div class="repo-card animate-on-scroll">
      <h3 class="repo-name">
        <a href="${repo.html_url}" target="_blank" rel="noopener">
          ${repo.name}
        </a>
      </h3>
      <p class="repo-description">
        ${repo.description || 'No description'}
      </p>
      <div class="repo-meta">
        ${repo.language ? `<span class="repo-language">${repo.language}</span>` : ''}
        <span class="repo-stars">★ ${repo.stargazers_count}</span>
        <span class="repo-forks">🍴 ${repo.forks_count}</span>
      </div>
    </div>
  `;
}

/**
 * リポジトリ一覧を表示
 */
export async function renderGitHubRepos(container, username) {
  const targetEl = document.querySelector(container);
  if (!targetEl) return;

  // ローディング表示
  targetEl.innerHTML = '<div class="loading">Loading...</div>';

  const repos = await fetchGitHubRepos(username);

  if (repos.length === 0) {
    targetEl.innerHTML = '<p>リポジトリが見つかりませんでした</p>';
    return;
  }

  targetEl.innerHTML = repos.map(createRepoCard).join('');
}
