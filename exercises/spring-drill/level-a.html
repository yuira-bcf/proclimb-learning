// DiscountStrategy.java
public interface DiscountStrategy {
    int apply(int price);
    String getType();
}

// NoneDiscount.java
@Component
public class NoneDiscount implements DiscountStrategy {
    public int apply(int price) { return price; }
    public String getType() { return "none"; }
}

// PercentDiscount.java
@Component
public class PercentDiscount implements DiscountStrategy {
    public int apply(int price) { return (int)(price * 0.9); }
    public String getType() { return "percent"; }
}

// FixedDiscount.java
@Component
public class FixedDiscount implements DiscountStrategy {
    public int apply(int price) { return Math.max(0, price - 500); }
    public String getType() { return "fixed"; }
}

// PriceService.java
@Service
public class PriceService {
    private final Map<String, DiscountStrategy> strategies;

    public PriceService(List<DiscountStrategy> strategyList) {
        this.strategies = strategyList.stream()
            .collect(Collectors.toMap(
                DiscountStrategy::getType, s -> s));
    }

    public int calculate(int amount, String discountType) {
        DiscountStrategy strategy = strategies.getOrDefault(
            discountType, strategies.get("none"));
        return strategy.apply(amount);
    }
}
解説
  • SpringはインタフェースのListを注入すると、全実装Beanを自動収集します。新しい割引戦略を追加してもService側の変更が不要です。
  • if-else/switchの分岐をMap検索に置き換えることで、Open-Closed Principle（拡張に開き、修正に閉じる）を実現しています。
  • getOrDefault でデフォルト戦略にフォールバックすることで、未知のタイプでもエラーになりません。

lightbulbヒント

@Around("within(@org.springframework.stereotype.Service *)") でService層を対象にします。ProceedingJoinPoint の proceed() でメソッドを実行し、前後で時間を計測します。

check_circle解答・解説 
@Aspect
@Component
public class ServiceLoggingAspect {

    private static final Logger log =
        LoggerFactory.getLogger(ServiceLoggingAspect.class);

    @Around("within(@org.springframework.stereotype.Service *)")
    public Object logServiceMethod(ProceedingJoinPoint pjp)
            throws Throwable {
        String className = pjp.getTarget().getClass().getSimpleName();
        String methodName = pjp.getSignature().getName();
        String target = className + "." + methodName;

        log.info("[START] {}", target);
        long start = System.currentTimeMillis();
        try {
            Object result = pjp.proceed();
            long elapsed = System.currentTimeMillis() - start;
            log.info("[END] {} ({}ms)", target, elapsed);
            return result;
        } catch (Throwable ex) {
            log.error("[ERROR] {}: {}", target, ex.getMessage());
            throw ex;
        }
    }
}
解説
  • @Around はメソッドの実行前後をラップできる最も強力なアドバイスタイプです。
  • pjp.proceed() を呼ばないと元のメソッドが実行されないため、必ず呼び出す必要があります。
  • AOPにより、全Serviceメソッドに手動でログを追加する必要がなくなり、DRY原則を実現します。

// OrderResponse.java
public class OrderResponse {
    private Long id;
    private String customerName;
    private String status;
    private Map<String, String> links = new LinkedHashMap<>();

    public static OrderResponse from(Order order, String baseUrl) {
        OrderResponse res = new OrderResponse();
        res.id = order.getId();
        res.customerName = order.getCustomerName();
        res.status = order.getStatus();
        res.links.put("self", baseUrl + "/api/orders/" + order.getId());
        res.links.put("items",
            baseUrl + "/api/orders/" + order.getId() + "/items");
        if ("PENDING".equals(order.getStatus())) {
            res.links.put("cancel",
                baseUrl + "/api/orders/" + order.getId() + "/cancel");
        }
        return res;
    }
}

// OrderController.java
@GetMapping("/{id}")
public OrderResponse getOrder(@PathVariable Long id,
        HttpServletRequest request) {
    Order order = orderService.findById(id);
    String baseUrl = ServletUriComponentsBuilder
        .fromRequestUri(request)
        .replacePath("").toUriString();
    return OrderResponse.from(order, baseUrl);
}

<!-- ReportMapper.xml -->
<select id="monthlySales" resultType="MonthlySalesReport">
  SELECT
    m.month,
    COALESCE(SUM(oi.price * oi.quantity), 0) AS totalSales,
    COUNT(DISTINCT o.id) AS orderCount,
    CASE WHEN COUNT(DISTINCT o.id) = 0 THEN 0
         ELSE COALESCE(SUM(oi.price * oi.quantity), 0)
              / COUNT(DISTINCT o.id)
    END AS avgOrderAmount
  FROM generate_series(1, 12) AS m(month)
  LEFT JOIN orders o
    ON EXTRACT(MONTH FROM o.order_date) = m.month
    AND EXTRACT(YEAR FROM o.order_date) = #{year}
  LEFT JOIN order_items oi ON o.id = oi.order_id
  GROUP BY m.month
  ORDER BY m.month
</select>

// UniqueEmail.java
@Target(ElementType.TYPE)
@Retention(RetentionPolicy.RUNTIME)
@Constraint(validatedBy = UniqueEmailValidator.class)
public @interface UniqueEmail {
    String message() default "このメールアドレスは既に使用されています";
    Class<?>[] groups() default {};
    Class<? extends Payload>[] payload() default {};
}

// UniqueEmailValidator.java
@Component
public class UniqueEmailValidator
        implements ConstraintValidator<UniqueEmail, UserForm> {

    private final UserMapper userMapper;

    public UniqueEmailValidator(UserMapper userMapper) {
        this.userMapper = userMapper;
    }

    @Override
    public boolean isValid(UserForm form,
            ConstraintValidatorContext context) {
        if (form.getEmail() == null) return true;
        User existing = userMapper.findByEmail(form.getEmail());
        if (existing == null) return true;
        // 更新時: 自身のレコードは除外
        if (form.getId() != null
                && existing.getId().equals(form.getId())) {
            return true;
        }
        return false;
    }
}
解説
  • ValidatorはSpring Beanとして管理されるため、Mapper等の他のBeanを注入できます。
  • 更新時に自身のメールアドレスが「重複」として検出されないよう、ID一致チェックで除外します。
  • DB制約（UNIQUE制約）も併用するのが安全です。バリデーションはユーザー向けメッセージ、DB制約は最終防衛線です。

// JwtTokenProvider.java
@Component
public class JwtTokenProvider {
    @Value("${jwt.secret}") private String secret;
    private static final long EXPIRATION_MS = 3600000; // 1時間

    public String generateToken(String username,
            List<String> roles) {
        return Jwts.builder()
            .subject(username)
            .claim("roles", roles)
            .issuedAt(new Date())
            .expiration(new Date(
                System.currentTimeMillis() + EXPIRATION_MS))
            .signWith(Keys.hmacShaKeyFor(secret.getBytes()))
            .compact();
    }

    public Claims parseToken(String token) {
        return Jwts.parser()
            .verifyWith(Keys.hmacShaKeyFor(secret.getBytes()))
            .build()
            .parseSignedClaims(token)
            .getPayload();
    }
}

// JwtAuthenticationFilter.java
@Component
public class JwtAuthenticationFilter
        extends OncePerRequestFilter {
    private final JwtTokenProvider tokenProvider;

    @Override
    protected void doFilterInternal(HttpServletRequest request,
            HttpServletResponse response, FilterChain chain)
            throws ServletException, IOException {
        String header = request.getHeader("Authorization");
        if (header != null && header.startsWith("Bearer ")) {
            String token = header.substring(7);
            try {
                Claims claims = tokenProvider.parseToken(token);
                String username = claims.getSubject();
                @SuppressWarnings("unchecked")
                List<String> roles =
                    claims.get("roles", List.class);
                var authorities = roles.stream()
                    .map(SimpleGrantedAuthority::new)
                    .toList();
                var auth = new UsernamePasswordAuthenticationToken(
                    username, null, authorities);
                SecurityContextHolder.getContext()
                    .setAuthentication(auth);
            } catch (ExpiredJwtException ex) {
                response.setStatus(401);
                response.setContentType("application/json");
                response.getWriter().write(
                    "{\"error\":\"トークンの有効期限が切れています\"}");
                return;
            }
        }
        chain.doFilter(request, response);
    }
}

// AuthController.java
@RestController
@RequestMapping("/api/auth")
public class AuthController {
    private final AuthenticationManager authManager;
    private final JwtTokenProvider tokenProvider;

    @PostMapping("/login")
    public ResponseEntity<Map<String, String>> login(
            @RequestBody LoginRequest req) {
        var authToken = new UsernamePasswordAuthenticationToken(
            req.getUsername(), req.getPassword());
        Authentication auth = authManager.authenticate(authToken);
        List<String> roles = auth.getAuthorities().stream()
            .map(GrantedAuthority::getAuthority).toList();
        String jwt = tokenProvider.generateToken(
            auth.getName(), roles);
        return ResponseEntity.ok(Map.of("token", jwt));
    }
}

  • JWT秘密鍵は環境変数やVaultから取得し、ソースコードにハードコーディングしないのが鉄則です。

public class ArticleService {

    public boolean isOwner(Long articleId, String username) {
        Article article = articleMapper.findById(articleId);
        return article != null
            && article.getAuthorName().equals(username);
    }

    @PreAuthorize("hasRole('ADMIN') or "
        + "@articleService.isOwner(#id, authentication.name)")
    @Transactional
    public ArticleResponse update(Long id, ArticleForm form) {
        Article article = articleMapper.findById(id);
        if (article == null) throw new EntityNotFoundException(
            "Article", id);
        article.setTitle(form.getTitle());
        article.setContent(form.getContent());
        article.setUpdatedAt(LocalDateTime.now());
        articleMapper.update(article);
        return ArticleResponse.from(article);
    }

    @PreAuthorize("hasRole('ADMIN')")
    @Transactional
    public void delete(Long id) {
        if (articleMapper.findById(id) == null)
            throw new EntityNotFoundException("Article", id);
        articleMapper.deleteById(id);
    }

public class CustomPermissionEvaluator

    private final ArticleMapper articleMapper;

    @Override
    public boolean hasPermission(Authentication auth,
            Object targetId, Object permission) {
        if ("ARTICLE_EDIT".equals(permission)) {
            Article article = articleMapper.findById(
                (Long) targetId);
            return article != null &&
                article.getAuthorName().equals(auth.getName());
        }
        return false;
    }

    @Override
    public boolean hasPermission(Authentication auth,
            Serializable targetId, String targetType,
            Object permission) {
        return hasPermission(auth, targetId, permission);
    }

// CouponService.java
@Service
public class CouponService {
    private final CouponMapper couponMapper;

    @Transactional
    public CouponResponse create(CouponForm form) {
        if (couponMapper.findByCode(form.getCode()) != null)
            throw new ConflictException("code", form.getCode());
        Coupon coupon = new Coupon(form.getCode(),
            form.getDiscount(), form.getExpiresAt());
        couponMapper.insert(coupon);
        return CouponResponse.from(coupon);
    }

    @Transactional
    public ApplyResult apply(String code, int amount) {
        Coupon coupon = couponMapper.findByCode(code);
        if (coupon == null)
            throw new ResourceNotFoundException("Coupon: " + code);
        if (coupon.getExpiresAt().isBefore(LocalDate.now()))
            throw new BadRequestException("クーポンの有効期限切れ");
        if (coupon.isUsed())
            throw new BadRequestException("使用済みクーポン");
        int discount = amount * coupon.getDiscount() / 100;
        couponMapper.markAsUsed(code);
        return new ApplyResult(amount, discount,
            amount - discount);
    }
}

// usecase/RegisterUserUseCase.java
@Service
public class RegisterUserUseCase {
    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;

    @Transactional
    public User execute(RegisterUserCommand command) {
        Email email = new Email(command.email());
        userRepository.findByEmail(email).ifPresent(u -> {
            throw new ConflictException("email", command.email());
        });
        Password hashedPassword = Password.fromRaw(
            command.password(), passwordEncoder);
        User user = new User(command.name(), email,
            hashedPassword);
        userRepository.save(user);
        return user;
    }
}

        return Optional.ofNullable(userMapper.findById(id));
    }
    public Optional<User> findByEmail(Email email) {
        return Optional.ofNullable(
            userMapper.findByEmail(email.value()));
    }
    public void save(User user) {
        if (user.getId() == null) userMapper.insert(user);
        else userMapper.update(user);
