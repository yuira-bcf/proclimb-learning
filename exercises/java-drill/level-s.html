import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

enum Category {
    FOOD("食費"), TRANSPORT("交通費"), ENTERTAINMENT("娯楽"),
    SALARY("給与"), OTHER("その他");

    private final String label;
    Category(String label) { this.label = label; }
    public String getLabel() { return label; }
}

class Entry {
    private LocalDate date;
    private int amount;
    private Category category;
    private String memo;

    public Entry(LocalDate date, int amount, Category category, String memo) {
        this.date = date;
        this.amount = amount;
        this.category = category;
        this.memo = memo;
    }

    public LocalDate getDate() { return date; }
    public int getAmount() { return amount; }
    public Category getCategory() { return category; }
    public String getMemo() { return memo; }
}

class Ledger {
    private List<Entry> entries = new ArrayList<>();

    public void add(Entry entry) {
        entries.add(entry);
    }

    public int totalByCategory(Category category) {
        int total = 0;
        for (Entry e : entries) {
            if (e.getCategory() == category) {
                total += e.getAmount();
            }
        }
        return total;
    }

    public String monthlyReport(int year, int month) {
        StringBuilder sb = new StringBuilder();
        sb.append(String.format("=== %d年%d月 月次レポート ===\n", year, month));
        sb.append(String.format("%-10s %-8s %8s %s\n", "日付", "カテゴリ", "金額", "メモ"));

        List<Entry> monthly = new ArrayList<>();
        for (Entry e : entries) {
            if (e.getDate().getYear() == year && e.getDate().getMonthValue() == month) {
                monthly.add(e);
            }
        }

        for (Entry e : monthly) {
            String sign = e.getAmount() >= 0 ? "+" : "";
            sb.append(String.format("%-10s %-8s %s%,d %s\n",
                e.getDate(), e.getCategory().getLabel(),
                sign, e.getAmount(), e.getMemo()));
        }

        sb.append("\n--- カテゴリ別集計 ---\n");
        int totalIncome = 0;
        int totalExpense = 0;
        for (Category c : Category.values()) {
            int catTotal = 0;
            for (Entry e : monthly) {
                if (e.getCategory() == c) {
                    catTotal += e.getAmount();
                }
            }
            if (catTotal != 0) {
                String sign = catTotal >= 0 ? "+" : "";
                sb.append(String.format("%-8s %s%,d\n", c.getLabel() + ":", sign, catTotal));
            }
            if (catTotal >= 0) {
                totalIncome += catTotal;
            } else {
                totalExpense += catTotal;
            }
        }

        sb.append(String.format("\n収入合計: +%,d\n", totalIncome));
        sb.append(String.format("支出合計: %,d\n", totalExpense));
        sb.append(String.format("残高:     +%,d", totalIncome + totalExpense));
        return sb.toString();
    }
}

import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;

class Student {
    private String id;
    private String name;

    public Student(String id, String name) {
        this.id = id;
        this.name = name;
    }

    public String getId() { return id; }
    public String getName() { return name; }
}

class Grade {
    private Student student;
    private String subject;
    private int score;

    public Grade(Student student, String subject, int score) {
        this.student = student;
        this.subject = subject;
        this.score = score;
    }

    public Student getStudent() { return student; }
    public String getSubject() { return subject; }
    public int getScore() { return score; }
}

class GradeManager {
    private List<Grade> grades = new ArrayList<>();

    public void addGrade(Grade grade) {
        grades.add(grade);
    }

    public double averageByStudent(String studentId) {
        int sum = 0;
        int count = 0;
        for (Grade g : grades) {
            if (g.getStudent().getId().equals(studentId)) {
                sum += g.getScore();
                count++;
            }
        }
        if (count == 0) return 0.0;
        return Math.round(sum / (double) count * 10.0) / 10.0;
    }

    public double averageBySubject(String subject) {
        int sum = 0;
        int count = 0;
        for (Grade g : grades) {
            if (g.getSubject().equals(subject)) {
                sum += g.getScore();
                count++;
            }
        }
        if (count == 0) return 0.0;
        return Math.round(sum / (double) count * 10.0) / 10.0;
    }

    public String ranking(String subject) {
        List<Grade> subjectGrades = new ArrayList<>();
        for (Grade g : grades) {
            if (g.getSubject().equals(subject)) {
                subjectGrades.add(g);
            }
        }
        Collections.sort(subjectGrades,
            Comparator.comparingInt(Grade::getScore).reversed());

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < subjectGrades.size(); i++) {
            Grade g = subjectGrades.get(i);
            sb.append(String.format("%d位: %s (%d点)\n",
                i + 1, g.getStudent().getName(), g.getScore()));
        }
        return sb.toString().trim();
    }

    public String passOrFail(String studentId, int threshold) {
        for (Grade g : grades) {
            if (g.getStudent().getId().equals(studentId)
                    && g.getScore() < threshold) {
                return "不合格";
            }
        }
        return "合格";
    }
}

import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.ArrayList;
import java.util.List;

class LogEntry {
    private static final DateTimeFormatter FMT =
        DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss");

    private LocalDateTime timestamp;
    private String level;
    private String message;

    public LogEntry(LocalDateTime timestamp, String level, String message) {
        this.timestamp = timestamp;
        this.level = level;
        this.message = message;
    }

    public static LogEntry parse(String line) {
        String dateStr = line.substring(0, 19);
        LocalDateTime ts = LocalDateTime.parse(dateStr, FMT);

        int bracketStart = line.indexOf('[');
        int bracketEnd = line.indexOf(']');
        String level = line.substring(bracketStart + 1, bracketEnd);

        String message = line.substring(bracketEnd + 2);
        return new LogEntry(ts, level, message);
    }

    public LocalDateTime getTimestamp() { return timestamp; }
    public String getLevel() { return level; }
    public String getMessage() { return message; }

    @Override
    public String toString() {
        return timestamp.format(FMT) + " [" + level + "] " + message;
    }
}

class LogAnalyzer {
    private List<LogEntry> logs = new ArrayList<>();

    public void addLog(String line) {
        logs.add(LogEntry.parse(line));
    }

    public List<LogEntry> filterByLevel(String level) {
        List<LogEntry> result = new ArrayList<>();
        for (LogEntry entry : logs) {
            if (entry.getLevel().equals(level)) {
                result.add(entry);
            }
        }
        return result;
    }

    public List<LogEntry> searchByKeyword(String keyword) {
        List<LogEntry> result = new ArrayList<>();
        for (LogEntry entry : logs) {
            if (entry.getMessage().contains(keyword)
                    || entry.toString().contains(keyword)) {
                result.add(entry);
            }
        }
        return result;
    }

    public String summary() {
        int infoCount = 0, warnCount = 0, errorCount = 0;
        for (LogEntry entry : logs) {
            switch (entry.getLevel()) {
                case "INFO":  infoCount++;  break;
                case "WARN":  warnCount++;  break;
                case "ERROR": errorCount++; break;
            }
        }
        StringBuilder sb = new StringBuilder();

class TestRunner {
    private List<TestResult> results = new ArrayList<>();

    public void assertEquals(String testName, Object expected, Object actual) {
        if (Objects.equals(expected, actual)) {
            results.add(new TestResult(testName, true, ""));
        } else {
            String msg = "expected: <" + expected + "> but was: <" + actual + ">";
            results.add(new TestResult(testName, false, msg));
        }
    }

    public void assertTrue(String testName, boolean condition) {
        if (condition) {
            results.add(new TestResult(testName, true, ""));
        } else {
            results.add(new TestResult(testName, false, "condition was false"));
        }
    }

    public void assertThrows(String testName,
            Class<? extends Exception> exType, Runnable action) {
        try {
            action.run();
            String msg = "expected " + exType.getSimpleName()
                       + " but none was thrown";
            results.add(new TestResult(testName, false, msg));
        } catch (Exception e) {
            if (exType.isInstance(e)) {
                results.add(new TestResult(testName, true, ""));
            } else {
                String msg = "expected " + exType.getSimpleName()
                           + " but was " + e.getClass().getSimpleName();
                results.add(new TestResult(testName, false, msg));
            }
        }
    }

    public void runAll() {
        System.out.println("=== テスト実行結果 ===");
        int passCount = 0;
        for (TestResult r : results) {
            if (r.isPassed()) {
                System.out.println("[PASS] " + r.getTestName());
                passCount++;
            } else {
                System.out.println("[FAIL] " + r.getTestName()
                    + " - " + r.getMessage());
            }
        }
        int total = results.size();
        int failCount = total - passCount;
        double rate = Math.round(passCount / (double) total * 1000.0) / 10.0;

class Theater {
    private Seat[][] seats;
    private static final int ROWS = 10;
    private static final int COLS = 5;

    public Theater() {
        seats = new Seat[ROWS][COLS];
        for (int r = 0; r < ROWS; r++) {
            for (int c = 0; c < COLS; c++) {
                seats[r][c] = new Seat(r, c);
            }
        }
    }

    public void reserve(int row, int col, String name) {
        seats[row][col].reserve(name);
    }

    public void cancel(int row, int col) {
        seats[row][col].cancel();
    }

    public String display() {
        StringBuilder sb = new StringBuilder();
        sb.append("    ");
        for (int c = 0; c < COLS; c++) {
            sb.append(String.format("%-4d", c + 1));
        }
        sb.append("\n");

        for (int r = 0; r < ROWS; r++) {
            sb.append((char)('A' + r)).append(" ");
            for (int c = 0; c < COLS; c++) {
                sb.append(seats[r][c].isReserved() ? "[X] " : "[ ] ");
            }
            sb.append("\n");
        }
        return sb.toString().trim();
    }

    public int countEmpty() {
        int count = 0;
        for (int r = 0; r < ROWS; r++) {
            for (int c = 0; c < COLS; c++) {
                if (!seats[r][c].isReserved()) {
                    count++;
                }
            }
        }
        return count;
    }

    public int findConsecutive(int row, int count) {
        int consecutive = 0;
        int startCol = 0;
        for (int c = 0; c < COLS; c++) {
            if (!seats[row][c].isReserved()) {
                if (consecutive == 0) {
                    startCol = c;
                }
                consecutive++;
                if (consecutive == count) {
                    return startCol;
                }
            } else {
                consecutive = 0;
            }
        }
        return -1;
    }

    public int getTotalSeats() {
        return ROWS * COLS;
    }
}

            for (int c = 0; c < getWidth(); c++) {

interface Filter {
    Image apply(Image source);
}

class BrightnessFilter implements Filter {
    private int amount;

    public BrightnessFilter(int amount) {
        this.amount = amount;
    }

    @Override
    public Image apply(Image source) {
        Image result = source.copy();
        for (int r = 0; r < result.getHeight(); r++) {
            for (int c = 0; c < result.getWidth(); c++) {
                result.setPixel(r, c, source.getPixel(r, c) + amount);
            }
        }
        return result;
    }
}

class InvertFilter implements Filter {
    @Override
    public Image apply(Image source) {
        Image result = source.copy();
        for (int r = 0; r < result.getHeight(); r++) {
            for (int c = 0; c < result.getWidth(); c++) {
                result.setPixel(r, c, 255 - source.getPixel(r, c));
            }
        }
        return result;
    }
}

class BlurFilter implements Filter {
    @Override
    public Image apply(Image source) {
        Image result = source.copy();
        int height = source.getHeight();
        int width  = source.getWidth();

        for (int r = 0; r < height; r++) {
            for (int c = 0; c < width; c++) {
                int sum = 0;
                int count = 0;
                for (int dr = -1; dr <= 1; dr++) {
                    for (int dc = -1; dc <= 1; dc++) {
                        int nr = r + dr;
                        int nc = c + dc;
                        if (nr >= 0 && nr < height
                                && nc >= 0 && nc < width) {
                            sum += source.getPixel(nr, nc);
                            count++;
                        }
                    }
                }
                result.setPixel(r, c, sum / count);
            }
        }
        return result;
    }
}

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;

enum Priority {
    HIGH("高"), MEDIUM("中"), LOW("低");

    private final String label;
    Priority(String label) { this.label = label; }
    public String getLabel() { return label; }
}

enum Status {
    TODO("TODO"), IN_PROGRESS("進行中"), DONE("完了");

    private final String label;
    Status(String label) { this.label = label; }
    public String getLabel() { return label; }

    public boolean canTransitionTo(Status next) {
        switch (this) {
            case TODO:        return next == IN_PROGRESS;
            case IN_PROGRESS: return next == DONE;
            case DONE:        return false;
            default:          return false;
        }
    }
}

class Task {
    private int id;
    private String title;
    private Priority priority;
    private Status status;
    private LocalDate deadline;

    public Task(int id, String title, Priority priority, LocalDate deadline) {
        this.id = id;
        this.title = title;
        this.priority = priority;
        this.status = Status.TODO;
        this.deadline = deadline;
    }

    public void changeStatus(Status next) {
        if (!status.canTransitionTo(next)) {
            throw new IllegalStateException(
                "不正な遷移: " + status.getLabel() + " → " + next.getLabel());
        }
        this.status = next;
    }

    public int getId() { return id; }
    public String getTitle() { return title; }
    public Priority getPriority() { return priority; }
    public Status getStatus() { return status; }
    public LocalDate getDeadline() { return deadline; }

    @Override
    public String toString() {
        return "#" + id + " [" + priority.getLabel() + "] "
            + title + " (" + status.getLabel() + ") 期限: " + deadline;
    }
}

class TaskManager {
    private List<Task> tasks = new ArrayList<>();

    public void add(Task task) { tasks.add(task); }

    public List<Task> filterByStatus(Status status) {
        List<Task> result = new ArrayList<>();
        for (Task t : tasks) {
            if (t.getStatus() == status) {
                result.add(t);
            }
        }
        return result;
    }

    public List<Task> filterByPriority(Priority priority) {
        List<Task> result = new ArrayList<>();
        for (Task t : tasks) {
            if (t.getPriority() == priority) {
                result.add(t);
            }
        }
        return result;
    }

    public List<Task> sortByDeadline() {
        List<Task> sorted = new ArrayList<>(tasks);
        Collections.sort(sorted,
            Comparator.comparing(Task::getDeadline));
        return sorted;
    }

    public String statistics() {
        int todoCount = 0, progressCount = 0, doneCount = 0;
        int overdueCount = 0;
        LocalDate today = LocalDate.now();

        for (Task t : tasks) {
            switch (t.getStatus()) {
                case TODO:        todoCount++;     break;
                case IN_PROGRESS: progressCount++; break;
                case DONE:        doneCount++;     break;
            }
            if (t.getStatus() != Status.DONE
                    && t.getDeadline().isBefore(today)) {
                overdueCount++;
            }
        }

        StringBuilder sb = new StringBuilder();
        sb.append("TODO:     ").append(todoCount).append("件\n");
        sb.append("進行中:   ").append(progressCount).append("件\n");
        sb.append("完了:     ").append(doneCount).append("件\n");
        sb.append("期限切れ: ").append(overdueCount).append("件");
        return sb.toString();
    }
}

import java.util.ArrayList;

class Employee {
    private static int nextId = 1;

    private int id;
    private String name;
    private String department;
    private LocalDate hireDate;
    private int salary;

    public Employee(String name, String department, LocalDate hireDate, int salary) {
        this.id = nextId++;
        this.name = name;
        this.department = department;
        this.hireDate = hireDate;
        this.salary = salary;
    }

    public int getId() { return id; }
    public String getName() { return name; }
    public String getDepartment() { return department; }
    public LocalDate getHireDate() { return hireDate; }
    public int getSalary() { return salary; }

    public long getYearsOfService() {
        return ChronoUnit.YEARS.between(hireDate, LocalDate.now());
    }

    @Override
    public String toString() {
        return String.format("[%d] %s (%s) 入社:%s 勤続%d年 給与:%,d",
            id, name, department, hireDate, getYearsOfService(), salary);
    }
}

class EmployeeManager {
    private List<Employee> employees = new ArrayList<>();

    public void add(Employee emp) {
        employees.add(emp);
    }

    public List<Employee> findByDepartment(String dept) {
        List<Employee> result = new ArrayList<>();
        for (Employee e : employees) {
            if (e.getDepartment().equals(dept)) {
                result.add(e);
            }
        }
        return result;
    }

    public Map<String, Integer> totalSalaryByDepartment() {
        Map<String, Integer> map = new HashMap<>();
        for (Employee e : employees) {
            map.merge(e.getDepartment(), e.getSalary(), Integer::sum);
        }
        return map;
    }

    public List<Employee> salaryRanking() {
        List<Employee> sorted = new ArrayList<>(employees);
        sorted.sort(Comparator.comparingInt(Employee::getSalary).reversed());
        return sorted;
    }

    public String exportCsv() {
        StringBuilder sb = new StringBuilder();
        sb.append("id,name,department,hireDate,salary\n");
        for (Employee e : employees) {
            sb.append(String.format("%d,%s,%s,%s,%d\n",
                e.getId(), e.getName(), e.getDepartment(),
                e.getHireDate(), e.getSalary()));
        }
        return sb.toString().trim();
    }
}

class RecipeBook {

    public void add(Recipe recipe) {
        recipes.add(recipe);
    }

    public List<Recipe> searchByIngredient(String ingredient) {
        List<Recipe> result = new ArrayList<>();
        for (Recipe r : recipes) {
            if (r.getIngredients().contains(ingredient)) {
                result.add(r);
            }
        }
        return result;

        List<Recipe> result = new ArrayList<>();
        for (Recipe r : recipes) {
            if (r.getDifficulty() == diff) {
                result.add(r);
            }
        }
        return result;
    }

        List<Recipe> sorted = new ArrayList<>(recipes);

    private List<CartItem> items = new ArrayList<>();

    public void addItem(Product product, int qty) {
        items.add(new CartItem(product, qty));
    }

    public int getTotalQuantity() {
        int total = 0;
        for (CartItem item : items) {
            total += item.getQuantity();
        }
        return total;
    }

    public int calculateSubtotal() {
        int subtotal = 0;
        for (CartItem item : items) {
            subtotal += item.getProduct().getPrice() * item.getQuantity();
        }
        return subtotal;
    }

    public int calculateShipping() {
        int shipping = 0;
        for (CartItem item : items) {
            if (item.getProduct() instanceof PhysicalProduct) {
                PhysicalProduct pp = (PhysicalProduct) item.getProduct();
                shipping += pp.getShippingCost() * item.getQuantity();
            }
        }
        return shipping;
    }

    public int calculateTotal(DiscountStrategy strategy) {
        int subtotal = calculateSubtotal();
        int discounted = strategy.apply(subtotal);
        return discounted + calculateShipping();
    }

    public void printCart() {
        for (CartItem item : items) {
            Product p = item.getProduct();
            String type = (p instanceof PhysicalProduct) ? "物理" : "デジタル";
            int shipping = 0;
            if (p instanceof PhysicalProduct) {
                shipping = ((PhysicalProduct) p).getShippingCost() * item.getQuantity();
            }
            System.out.printf("[%s] %s ¥%,d × %d (送料: ¥%,d)%n",
                type, p.getName(), p.getPrice(), item.getQuantity(), shipping);
            System.out.println("  → " + p.getDescription());
        }
    }

        int subtotal = strategy.apply(calculateSubtotal());
        int shipping = calculateShipping();
        System.out.printf("小計: ¥%,d  送料: ¥%,d  合計: ¥%,d%n",
            subtotal, shipping, subtotal + shipping);
    }

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

class StoreException extends Exception {
    public StoreException(String message) {
        super(message);
    }
}

class KeyNotFoundException extends StoreException {
    public KeyNotFoundException(String key) {
        super("キー '" + key + "' は存在しません");
    }
}

class DuplicateKeyException extends StoreException {
    public DuplicateKeyException(String key) {
        super("キー '" + key + "' は既に存在します");
    }
}

class StoreCorruptException extends StoreException {
    public StoreCorruptException(String detail) {
        super("データ整合性エラー: " + detail);
    }
}

interface KeyValueStore {
    void put(String key, String value) throws StoreException;
    String get(String key) throws StoreException;
    void delete(String key) throws StoreException;
    boolean exists(String key);
    List<String> keys();
    List<String> values();
}

class InMemoryStore implements KeyValueStore {
    private Map<String, String> store = new HashMap<>();

    @Override
    public void put(String key, String value) throws DuplicateKeyException {
        if (store.containsKey(key)) {
            throw new DuplicateKeyException(key);
        }
        store.put(key, value);
    }

    @Override
    public String get(String key) throws KeyNotFoundException {
        if (!store.containsKey(key)) {
            throw new KeyNotFoundException(key);
        }
        return store.get(key);
    }

    @Override
    public void delete(String key) throws KeyNotFoundException {
        if (!store.containsKey(key)) {
            throw new KeyNotFoundException(key);
        }
        store.remove(key);
    }

    @Override
    public boolean exists(String key) {
        return store.containsKey(key);
    }

    @Override
    public List<String> keys() {
        return new ArrayList<>(store.keySet());
    }

    @Override
    public List<String> values() {
        return new ArrayList<>(store.values());
    }

    public void batchPut(String[][] entries) throws StoreException {
        Map<String, String> snapshot = new HashMap<>(store);
        try {
            for (String[] entry : entries) {
                put(entry[0], entry[1]);
            }
        } catch (StoreException e) {
            store = new HashMap<>(snapshot);
            throw e;
        }
    }
}

class LibrarySystem {
    private Map<String, Book> books = new HashMap<>();
    private Map<String, Member> members = new HashMap<>();
    private List<Loan> loans = new ArrayList<>();

    public void addBook(Book book) {
        books.put(book.getIsbn(), book);
    }

    public void addMember(Member member) {
        members.put(member.getId(), member);
    }

    private int currentLoanCount(String isbn) {
        int count = 0;
        for (Loan loan : loans) {
            if (loan.getBook().getIsbn().equals(isbn)
                    && loan.getStatus() != LoanStatus.RETURNED) {
                count++;
            }
        }
        return count;
    }

    public void lend(String isbn, String memberId, LocalDate today) {
        Book book = books.get(isbn);
        Member member = members.get(memberId);
        if (book == null || member == null) {
            System.out.println("[エラー] 書籍または会員が見つかりません");
            return;
        }
        if (currentLoanCount(isbn) >= book.getTotalCopies()) {
            System.out.println("[エラー] 在庫なし: " + book.getTitle());
            return;
        }
        LocalDate dueDate = today.plusDays(14);
        Loan loan = new Loan(book, member, today, dueDate);
        loans.add(loan);
        System.out.printf("貸出: 「%s」→ %s (期限: %s)%n",
            book.getTitle(), member.getName(), dueDate);
    }

    public void returnBook(String isbn, String memberId, LocalDate today) {
        for (Loan loan : loans) {
            if (loan.getBook().getIsbn().equals(isbn)
                    && loan.getMember().getId().equals(memberId)
                    && loan.getStatus() != LoanStatus.RETURNED) {
                loan.returnBook(today);
                System.out.printf("返却: 「%s」← %s%n",
                    loan.getBook().getTitle(), loan.getMember().getName());
                return;
            }
        }
        System.out.println("[エラー] 該当する貸出記録がありません");
    }

    public void checkOverdue(LocalDate today) {
        for (Loan loan : loans) {
            if (loan.getStatus() == LoanStatus.ON_LOAN
                    && loan.getDueDate().isBefore(today)) {
                loan.markOverdue();
                long overdueDays = ChronoUnit.DAYS.between(loan.getDueDate(), today);
                System.out.printf("延滞: 「%s」— %s (%d日超過)%n",
                    loan.getBook().getTitle(),
                    loan.getMember().getName(), overdueDays);
            }
        }
    }

    public void getMemberHistory(String memberId) {
        Member member = members.get(memberId);
        if (member == null) return;
        System.out.printf("--- %sの貸出履歴 ---%n", member.getName());
        for (Loan loan : loans) {
            if (loan.getMember().getId().equals(memberId)) {
                String dateRange = loan.getReturnDate() != null
                    ? loan.getLoanDate() + "〜" + loan.getReturnDate()
                    : loan.getLoanDate() + "〜";
                System.out.printf("「%s」 %s [%s]%n",
                    loan.getBook().getTitle(), dateRange,
                    loan.getStatus().getLabel());
            }
        }
    }

    public void getPopularBooks() {
        Map<String, Integer> countMap = new HashMap<>();
        Map<String, String> titleMap = new HashMap<>();
        for (Loan loan : loans) {
            String isbn = loan.getBook().getIsbn();
            countMap.merge(isbn, 1, Integer::sum);
            titleMap.put(isbn, loan.getBook().getTitle());
        }

        List<Map.Entry<String, Integer>> sorted = new ArrayList<>(countMap.entrySet());
        sorted.sort((a, b) -> b.getValue() - a.getValue());

        int rank = 1;
        int prevCount = -1;
        for (int i = 0; i < sorted.size(); i++) {
            Map.Entry<String, Integer> entry = sorted.get(i);
            if (entry.getValue() != prevCount) {
                rank = i + 1;
            }
            System.out.printf("%d位: %s (%d回)%n",
                rank, titleMap.get(entry.getKey()), entry.getValue());
            prevCount = entry.getValue();
        }
    }

    public void monthlyReport(int year, int month) {
        int lendCount = 0;
        int returnCount = 0;
        int overdueCount = 0;
        Map<String, Boolean> activeMembers = new HashMap<>();

        for (Loan loan : loans) {
            LocalDate ld = loan.getLoanDate();
            if (ld.getYear() == year && ld.getMonthValue() == month) {
                lendCount++;
                activeMembers.put(loan.getMember().getId(), true);
            }
            if (loan.getReturnDate() != null) {
                LocalDate rd = loan.getReturnDate();
                if (rd.getYear() == year && rd.getMonthValue() == month) {
                    returnCount++;
                }
            }
            if (loan.getStatus() == LoanStatus.OVERDUE) {
                overdueCount++;
            }
        }

        System.out.printf("--- %d年%d月 月次統計 ---%n", year, month);
        System.out.printf("貸出件数: %d%n", lendCount);
        System.out.printf("返却件数: %d%n", returnCount);
        System.out.printf("延滞件数: %d%n", overdueCount);
        System.out.printf("アクティブ会員数: %d%n", activeMembers.size());
    }
}

解説
  • Map.merge(key, 1, Integer::sum) は集計の定番パターンです。キーが存在すれば値を加算、なければ初期値1をセットします。
  • ChronoUnit.DAYS.between(date1, date2) で2つの日付間の日数を計算します。Period.between() との違い（Periodは年月日、ChronoUnitは単一単位）を理解しておきましょう。
  • ランキングで同率順位を処理するために、前回のカウントと比較して順位を更新する制御を入れています。
  • LocalDate.plusDays(14) で貸出期限を計算。isBefore() で延滞判定。これらのメソッドはJavaSilver頻出です。
  • HashMap を使ってアクティブ会員の重複を排除しています。HashSet でも同様の結果が得られます。

class ProjectManager {
    private Map<String, Task> tasks = new HashMap<>();
    private Map<String, TaskMember> members = new HashMap<>();
    private List<String> taskOrder = new ArrayList<>();

    public void addTask(Task task) {
        tasks.put(task.getId(), task);

        }
    }

        if (task == null) return false;

            if (dep == null || dep.getStatus() != Status.DONE) {
                return false;
            }
        }
        return true;

            String assigneeName = t.getAssignee() != null ? t.getAssignee().getName() : "未割当";
            String deps = t.getDependsOn().isEmpty() ? "" : " 依存:" + t.getDependsOn();
            System.out.printf("[%s] %-10s %-3s %-6s %02d/%02d-%02d/%02d (%d日) 担当:%s%s%n",
                t.getId(), t.getTitle(),
                t.getPriority().getLabel(), t.getStatus().getLabel(),
                t.getStartDate().getMonthValue(), t.getStartDate().getDayOfMonth(),
                t.getEndDate().getMonthValue(), t.getEndDate().getDayOfMonth(),
                t.getDurationDays(), assigneeName, deps);
        }
    }

    public void printGanttChart() {
        LocalDate projectStart = null;
        LocalDate projectEnd = null;
        for (Task t : tasks.values()) {
            if (projectStart == null || t.getStartDate().isBefore(projectStart)) {
                projectStart = t.getStartDate();
            }
            if (projectEnd == null || t.getEndDate().isAfter(projectEnd)) {
                projectEnd = t.getEndDate();
            }
        }

        int totalDays = (int) ChronoUnit.DAYS.between(projectStart, projectEnd) + 1;

        for (String id : taskOrder) {
            Task t = tasks.get(id);
            int offset = (int) ChronoUnit.DAYS.between(projectStart, t.getStartDate());
            int duration = (int) t.getDurationDays();

            StringBuilder bar = new StringBuilder();
            bar.append("|");
            for (int d = 0; d < totalDays; d++) {
                if (d >= offset && d < offset + duration) {
                    bar.append("#");
                } else {
                    bar.append(".");
                }
            }
            bar.append("|");
            System.out.printf("%s %-10s %s%n", t.getId(), t.getTitle(), bar);
        }
    }

    public void criticalPath() {
        Map<String, Long> memo = new HashMap<>();
        Map<String, List<String>> pathMemo = new HashMap<>();

        for (String id : taskOrder) {
            calcLongestPath(id, memo, pathMemo);
        }

        String longestEnd = null;
        long maxDuration = 0;
        for (String id : taskOrder) {
            if (memo.get(id) > maxDuration) {
                maxDuration = memo.get(id);
                longestEnd = id;
            }
        }

        List<String> path = pathMemo.get(longestEnd);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < path.size(); i++) {
            if (i > 0) sb.append(" → ");
            sb.append(path.get(i));
        }
        System.out.printf("%s (合計: %d日)%n", sb, maxDuration);
    }

    private long calcLongestPath(String taskId, Map<String, Long> memo,
                                  Map<String, List<String>> pathMemo) {
        if (memo.containsKey(taskId)) {
            return memo.get(taskId);
        }

        Task task = tasks.get(taskId);
        long taskDuration = task.getDurationDays();

        if (task.getDependsOn().isEmpty()) {
            memo.put(taskId, taskDuration);
            List<String> path = new ArrayList<>();
            path.add(taskId);
            pathMemo.put(taskId, path);
            return taskDuration;
        }

        long maxDepDuration = 0;
        String maxDepId = null;
        for (String depId : task.getDependsOn()) {
            long depDuration = calcLongestPath(depId, memo, pathMemo);
            if (depDuration > maxDepDuration) {
                maxDepDuration = depDuration;
                maxDepId = depId;
            }
        }

        long total = maxDepDuration + taskDuration;
        memo.put(taskId, total);

        List<String> path = new ArrayList<>(pathMemo.get(maxDepId));
        path.add(taskId);
        pathMemo.put(taskId, path);

        return total;
    }

    public void progressReport() {
        int totalTasks = tasks.size();
        int doneTasks = 0;
        Map<Priority, int[]> priorityStats = new HashMap<>();
        for (Priority p : Priority.values()) {
            priorityStats.put(p, new int[]{0, 0});
        }

        for (Task t : tasks.values()) {
            priorityStats.get(t.getPriority())[0]++;
            if (t.getStatus() == Status.DONE) {
                doneTasks++;
                priorityStats.get(t.getPriority())[1]++;
            }
        }

        System.out.printf("全体進捗: %.1f%% (%d/%d 完了)%n",
            (double) doneTasks / totalTasks * 100, doneTasks, totalTasks);

        for (Priority p : Priority.values()) {
            int[] stats = priorityStats.get(p);
            if (stats[0] == 0) {
                System.out.printf("  %s優先度: — (該当なし)%n", p.getLabel());
            } else {
                System.out.printf("  %s優先度: %.1f%% (%d/%d)%n",
                    p.getLabel(), (double) stats[1] / stats[0] * 100,
                    stats[1], stats[0]);
            }
        }
    }

解説
  • クリティカルパス計算ではメモ化（HashMap）を使った再帰で各タスクの最長経路を求めます。動的計画法（DP）の基本パターンです。
  • List.of("T1", "T2") で不変リストを簡潔に作成。コンストラクタ内で new ArrayList<>() にコピーすることで外部からの変更を防いでいます。
  • int[] 配列を HashMap の値にすることで、1つのキーに対して複数のカウンタ（全体数・完了数）を管理しています。
  • ChronoUnit.DAYS.between() でガントチャートのオフセットと幅を計算。StringBuilder で1文字ずつバーを組み立てます。
  • 依存関係のチェック（canStart）では、全ての依存先タスクが Status.DONE であることを確認します。enumの == 比較はJavaSilver頻出です。
