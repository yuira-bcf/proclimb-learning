function jsonQuery(data, path) {
  const tokens = [];
  const regex = /(\w+)|\[(\d+|\*)\]/g;
  let match;
  while ((match = regex.exec(path)) !== null) {
    if (match[1]) tokens.push({ type: 'key', value: match[1] });
    else if (match[2] === '*') tokens.push({ type: 'wildcard' });
    else tokens.push({ type: 'index', value: parseInt(match[2]) });
  }

  function resolve(current, index) {
    if (index >= tokens.length || current == null) return current;
    const token = tokens[index];

    if (token.type === 'key') {
      return resolve(current[token.value], index + 1);
    }
    if (token.type === 'index') {
      return resolve(current[token.value], index + 1);
    }
    if (token.type === 'wildcard') {
      return current.map(item => resolve(item, index + 1));
    }
  }

  return resolve(data, 0);
}
解説
  • 正規表現でパスをトークンに分割し、再帰関数でトークンを順番に消費しながらデータを走査しています。
  • ワイルドカード [*] は map で各要素に残りのパスを再帰適用し、結果を配列で返します。
  • jmespath、JSONPath、lodash の _.get() の基本原理です。

function createScheduler(concurrency) {
  const queue = [];
  let running = 0;
  let resolveAll = null;
  let totalAdded = 0;
  let totalDone = 0;

  function runNext() {
    while (running < concurrency && queue.length > 0) {
      const task = queue.shift();
      running++;
      task().then(() => {
        running--;
        totalDone++;
        if (totalDone === totalAdded && queue.length === 0 && resolveAll) {
          resolveAll();
        }
        runNext();
      });
    }
  }

  return {
    add(task) {
      totalAdded++;
      queue.push(task);
      runNext();
    },
    waitAll() {
      if (totalDone === totalAdded && queue.length === 0) {
        return Promise.resolve();
      }
      return new Promise(resolve => { resolveAll = resolve; });
    }
  };
}
解説
  • running カウンタで同時実行数を制御し、タスク完了時に次のタスクを自動起動しています。
  • waitAll は全タスク完了時に resolve される Promise を返します。
  • API レート制限、画像の並列ダウンロード、バッチ処理などで使われるパターンです。p-limit ライブラリも同じ原理です。

function createDB() {
  const collections = {};

  return {
    createCollection(name) {
      let docs = [];
      let nextId = 1;

      const collection = {
        insert(doc) {
          const newDoc = { _id: nextId++, ...doc };
          docs.push(newDoc);
          return newDoc;
        },
        find(query) {
          const entries = Object.entries(query);
          return docs.filter(doc =>
            entries.every(([key, val]) => doc[key] === val)
          );
        },
        findAll() {
          return [...docs];
        },
        update(query, updates) {
          const targets = collection.find(query);
          targets.forEach(doc => Object.assign(doc, updates));
          return targets.length;
        },
        remove(query) {
          const before = docs.length;
          const entries = Object.entries(query);
          docs = docs.filter(doc =>
            !entries.every(([key, val]) => doc[key] === val)
          );
          return before - docs.length;
        }
      };

      collections[name] = collection;
      return collection;
    }
  };
}
解説
  • every で全条件一致をチェックするパターンは、MongoDBのクエリ基本動作と同じです。

function createHistory(initialState) {
  const states = [structuredClone(initialState)];
  let index = 0;

  return {
    current() { return structuredClone(states[index]); },
    push(state) {
      states.splice(index + 1);
      states.push(structuredClone(state));
      index = states.length - 1;
    },
    undo() {
      if (index > 0) index--;
      return this.current();
    },
    redo() {
      if (index < states.length - 1) index++;
      return this.current();
    },
    canUndo() { return index > 0; },
    canRedo() { return index < states.length - 1; }
  };
}

function createSearchUI(containerId, products) {
  const container = document.getElementById(containerId);
  const state = { query: '', category: 'all', sortDir: 'asc' };

  // デバウンス
  function debounce(fn, delay) {
    let timer;
    return (...args) => { clearTimeout(timer); timer = setTimeout(() => fn(...args), delay); };
  }

  function getFiltered() {
    let result = products;
    if (state.query) {
      const q = state.query.toLowerCase();
      result = result.filter(p => p.name.toLowerCase().includes(q));
    }
    if (state.category !== 'all') {
      result = result.filter(p => p.category === state.category);
    }
    result = [...result].sort((a, b) =>
      state.sortDir === 'asc' ? a.price - b.price : b.price - a.price
    );
    return result;
  }

  function render() {
    const filtered = getFiltered();
    const categories = [...new Set(products.map(p => p.category))];
    container.innerHTML = '';

    // 検索UI
    const controls = document.createElement('div');
    controls.innerHTML = `
      <input id="search" placeholder="検索..." value="${state.query}">
      <select id="catFilter">
        <option value="all">全カテゴリ</option>
        ${categories.map(c => `<option value="${c}" ${c===state.category?'selected':''}>${c}</option>`).join('')}

function createTodoApp(containerId) {
  const container = document.getElementById(containerId);
  const STORAGE_KEY = 'todoApp';
  let todos = JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
  let filter = 'all';

  function save() {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(todos));
  }

  function emitChange(action) {
    container.dispatchEvent(new CustomEvent('todo:change', {
      detail: { action, todos: [...todos] },
      bubbles: true
    }));
  }

  function getFiltered() {
    if (filter === 'active') return todos.filter(t => !t.done);
    if (filter === 'completed') return todos.filter(t => t.done);
    return todos;
  }

  function render() {
    const filtered = getFiltered();
    const remaining = todos.filter(t => !t.done).length;

    container.innerHTML = `
      <div class="todo-input">
        <input id="newTodo" placeholder="新しいTodo..." />
        <button data-action="add">追加</button>
      </div>
      <div class="todo-filters">
        <button data-filter="all" class="${filter === 'all' ? 'active' : ''}">全て</button>
        <button data-filter="active" class="${filter === 'active' ? 'active' : ''}">未完了</button>
        <button data-filter="completed" class="${filter === 'completed' ? 'active' : ''}">完了済</button>
      </div>
      <ul class="todo-list">
        ${filtered.map(t => `
          <li data-id="${t.id}">
            <input type="checkbox" data-action="toggle" ${t.done ? 'checked' : ''} />
            <span style="${t.done ? 'text-decoration:line-through;color:#999' : ''}">${t.text}</span>
            <button data-action="delete">削除</button>
          </li>
        `).join('')}
      </ul>
      <p>残り ${remaining} 件</p>
    `;
  }

  // イベント委譲
  container.addEventListener('click', (e) => {
    const action = e.target.dataset.action;
    const filterVal = e.target.dataset.filter;

    if (action === 'add') {
      const input = container.querySelector('#newTodo');
      const text = input.value.trim();
      if (!text) return;
      todos.push({ id: Date.now(), text, done: false });
      save(); render(); emitChange('add');
    }

    if (action === 'toggle') {
      const id = Number(e.target.closest('li').dataset.id);
      const todo = todos.find(t => t.id === id);
      if (todo) { todo.done = !todo.done; save(); render(); emitChange('toggle'); }
    }

    if (action === 'delete') {
      const id = Number(e.target.closest('li').dataset.id);
      todos = todos.filter(t => t.id !== id);
      save(); render(); emitChange('delete');
    }

    if (filterVal) {
      filter = filterVal;
      render();
    }
  });

  // Enterキーで追加
  container.addEventListener('keydown', (e) => {
    if (e.key === 'Enter' && e.target.id === 'newTodo') {
      const text = e.target.value.trim();
      if (!text) return;
      todos.push({ id: Date.now(), text, done: false });
      save(); render(); emitChange('add');
    }
  });

  render();
}

function createValidatedForm(containerId, schema) {
  const container = document.getElementById(containerId);
  const errors = {};
  const values = {};

  // 初期化
  for (const field of Object.keys(schema)) {
    values[field] = '';
    errors[field] = null;
  }

  function validate(field) {
    const rules = schema[field].rules;
    for (const rule of rules) {
      const error = rule(values[field], values);
      if (error) { errors[field] = error; return false; }
    }
    errors[field] = null;
    return true;
  }

  function isAllValid() {
    return Object.keys(schema).every(f => {
      const rules = schema[f].rules;
      return rules.every(rule => rule(values[f], values) === null);
    });
  }

  function render() {
    container.innerHTML = `
      <form id="validatedForm">
        ${Object.entries(schema).map(([field, config]) => `
          <div class="form-field">
            <label>${config.label}</label>
            <input name="${field}" type="${field.includes('password') || field === 'confirm' ? 'password' : 'text'}"
              value="${values[field]}" />
            <div class="field-status">
              ${errors[field] ? `<span class="error">${errors[field]}</span>`
                : values[field] && !errors[field] ? '<span class="success">OK</span>' : ''}
            </div>
          </div>
        `).join('')}
        <button type="submit" ${isAllValid() && Object.values(values).some(v => v) ? '' : 'disabled'}>送信</button>
      </form>
    `;
  }

  // イベント委譲
  container.addEventListener('blur', (e) => {
    if (e.target.name && e.target.name in schema) {
      values[e.target.name] = e.target.value;
      validate(e.target.name);
      render();
    }
  }, true); // capture フェーズで blur を検知

  container.addEventListener('input', (e) => {
    if (e.target.name && e.target.name in schema) {
      values[e.target.name] = e.target.value;
      errors[e.target.name] = null;
      render();
      // カーソル位置を復元
      const input = container.querySelector(`[name="${e.target.name}"]`);
      if (input) { input.focus(); input.selectionStart = input.selectionEnd = e.target.selectionStart; }

      container.dispatchEvent(new CustomEvent('form:submit', {
        detail: { ...values }, bubbles: true
      }));
    }
    render();
  });

  render();
}

    }

    // 重複排除（GETのみ）
    if (method === 'GET' && pending.has(url)) {
      return pending.get(url);
    }

    // リクエスト設定
    let config = {
      method, headers: { 'Content-Type': 'application/json' },
      body: body ? JSON.stringify(body) : undefined
    };

    // リクエストインターセプター
    for (const fn of interceptors.request.fns) {
      config = fn(config);
    }

    // リトライ付きfetch
    const promise = (async () => {
      let lastError;
      for (let attempt = 0; attempt <= maxRetries; attempt++) {
        try {
          const res = await fetch(url, config);
          if (!res.ok && attempt < maxRetries) throw new Error(`HTTP ${res.status}`);

          let response = { status: res.status, url, data: await res.json() };
          for (const fn of interceptors.response.fns) {
            response = fn(response);
          }

          // キャッシュ保存（GETのみ）
          if (method === 'GET' && cacheTTL > 0) {
            cache.set(url, { data: response.data, expiry: Date.now() + cacheTTL });
          }

          return response.data;
        } catch (error) {
          lastError = error;
          if (attempt < maxRetries) {
            await new Promise(r => setTimeout(r, baseDelay * Math.pow(2, attempt)));
          }
        }
      }
      throw lastError;
    })();

    if (method === 'GET') {
      pending.set(url, promise);
      promise.finally(() => pending.delete(url));
    }

    return promise;
  }

  return {
    interceptors,
    get: (path) => request('GET', path),
    post: (path, body) => request('POST', path, body),
    put: (path, body) => request('PUT', path, body),
    delete: (path) => request('DELETE', path)
  };
}

    filter(predicate) {
      const source = asyncIterable;
      return pipeline(async function* () {
        for await (const item of source) {
          if (predicate(item)) yield item;
        }
      }());
    },

      const source = asyncIterable;
      return pipeline(async function* () {
        for await (const item of source) {
          yield transform(item);
        }
      }());
    },

    take(n) {
      const source = asyncIterable;
      return pipeline(async function* () {
        let count = 0;
        for await (const item of source) {
          yield item;
          if (++count >= n) return;
        }
      }());

      const result = [];
      for await (const item of asyncIterable) {
        result.push(item);
      }
      return result;
    }
  };

function createGlobalErrorHandler({ onReport, onUserNotify, debounceMs = 5000 } = {}) {
  const reportQueue = [];
  const seen = new Set();
  const stats = { total: 0, byType: {}, suppressed: 0 };
  let timer = null;

  function classifyError(error) {
    if (error instanceof TypeError && error.message.includes('fetch')) return 'network';
    if (error.name === 'ValidationError') return 'validation';
    if (error instanceof SyntaxError || error instanceof ReferenceError) return 'runtime';
    return 'unknown';
  }

  function getFingerprint(error) {
    const stack = error.stack?.split('\n')[1]?.trim() || '';
    return `${error.message}|${stack}`;
  }

  function getUserMessage(type) {
    const messages = {
      network: 'ネットワークエラーが発生しました',
      validation: '入力内容に問題があります',
      runtime: '予期しないエラーが発生しました',
      unknown: 'エラーが発生しました'
    };
    return messages[type] || messages.unknown;
  }

  function scheduleFlush() {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      if (reportQueue.length > 0) {
        onReport?.([...reportQueue]);
        reportQueue.length = 0;
      }
      timer = null;
    }, debounceMs);
  }

  function handleError(error) {
    const fingerprint = getFingerprint(error);
    if (seen.has(fingerprint)) {
      stats.suppressed++;
      return;
    }
    seen.add(fingerprint);

    const type = classifyError(error);
    stats.total++;
    stats.byType[type] = (stats.byType[type] || 0) + 1;

    reportQueue.push({
      type, message: error.message, stack: error.stack,
      timestamp: new Date().toISOString(), fingerprint
    });

    onUserNotify?.(getUserMessage(type));
    scheduleFlush();
  }

  window.addEventListener('error', (e) => handleError(e.error || new Error(e.message)));
  window.addEventListener('unhandledrejection', (e) => {
    const error = e.reason instanceof Error ? e.reason : new Error(String(e.reason));
    handleError(error);
  });

  return {
    captureError: handleError,
    getStats: () => ({ ...stats, byType: { ...stats.byType } }),
    destroy() {
      clearTimeout(timer);
      // リスナー削除は省略（実装時は参照を保持して削除）
    }
  };
}

function createApp(selector) {
  const container = document.querySelector(selector);
  const routes = [];
  const middlewares = [];
  let currentComponent = null;
  let currentState = {};

  function state(initial) {
    const reactive = { value: initial };
    return new Proxy(reactive, {
      set(target, key, value) {
        target[key] = value;
        if (key === 'value') renderCurrent();
        return true;
      }
    });
  }

  function renderCurrent() {
    if (!currentComponent?.render) return;
    const html = currentComponent.render(currentState);
    container.innerHTML = html;

    // methods をグローバルに公開（簡易実装）
    if (currentComponent.methods) {
      for (const [name, fn] of Object.entries(currentComponent.methods)) {
        window[name] = () => { fn(currentState); };
      }
    }
  }

  function matchRoute(path) {
    for (const route of routes) {
      if (route.path === path) return route;
    }
    return null;
  }

  function navigate(path) {
    const ctx = { path };

    // ミドルウェア実行
    let idx = 0;
    function next() {
      if (idx < middlewares.length) {
        middlewares[idx++](ctx, next);
      } else {
        // ルートマッチング
        const route = matchRoute(path);
        if (!route) return;
        currentComponent = route.component;
        currentState = currentComponent.setup ? currentComponent.setup() : {};
        renderCurrent();
      }
    }
    next();
  }

  // リンクのクリックをインターセプト
  document.addEventListener('click', (e) => {
    const a = e.target.closest('a[href]');
    if (a && a.getAttribute('href').startsWith('/')) {
      e.preventDefault();
      history.pushState(null, '', a.getAttribute('href'));
      navigate(a.getAttribute('href'));
    }
  });

  window.addEventListener('popstate', () => navigate(location.pathname));

  return {
    state,
    route(path, component) {
      routes.push({ path, component });
    },
    use(middleware) {
      middlewares.push(middleware);
    },
    start() {
      navigate(location.pathname);
    }
  };
}

function createCLI(name, version) {
  const commands = new Map();
  const hooks = { beforeCommand: [], afterCommand: [] };

  function createCommand(cmdName) {
    const cmd = {
      name: cmdName,
      desc: '',
      options: [],
      _action: null,
      description(text) { cmd.desc = text; return cmd; },
      option(flags, desc, defaultValue) {
        const parts = flags.split(/[,\s]+/);
        const longFlag = parts.find(p => p.startsWith('--'))?.replace('--', '');
        const hasValue = flags.includes('<');
        cmd.options.push({ flags, desc, longFlag, hasValue, defaultValue });
        return cmd;
      },
      action(fn) { cmd._action = fn; return cmd; }
    };
    commands.set(cmdName, cmd);
    return cmd;
  }

  function parseOptions(args, optionDefs) {
    const result = {};
    optionDefs.forEach(opt => { result[opt.longFlag] = opt.defaultValue; });

    for (let i = 0; i < args.length; i++) {
      const arg = args[i];
      const optDef = optionDefs.find(o =>
        o.flags.split(/[,\s]+/).some(f => f.replace(/\s*<.*>/, '') === arg)
      );
      if (optDef) {
        result[optDef.longFlag] = optDef.hasValue ? args[++i] : true;
      }
    }
    return result;
  }

  function showHelp() {
    console.log(`${name} v${version}`);
    console.log('Commands:');
    for (const [cmdName, cmd] of commands) {
      console.log(`  ${cmdName.padEnd(10)} — ${cmd.desc}`);
    }
  }

  return {
    command: createCommand,

    use(plugin) {
      plugin.install({
        hook(event, fn) {
          if (hooks[event]) hooks[event].push(fn);
        }
      });
    },

    run(argv) {
      if (argv.includes('--help') || argv.includes('-h')) {
        showHelp();
        return;
      }

      const cmdName = argv[0];
      const cmd = commands.get(cmdName);
      if (!cmd) {
        console.log(`Unknown command: ${cmdName}`);
        showHelp();
        return;
      }

      const options = parseOptions(argv.slice(1), cmd.options);

      hooks.beforeCommand.forEach(fn => fn(cmdName, options));
      cmd._action(options);
      hooks.afterCommand.forEach(fn => fn(cmdName, options));
    }
  };
}
