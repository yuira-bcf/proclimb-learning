<select id="countDailyCompletedInSprint" resultType="map">
    SELECT DATE(updated_at) AS completed_date,
           COUNT(*) AS cnt
    FROM todo
    WHERE sprint_id = #{sprintId}
      AND status = 'COMPLETED'

      AND updated_at <= #{endDate}

BurndownService.java
import org.springframework.stereotype.Service;
import java.time.LocalDate;
import java.util.*;

@Service
public class BurndownService {

    private final SprintMapper sprintMapper;

    public BurndownService(SprintMapper sprintMapper) {
        this.sprintMapper = sprintMapper;
    }

    /**
     * バーンダウンチャート用データを構築
     */
    public Map<String, Object> getBurndownData(Integer sprintId) {
        Sprint sprint = sprintMapper.findById(sprintId);
        int totalTasks = sprintMapper.countTasksInSprint(sprintId);

        // 日別完了数を取得
        List<Map<String, Object>> dailyData =
            sprintMapper.countDailyCompletedInSprint(
                sprintId, sprint.getStartDate(), sprint.getEndDate()
            );
        Map<String, Integer> dailyCompleted = new HashMap<>();
        for (Map<String, Object> row : dailyData) {
            dailyCompleted.put(
                row.get("completed_date").toString(),
                ((Number) row.get("cnt")).intValue()
            );
        }

        // 日付ラベルと残タスク数を計算
        List<String> labels = new ArrayList<>();
        List<Integer> actualData = new ArrayList<>();
        List<Double> idealData = new ArrayList<>();

        long days = sprint.getDurationDays();
        double idealDecrement = (double) totalTasks / (days - 1);
        int cumulativeCompleted = 0;

        LocalDate current = sprint.getStartDate();
        LocalDate end = sprint.getEndDate();
        int dayIndex = 0;

        while (!current.isAfter(end)) {
            labels.add(current.toString());

            // 実績線: 残タスク数 = 総数 - 累積完了数
            cumulativeCompleted += dailyCompleted
                .getOrDefault(current.toString(), 0);
            actualData.add(totalTasks - cumulativeCompleted);

            // 理想線: 均等に減少
            idealData.add(
                Math.max(0, totalTasks - idealDecrement * dayIndex));

            current = current.plusDays(1);
            dayIndex++;
        }

        return Map.of(
            "labels", labels,
            "actual", actualData,
            "ideal", idealData,
            "sprint", sprint,
            "totalTasks", totalTasks
        );
    }
}
warning理想線の計算

理想線（Ideal Line）はスプリント開始時の総タスク数を日数で均等割りした直線です。実績線がこれより上にある場合は「遅れている」、下にある場合は「進んでいる」ことを意味します。

BurndownController.java
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
public class BurndownController {

    private final BurndownService burndownService;
    private final SprintMapper sprintMapper;

    public BurndownController(BurndownService burndownService,
                               SprintMapper sprintMapper) {
        this.burndownService = burndownService;
        this.sprintMapper = sprintMapper;
    }

    @GetMapping("/burndown")
    public String burndown(
            @RequestParam(required = false) Integer sprintId,
            @AuthenticationPrincipal LoginUser loginUser,
            Model model) {

        Integer userId = loginUser.getUserId();
        List<Sprint> sprints = sprintMapper.findByUserId(userId);
        model.addAttribute("sprints", sprints);

        if (sprintId != null) {
            Map<String, Object> data =
                burndownService.getBurndownData(sprintId);
            model.addAttribute("data", data);
            model.addAttribute("selectedSprintId", sprintId);

            // 進捗ステータス判定
            List<Integer> actual =
                (List<Integer>) data.get("actual");
            List<Double> ideal =
                (List<Double>) data.get("ideal");
            int lastIdx = Math.min(
                actual.size(), ideal.size()) - 1;
            if (lastIdx >= 0) {
                double diff = actual.get(lastIdx)
                    - ideal.get(lastIdx);
                String status = diff < -1 ? "ahead"
                    : diff > 1 ? "behind" : "on_track";
                model.addAttribute("progressStatus", status);
            }
        }

        return "burndown";
    }
}

# Step 1: スプリントテーブル設計
sprintテーブル（name, start_date, end_date, initial_task_count, user_id）を作成し、todoテーブルにsprint_id列を追加してください。
