CREATE TABLE label_colors (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    label_type VARCHAR(20) NOT NULL,       -- 'CATEGORY' or 'PRIORITY'
    label_value VARCHAR(50) NOT NULL,       -- カテゴリ名 or 優先度名
    color_code VARCHAR(7) NOT NULL,         -- '#FF5733' 形式
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    UNIQUE KEY uk_user_label (user_id, label_type, label_value)
);

Section 63-2

LabelColorController.java 
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
@RequestMapping("/settings/colors")
public class LabelColorController {

    private final LabelColorService labelColorService;

    public LabelColorController(LabelColorService labelColorService) {
        this.labelColorService = labelColorService;
    }

    /**
     * カラー設定画面を表示
     */
    @GetMapping
    public String showColorSettings(@AuthenticationPrincipal UserDetails user,
                                    Model model) {
        Integer userId = ((CustomUserDetails) user).getUserId();
        model.addAttribute("colors", labelColorService.getUserColors(userId));
        model.addAttribute("defaults", labelColorService.getDefaultColors());
        return "settings/color-settings";
    }

    /**
     * カラー設定をAjaxで保存
     */
    @PostMapping("/api/save")
    @ResponseBody
    public Map<String, String> saveColor(
            @AuthenticationPrincipal UserDetails user,
            @RequestBody LabelColorForm form) {

        Integer userId = ((CustomUserDetails) user).getUserId();

        // カラーコードのバリデーション
        if (!form.getColorCode().matches("^#[0-9A-Fa-f]{6}$")) {
            return Map.of("status", "error", "message", "不正なカラーコードです");
        }

        labelColorService.saveColor(userId, form);
        return Map.of("status", "ok");
    }
}

LabelColorService.java 
import org.springframework.stereotype.Service;
import java.util.*;

@Service
public class LabelColorService {

    private final LabelColorMapper labelColorMapper;

    /** システムデフォルト色 */
    private static final Map<String, String> DEFAULT_COLORS = Map.of(
        "仕事",     "#3b82f6",
        "プライベート", "#22c55e",
        "勉強",     "#f59e0b",
        "健康",     "#ef4444",
        "その他",    "#8b5cf6"
    );

    public LabelColorService(LabelColorMapper labelColorMapper) {
        this.labelColorMapper = labelColorMapper;
    }

    /**
     * ユーザーのカラー設定をMap形式で取得
     */
    public Map<String, String> getUserColors(Integer userId) {
        Map<String, String> colorMap = new LinkedHashMap<>(DEFAULT_COLORS);
        List<LabelColor> customs = labelColorMapper.findByUserId(userId);
        for (LabelColor lc : customs) {
            colorMap.put(lc.getLabelValue(), lc.getColorCode());
        }
        return colorMap;
    }

    /**
     * CSS変数用の文字列を生成
     */
    public String generateCssVariables(Integer userId) {
        Map<String, String> colors = getUserColors(userId);
        StringBuilder sb = new StringBuilder(":root {\n");
        for (Map.Entry<String, String> entry : colors.entrySet()) {
            String varName = "--label-" + entry.getKey().hashCode();
            sb.append("  ").append(varName).append(": ")
              .append(entry.getValue()).append(";\n");
        }
        sb.append("}");
        return sb.toString();
    }

    public Map<String, String> getDefaultColors() {
        return DEFAULT_COLORS;
    }

    public void saveColor(Integer userId, LabelColorForm form) {
        LabelColor lc = new LabelColor();
        lc.setUserId(userId);
        lc.setLabelType(form.getLabelType());
        lc.setLabelValue(form.getLabelValue());
        lc.setColorCode(form.getColorCode());
        labelColorMapper.upsert(lc);
    }
}

/**
 * カラーピッカー変更時にCSS変数を即座に更新
 */
function applyColorToPage(labelName, colorCode) {
    const varName = '--label-' + labelName.hashCode();
    document.documentElement.style.setProperty(varName, colorCode);
}

// 簡易hashCode（Java互換）
String.prototype.hashCode = function() {
    let hash = 0;
    for (let i = 0; i < this.length; i++) {
        const char = this.charCodeAt(i);
        hash = ((hash << 5) - hash) + char;
        hash |= 0;
    }
    return hash;
};
lightbulbCSS変数の利点

CSS Custom Properties（var(--xxx)）を使えば、JavaScriptからstyle.setProperty一つで全ページのラベル色を一括更新できます。個別のDOM操作は不要です。

LabelColorController.java（リセット処理） 
/**
 * 個別ラベル色のリセット
 */
@PostMapping("/api/reset")
@ResponseBody
public Map<String, String> resetColor(
        @AuthenticationPrincipal UserDetails user,
        @RequestBody LabelColorForm form) {

    Integer userId = ((CustomUserDetails) user).getUserId();
    labelColorService.resetColor(userId, form.getLabelType(),
                                  form.getLabelValue());

    String defaultColor = labelColorService
        .getDefaultColors().get(form.getLabelValue());
    return Map.of("status", "ok", "defaultColor", defaultColor);
}

/**
 * 全ラベル色の一括リセット
 */
@PostMapping("/api/reset-all")
@ResponseBody
public Map<String, Object> resetAllColors(
        @AuthenticationPrincipal UserDetails user) {

    Integer userId = ((CustomUserDetails) user).getUserId();
    labelColorService.resetAllColors(userId);

    return Map.of("status", "ok",
                   "defaults", labelColorService.getDefaultColors());
}
JavaScript（リセットUI） 
/**
 * 個別リセット
 */
function resetColor(labelType, labelValue) {
    if (!confirm(labelValue + ' の色をデフォルトに戻しますか？')) return;

    fetch('/settings/colors/api/reset', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ labelType, labelValue })
    })
    .then(res => res.json())
    .then(data => {
        // ピッカーとプレビューをデフォルト色に更新
        document.getElementById('picker-' + labelValue).value
            = data.defaultColor;
        document.getElementById('preview-' + labelValue)
            .style.background = data.defaultColor;
    });
}

/**
 * 全リセット
 */
function resetAllColors() {
    if (!confirm('すべてのラベル色をデフォルトに戻しますか？')) return;

    fetch('/settings/colors/api/reset-all', { method: 'POST' })
    .then(res => res.json())
    .then(data => {
        Object.entries(data.defaults).forEach(([label, color]) => {
            document.getElementById('picker-' + label).value = color;
            document.getElementById('preview-' + label)
                .style.background = color;
        });
    });
}
lightbulbリセットの実装方式

リセットはlabel_colorsテーブルから該当レコードを削除するだけです。レコードがなければサービス層がデフォルト色を返すため、追加の処理は不要です。

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにラベル色カスタマイズ機能を追加してください。

【要件】
- label_colorsテーブルでユーザーごとのカラー設定を保存
- HTML5 input[type=color]でカラーピッカーを表示
- CSS変数で全ページのラベル色を動的に変更
- デフォルト色へのリセット機能（個別・一括）

【技術スタック】
- Spring Boot 3.x + MyBatis
- Thymeleaf（CSS変数のインライン注入）
- JavaScript fetch API（非同期保存）
- CSS Custom Properties（var(--xxx)）
gps_fixed ステップ別プロンプト 
# Step 1: カラー設定テーブル
label_colorsテーブルを作成してください。user_id, label_type, label_value, color_code列を持ち、UNIQUE制約で重複を防いでください。

# Step 2: カラーピッカーUI
/settings/colorsでラベル色設定画面を作成してください。input[type=color]で色を選択し、fetchで非同期保存してください。

# Step 3: CSS変数の動的注入
全ページのheadにユーザーのカラー設定をCSS変数として注入するThymeleafフラグメントを作成してください。

# Step 4: デフォルト色リセット
個別リセットと一括リセットのAPIを作成してください。リセット後はデフォルト色をレスポンスに含めてUIを更新してください。
lightbulb プロンプトのコツ

「カラーコードは必ず#RRGGBB形式でバリデーションしてください」と指定すると、不正な値の保存を防げます。
また、「プレビューは色変更と同時にリアルタイムで更新してください」と伝えるとUXが向上します。
