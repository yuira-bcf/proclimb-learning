Section 35-1

KeyboardEventリスナー設計

ToDoアプリにキーボードショートカットを追加するために、KeyboardEventをグローバルに監視する仕組みを設計します。keydownイベントで修飾キー（Ctrl/Meta）との組み合わせを判定します。

keyboard

keydownイベント

document全体でキー入力を監視するイベントリスナー

tune

修飾キー判定

ctrlKey / metaKey / shiftKey / altKeyの組み合わせ検出

block

preventDefault

ブラウザデフォルト動作を無効化して独自動作を割り当て

shortcut.js（グローバルキーリスナー） 
/**
 * ショートカットキー管理クラス
 * グローバルなKeyboardEventを一元管理する
 */
class ShortcutManager {

    constructor() {
        this.shortcuts = new Map();
        this.enabled = true;
        this._init();
    }

    _init() {
        document.addEventListener('keydown', (e) => {
            // 入力フォーカス中はショートカット無効
            const tag = e.target.tagName;
            if (tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT') {
                // Escapeだけは常に有効（フォーカス解除用）
                if (e.key !== 'Escape') return;
            }
            if (!this.enabled) return;
            this._handleKeydown(e);
        });
    }

    _handleKeydown(e) {
        const key = this._buildKey(e);
        const action = this.shortcuts.get(key);
        if (action) {
            e.preventDefault();
            action.handler();
        }
    }

    /**
     * 修飾キー + キー名 を文字列化
     * 例: "Ctrl+N", "Ctrl+Shift+F"
     */
    _buildKey(e) {
        const parts = [];
        if (e.ctrlKey || e.metaKey) parts.push('Ctrl');
        if (e.shiftKey) parts.push('Shift');
        if (e.altKey) parts.push('Alt');
        parts.push(e.key.toUpperCase());
        return parts.join('+');
    }

    /**
     * ショートカットを登録
     * @param {string} combo  例: "Ctrl+N"
     * @param {string} desc   説明文
     * @param {Function} handler 実行する関数
     */
    register(combo, desc, handler) {
        this.shortcuts.set(combo, { desc, handler });
    }
}

// シングルトンとしてエクスポート
const shortcutManager = new ShortcutManager();

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.Map;

@RestController
public class ShortcutController {

    /**
     * ショートカット一覧をJSONで返す（ヘルプモーダル用）
     */
    @GetMapping("/api/shortcuts")
    public List<Map<String, String>> getShortcuts() {
        return List.of(
            Map.of("key", "Ctrl + N", "desc", "新しいタスクを作成"),
            Map.of("key", "Ctrl + S", "desc", "フォームを保存"),
            Map.of("key", "Ctrl + F", "desc", "検索にフォーカス"),
            Map.of("key", "Ctrl + H", "desc", "ホームへ移動"),
            Map.of("key", "Ctrl + Enter", "desc", "完了/未完了切替"),
            Map.of("key", "Delete", "desc", "選択タスクを削除"),
            Map.of("key", "?", "desc", "ヘルプ表示")
        );
    }
}

keyboard-nav.js（リストナビゲーション） 
/**
 * タスクリストのキーボードナビゲーション
 */
class TaskKeyboardNav {

    constructor(listSelector = '.task-list') {
        this.list = document.querySelector(listSelector);
        if (!this.list) return;

        this.rows = Array.from(this.list.querySelectorAll('.task-row'));
        this.currentIndex = -1;
        this._bindKeys();
    }

    _bindKeys() {
        // 上下矢印でフォーカス移動
        shortcutManager.register('ARROWDOWN', '次のタスクへ', () => {
            this._move(1);
        });
        shortcutManager.register('ARROWUP', '前のタスクへ', () => {
            this._move(-1);
        });

        // Enterで詳細ページへ
        shortcutManager.register('ENTER', 'タスク詳細を開く', () => {
            if (this.currentIndex >= 0) {
                const link = this.rows[this.currentIndex]
                    .querySelector('a.task-link');
                if (link) link.click();
            }
        });

        // Escapeでフォーカス解除
        shortcutManager.register('ESCAPE', 'フォーカス解除', () => {
            this._clearFocus();
            document.activeElement.blur();
        });

        // スペースで選択トグル
        shortcutManager.register(' ', 'チェックボックス切替', () => {
            if (this.currentIndex >= 0) {
                const cb = this.rows[this.currentIndex]
                    .querySelector('input[type="checkbox"]');
                if (cb) cb.checked = !cb.checked;
            }
        });
    }

    _move(direction) {
        this._clearFocus();
        this.currentIndex += direction;
        // 範囲を制限
        if (this.currentIndex < 0) this.currentIndex = 0;
        if (this.currentIndex >= this.rows.length) {
            this.currentIndex = this.rows.length - 1;
        }
        this.rows[this.currentIndex].classList.add('focused');
        this.rows[this.currentIndex].scrollIntoView({
            block: 'nearest'
        });
    }

    _clearFocus() {
        this.rows.forEach(r => r.classList.remove('focused'));
    }
}

// ページ読み込み時に初期化
document.addEventListener('DOMContentLoaded', () => {
    new TaskKeyboardNav();
});

shortcut-help.js（モーダル表示ロジック） 
/**
 * ショートカットヘルプモーダルの表示/非表示を切り替え
 */
function toggleShortcutHelp() {
    const modal = document.getElementById('shortcut-help-modal');
    if (modal.style.display === 'none') {
        loadShortcutList();
        modal.style.display = 'flex';
    } else {
        modal.style.display = 'none';
    }
}

/**
 * サーバーAPIからショートカット一覧を取得して表示
 */
async function loadShortcutList() {
    const tbody = document.querySelector('#shortcut-list tbody');
    tbody.innerHTML = '';

    try {
        const res = await fetch('/api/shortcuts');
        const shortcuts = await res.json();

        shortcuts.forEach(s => {
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td><kbd>${s.key}</kbd></td>
                <td>${s.desc}</td>
            `;
            tbody.appendChild(tr);
        });
    } catch (err) {
        // APIエラー時はクライアント側の定義から表示
        shortcutManager.shortcuts.forEach((val, key) => {
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td><kbd>${key}</kbd></td>
                <td>${val.desc}</td>
            `;
            tbody.appendChild(tr);
        });
    }
}

// オーバーレイクリックで閉じる
document.addEventListener('click', (e) => {
    const modal = document.getElementById('shortcut-help-modal');
    if (e.target === modal) {
        modal.style.display = 'none';
    }
});

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにキーボードショートカット機能を追加してください。

【要件】
- グローバルKeyboardEventリスナーでショートカットを一元管理
- Ctrl+N: 新規タスク作成、Ctrl+S: 保存、Ctrl+F: 検索フォーカス
- 矢印キーでタスクリストのフォーカス移動、Enterで詳細表示
- ?キーでショートカット一覧のヘルプモーダルを表示
- 入力フォーム内ではショートカットを無効化（Escape除く）

【技術スタック】
- JavaScript（vanilla、フレームワーク不使用）
- Spring Boot 3.x + Thymeleaf
- REST API（ショートカット一覧取得用）

# Step 1: ShortcutManagerクラスの作成
JavaScriptでショートカットを一元管理するShortcutManagerクラスを作成してください。keydownイベントリスナー、修飾キー判定、ショートカット登録機能を含めてください。

# Step 2: ショートカット定義の登録
Ctrl+N（新規作成）、Ctrl+S（保存）、Ctrl+F（検索）、Ctrl+H（ホーム）、?（ヘルプ）などの操作を登録してください。

# Step 3: キーボードナビゲーション
矢印キーでタスク行を上下移動し、Enterで詳細ページを開き、Spaceでチェックボックスを切替、Escapeでフォーカス解除する機能を追加してください。

# Step 4: ヘルプモーダル
?キーで表示するショートカット一覧モーダルを実装してください。/api/shortcutsからJSONで取得し、テーブル表示してください。
