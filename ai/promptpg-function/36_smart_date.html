DatePattern.java（パターン定義列挙型） 
import java.util.regex.Pattern;

/**
 * 自然言語日付パターンの定義
 */
public enum DatePattern {

    TODAY("^(今日|きょう|today)$"),
    TOMORROW("^(明日|あした|あす|tomorrow)$"),
    DAY_AFTER_TOMORROW("^(明後日|あさって)$"),
    DAYS_LATER("^(\\d+)日後$"),
    WEEKS_LATER("^(\\d+)週間後$"),
    NEXT_WEEKDAY("^(来週|次の?)(月|火|水|木|金|土|日)(曜日?)?$"),
    THIS_WEEKDAY("^(今週の?)(月|火|水|木|金|土|日)(曜日?)?$"),
    MONTH_DAY("^(\\d{1,2})月(\\d{1,2})日$"),
    SLASH_DATE("^(\\d{1,2})/(\\d{1,2})$"),
    ISO_DATE("^\\d{4}-\\d{2}-\\d{2}$");

    private final Pattern pattern;

    DatePattern(String regex) {
        this.pattern = Pattern.compile(regex);
    }

    public Pattern getPattern() {
        return pattern;
    }
}
WeekdayMapper.java（曜日文字列→DayOfWeek変換） 
import java.time.DayOfWeek;
import java.util.Map;

public class WeekdayMapper {

    private static final Map<String, DayOfWeek> WEEKDAY_MAP = Map.of(
        "月", DayOfWeek.MONDAY,
        "火", DayOfWeek.TUESDAY,
        "水", DayOfWeek.WEDNESDAY,
        "木", DayOfWeek.THURSDAY,
        "金", DayOfWeek.FRIDAY,
        "土", DayOfWeek.SATURDAY,
        "日", DayOfWeek.SUNDAY
    );

    public static DayOfWeek fromJapanese(String dayStr) {
        DayOfWeek dow = WEEKDAY_MAP.get(dayStr);
        if (dow == null) {
            throw new IllegalArgumentException(
                "不明な曜日: " + dayStr);
        }
        return dow;
    }
}
lightbulbパターンの優先順位

列挙型の定義順がマッチングの優先順位になります。「今日」のような完全一致パターンを先に、「数字+日後」のような部分マッチパターンを後に配置することで、誤判定を防ぎます。

SmartDateParser.java（パーサー本体） 
import org.springframework.stereotype.Service;
import java.time.LocalDate;
import java.time.temporal.TemporalAdjusters;
import java.util.Optional;
import java.util.regex.Matcher;

@Service
public class SmartDateParser {

    /**
     * 自然言語テキストをLocalDateに変換
     * @param input ユーザー入力文字列
     * @return パース結果（失敗時はOptional.empty）
     */
    public Optional<LocalDate> parse(String input) {
        if (input == null || input.isBlank()) {
            return Optional.empty();
        }
        String text = input.trim();
        LocalDate today = LocalDate.now();

        // --- 今日 ---
        Matcher m = DatePattern.TODAY.getPattern().matcher(text);
        if (m.matches()) return Optional.of(today);

        // --- 明日 ---
        m = DatePattern.TOMORROW.getPattern().matcher(text);
        if (m.matches()) return Optional.of(today.plusDays(1));

        // --- 明後日 ---
        m = DatePattern.DAY_AFTER_TOMORROW.getPattern().matcher(text);
        if (m.matches()) return Optional.of(today.plusDays(2));

        // --- N日後 ---
        m = DatePattern.DAYS_LATER.getPattern().matcher(text);
        if (m.matches()) {
            int days = Integer.parseInt(m.group(1));
            return Optional.of(today.plusDays(days));
        }

        // --- N週間後 ---
        m = DatePattern.WEEKS_LATER.getPattern().matcher(text);
        if (m.matches()) {
            int weeks = Integer.parseInt(m.group(1));
            return Optional.of(today.plusWeeks(weeks));
        }

        // --- 来週○曜日 ---
        m = DatePattern.NEXT_WEEKDAY.getPattern().matcher(text);
        if (m.matches()) {
            var dow = WeekdayMapper.fromJapanese(m.group(2));
            return Optional.of(
                today.with(TemporalAdjusters.next(dow))
                     .plusWeeks(0)  // 「来週」は次の該当曜日
            );
        }

        // --- 今週○曜日 ---
        m = DatePattern.THIS_WEEKDAY.getPattern().matcher(text);
        if (m.matches()) {
            var dow = WeekdayMapper.fromJapanese(m.group(2));
            return Optional.of(
                today.with(TemporalAdjusters
                    .nextOrSame(dow))
            );
        }

        // --- M月D日 ---
        m = DatePattern.MONTH_DAY.getPattern().matcher(text);
        if (m.matches()) {
            int month = Integer.parseInt(m.group(1));
            int day = Integer.parseInt(m.group(2));
            LocalDate date = today.withMonth(month).withDayOfMonth(day);
            // 過去日の場合は翌年にする
            if (date.isBefore(today)) date = date.plusYears(1);
            return Optional.of(date);
        }

        // --- M/D形式 ---
        m = DatePattern.SLASH_DATE.getPattern().matcher(text);
        if (m.matches()) {
            int month = Integer.parseInt(m.group(1));
            int day = Integer.parseInt(m.group(2));
            LocalDate date = today.withMonth(month).withDayOfMonth(day);
            if (date.isBefore(today)) date = date.plusYears(1);
            return Optional.of(date);
        }

        // --- ISO形式 (yyyy-MM-dd) ---
        m = DatePattern.ISO_DATE.getPattern().matcher(text);
        if (m.matches()) {
            return Optional.of(LocalDate.parse(text));
        }

        return Optional.empty();
    }
}
SmartDateController.java（パースAPI） 
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

@RestController
@RequestMapping("/api/date")
public class SmartDateController {

    private final SmartDateParser parser;

    public SmartDateController(SmartDateParser parser) {
        this.parser = parser;
    }

    /**
     * 自然言語テキストを日付に変換するAPI
     * GET /api/date/parse?input=明日
     */
    @GetMapping("/parse")
    public ResponseEntity<Map<String, String>> parse(
            @RequestParam String input) {

        return parser.parse(input)
            .map(date -> ResponseEntity.ok(Map.of(
                "input", input,
                "date", date.toString(),
                "formatted", date.getMonthValue()
                    + "月" + date.getDayOfMonth() + "日"
            )))
            .orElse(ResponseEntity.ok(Map.of(
                "input", input,
                "error", "日付として認識できません"
            )));
    }
}
warning日付の範囲チェック

withDayOfMonth(31)のような不正な日付がパースされる可能性があります。DateTimeExceptionをcatchして、不正な場合はOptional.empty()を返すようにしましょう。

smart-date.js（リアルタイムパース） 
document.addEventListener('DOMContentLoaded', () => {
    const textInput = document.getElementById('deadline-text');
    const preview = document.getElementById('date-preview');
    const hidden = document.getElementById('deadline-hidden');

    if (!textInput) return;

    let debounceTimer;

    textInput.addEventListener('input', () => {
        clearTimeout(debounceTimer);
        const value = textInput.value.trim();

        if (!value) {
            preview.textContent = '';
            preview.className = 'date-preview';
            hidden.value = '';
            return;
        }

        // 300msのデバウンスでAPI呼び出し
        debounceTimer = setTimeout(async () => {
            try {
                const res = await fetch(
                    `/api/date/parse?input=${encodeURIComponent(value)}`
                );
                const data = await res.json();

                if (data.date) {
                    preview.textContent = `→ ${data.formatted}`;
                    preview.className = 'date-preview success';
                    hidden.value = data.date;
                } else {
                    preview.textContent = data.error;
                    preview.className = 'date-preview error';
                    hidden.value = '';
                }
            } catch (err) {
                preview.textContent = '通信エラー';
                preview.className = 'date-preview error';
            }
        }, 300);
    });
});
lightbulbデバウンスの重要性

キー入力のたびにAPIを呼び出すと、サーバーに過度な負荷がかかります。setTimeoutで300msのデバウンスを設定し、ユーザーの入力が落ち着いてからAPIを呼び出すようにしています。

/**

@GetMapping("/test")
public List<Map<String, String>> testPatterns() {
    List<String> samples = List.of(
        "今日", "明日", "明後日", "3日後", "2週間後",
        "来週金曜", "今週水曜", "12月25日", "3/15",
        "2026-04-01"
    );

    return samples.stream().map(input -> {
        var result = parser.parse(input);
        return Map.of(
            "input", input,
            "result", result.map(LocalDate::toString)
                            .orElse("パース失敗")
        );
    }).toList();
}
