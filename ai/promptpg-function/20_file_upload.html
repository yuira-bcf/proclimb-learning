Section 19-2

ファイルアップロード

TodoController.java（アップロード処理） 
import org.springframework.web.multipart.MultipartFile;

@Controller
@RequestMapping("/todo")
public class TodoController {

    private final TodoService todoService;
    private final FileStorageService fileStorageService;

    /**
     * ファイル添付フォームの表示
     */
    @GetMapping("/{id}/attach")
    public String showAttachForm(@PathVariable Integer id, Model model) {
        model.addAttribute("todo", todoService.findById(id));
        return "todo/attach";
    }

    /**
     * ファイルアップロード処理
     */
    @PostMapping("/{id}/attach")
    public String uploadFile(
            @PathVariable Integer id,
            @RequestParam("file") MultipartFile file,
            RedirectAttributes redirectAttributes) {

        // ファイルが空でないかチェック
        if (file.isEmpty()) {
            redirectAttributes.addFlashAttribute("error", "ファイルを選択してください");
            return "redirect:/todo/" + id + "/attach";
        }

        try {
            // ファイルを保存
            fileStorageService.store(id, file);
            redirectAttributes.addFlashAttribute("message", "ファイルをアップロードしました");
        } catch (Exception e) {
            redirectAttributes.addFlashAttribute("error", "アップロードに失敗しました");
        }

        return "redirect:/todo/" + id;
    }
}
FileStorageService.java（ファイル保存サービス） 
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;
import java.nio.file.*;
import java.util.UUID;

@Service
public class FileStorageService {

    private final Path uploadPath;
    private final AttachmentMapper attachmentMapper;

    public FileStorageService(
            @Value("${app.upload.dir}") String uploadDir,
            AttachmentMapper attachmentMapper) {
        this.uploadPath = Paths.get(uploadDir).toAbsolutePath().normalize();
        this.attachmentMapper = attachmentMapper;

        // アップロードディレクトリを作成
        try {
            Files.createDirectories(this.uploadPath);
        } catch (IOException e) {
            throw new RuntimeException("アップロードディレクトリの作成に失敗", e);
        }
    }

    /**
     * ファイルを保存
     */
    public TodoAttachment store(Integer todoId, MultipartFile file) throws IOException {
        // ファイル名の安全性チェック
        String originalFilename = StringUtils.cleanPath(file.getOriginalFilename());
        if (originalFilename.contains("..")) {
            throw new IllegalArgumentException("不正なファイル名です");
        }

        // UUIDで一意なファイル名を生成
        String storedFilename = UUID.randomUUID().toString() + getExtension(originalFilename);

        // ファイルを保存
        Path targetPath = this.uploadPath.resolve(storedFilename);
        Files.copy(file.getInputStream(), targetPath, StandardCopyOption.REPLACE_EXISTING);

        // DB にメタ情報を保存
        TodoAttachment attachment = new TodoAttachment();
        attachment.setTodoId(todoId);
        attachment.setOriginalFilename(originalFilename);
        attachment.setStoredFilename(storedFilename);
        attachment.setContentType(file.getContentType());
        attachment.setFileSize(file.getSize());
        attachment.setUploadedAt(LocalDateTime.now());

        attachmentMapper.insert(attachment);

        return attachment;
    }

    private String getExtension(String filename) {
        int dotIndex = filename.lastIndexOf(".");
        return dotIndex > 0 ? filename.substring(dotIndex) : "";
    }
}

Section 19-3

ファイルダウンロード

TodoController.java（ダウンロード処理） 
import org.springframework.core.io.Resource;
import org.springframework.core.io.UrlResource;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;

@Controller
@RequestMapping("/todo")
public class TodoController {

    /**
     * ファイルダウンロード
     */
    @GetMapping("/attachments/{attachmentId}/download")
    public ResponseEntity<Resource> downloadFile(@PathVariable Integer attachmentId) {
        TodoAttachment attachment = attachmentService.findById(attachmentId);

        try {
            Path filePath = uploadPath.resolve(attachment.getStoredFilename());
            Resource resource = new UrlResource(filePath.toUri());

            if (!resource.exists()) {
                throw new RuntimeException("ファイルが見つかりません");
            }

            // Content-Dispositionヘッダーでファイル名を指定
            String encodedFilename = URLEncoder.encode(
                attachment.getOriginalFilename(), StandardCharsets.UTF_8.toString()
            ).replaceAll("\\+", "%20");

            return ResponseEntity.ok()
                .contentType(MediaType.parseMediaType(attachment.getContentType()))
                .header(HttpHeaders.CONTENT_DISPOSITION,
                    "attachment; filename*=UTF-8''" + encodedFilename)
                .body(resource);

        } catch (IOException e) {
            throw new RuntimeException("ファイルの読み込みに失敗", e);
        }
    }

    /**
     * 添付ファイル削除
     */
    @PostMapping("/attachments/{attachmentId}/delete")
    public String deleteFile(
            @PathVariable Integer attachmentId,
            RedirectAttributes redirectAttributes) {

        TodoAttachment attachment = attachmentService.findById(attachmentId);
        Integer todoId = attachment.getTodoId();

        fileStorageService.delete(attachmentId);
        redirectAttributes.addFlashAttribute("message", "ファイルを削除しました");

        return "redirect:/todo/" + todoId;
    }
}

  • データベース（BLOB）：トランザクション一貫性、小〜中サイズ向き

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにファイル添付機能を追加してください。

【要件】
- MultipartFileを使用してファイルをアップロード
- application.propertiesでファイルサイズ上限（10MB）を設定
- TodoAttachmentエンティティで添付ファイルのメタ情報を管理
- FileStorageServiceでファイル保存（UUID形式のファイル名に変換）
- ファイル名のサニタイズ（パストラバーサル対策）
- ダウンロード機能（Content-Dispositionヘッダーでファイル名を指定）
- 添付ファイルの削除機能

【技術スタック】
- Spring Boot 3.x
- MultipartFile
- MyBatis
- Thymeleaf
gps_fixed ステップ別プロンプト 
# Step 1: 設定とエンティティ
application.propertiesにファイルアップロード設定を追加し、

# Step 2: FileStorageServiceの実装

TodoControllerにファイルアップロードのエンドポイントを追加し、

# Step 4: ダウンロード・削除機能
ResponseEntity<Resource>を返すダウンロードエンドポイントと、
添付ファイルを削除する機能を実装してください。
