Section 57-1

Notification API基礎

ブラウザのNotification APIを使うと、Webアプリからデスクトップ通知を表示できます。タブがバックグラウンドにある時でもユーザーに重要な情報を伝えられる強力な機能です。

notification.js（基本的な通知表示） 
/**
 * ブラウザ通知を表示する
 * @param {string} title - 通知タイトル
 * @param {Object} options - 通知オプション
 */
function showNotification(title, options = {}) {
    // Notification APIが使えるかチェック
    if (!('Notification' in window)) {
        console.warn('このブラウザはNotification APIに対応していません');
        return;
    }

    // 許可されている場合のみ通知
    if (Notification.permission === 'granted') {
        const notification = new Notification(title, {
            body: options.body || '',
            icon: options.icon || '/images/todo-icon.png',
            tag: options.tag || 'todo-notification',
            requireInteraction: options.sticky || false
        });

        // クリックでアプリにフォーカス
        notification.onclick = function() {
            window.focus();
            if (options.url) {
                window.location.href = options.url;
            }
            notification.close();
        };

        // 自動で閉じる（5秒後）
        if (!options.sticky) {
            setTimeout(() => notification.close(), 5000);
        }
    }
}

// 使用例
showNotification('タスク期限通知', {
    body: '「レポート提出」の期限が明日です',
    tag: 'deadline-reminder',
    url: '/todo/5'
});
許可状態の確認 
/**
 * 通知の許可状態を確認する
 * - 'granted'  : 許可済み（通知を表示できる）
 * - 'denied'   : 拒否済み（再リクエスト不可）
 * - 'default'  : 未回答（リクエスト可能）
 */
function checkNotificationStatus() {
    if (!('Notification' in window)) {
        return 'unsupported';
    }
    return Notification.permission;
    // 'granted' | 'denied' | 'default'
}

// 状態に応じたUI表示
const status = checkNotificationStatus();
const statusEl = document.getElementById('notification-status');

switch (status) {
    case 'granted':
        statusEl.textContent = '通知: 許可済み';
        statusEl.className = 'badge badge-success';
        break;
    case 'denied':
        statusEl.textContent = '通知: ブロック中';
        statusEl.className = 'badge badge-danger';
        break;
    case 'default':
        statusEl.textContent = '通知: 未設定';
        statusEl.className = 'badge badge-warning';
        break;
}

notification-permission.js（許可リクエスト） 
/**
 * 通知の許可をリクエストする
 * @returns {Promise<string>} 許可結果
 */
async function requestNotificationPermission() {
    if (!('Notification' in window)) {
        alert('このブラウザは通知に対応していません');
        return 'unsupported';
    }

    // 既に許可/拒否されている場合はリクエスト不要
    if (Notification.permission !== 'default') {
        return Notification.permission;
    }

    // ユーザーに許可をリクエスト
    const permission = await Notification.requestPermission();

    // 結果に応じてUIを更新
    updateNotificationUI(permission);

    if (permission === 'granted') {
        // テスト通知を送信
        showNotification('通知が有効になりました', {
            body: 'タスクの期限リマインダーを受け取れます'
        });
    }

    return permission;
}

/**
 * 通知設定UIを更新する
 */
function updateNotificationUI(permission) {
    const enableBtn = document.getElementById('enable-notification');
    const statusBadge = document.getElementById('notification-status');

    switch (permission) {
        case 'granted':
            enableBtn.style.display = 'none';
            statusBadge.textContent = '有効';
            statusBadge.className = 'badge badge-success';
            break;
        case 'denied':
            enableBtn.disabled = true;
            enableBtn.textContent = 'ブラウザ設定で許可してください';
            statusBadge.textContent = 'ブロック中';
            statusBadge.className = 'badge badge-danger';
            break;
    }
}

// ボタンクリックで許可リクエスト
document.getElementById('enable-notification')
    .addEventListener('click', requestNotificationPermission);

Section 57-3

期限リマインダー通知

サーバーから期限が近いタスクの情報を取得し、定期的にチェックしてブラウザ通知で知らせるリマインダー機能を実装します。

api

期限チェックAPI

期限が近いタスクを返すREST APIをサーバーに追加

定期ポーリング

setIntervalで定期的にサーバーに問い合わせ

リマインダー通知

期限24時間前・1時間前にデスクトップ通知を送信

ReminderController.java（期限チェックAPI） 
import org.springframework.web.bind.annotation.*;
import java.time.LocalDateTime;
import java.util.List;

@RestController
@RequestMapping("/api/reminders")
public class ReminderController {

    private final TodoMapper todoMapper;

    public ReminderController(TodoMapper todoMapper) {
        this.todoMapper = todoMapper;
    }

    /**
     * 期限が近いタスクを取得
     * @param hoursAhead 何時間先までを対象にするか
     */
    @GetMapping("/upcoming")
    public List<TodoDto> getUpcoming(
            @RequestParam(defaultValue = "24") int hoursAhead,
            @AuthenticationPrincipal UserDetails user) {

        LocalDateTime now = LocalDateTime.now();
        LocalDateTime deadline = now.plusHours(hoursAhead);

        return todoMapper.findUpcomingDeadlines(
            user.getUsername(), now, deadline
        );
    }
}
TodoMapper.xml（期限検索SQL） 
<select id="findUpcomingDeadlines" resultType="TodoDto">
    SELECT t.id, t.title, t.deadline,
           TIMESTAMPDIFF(HOUR, NOW(), t.deadline) AS hours_remaining
    FROM todo t
    JOIN users u ON t.user_id = u.id
    WHERE u.username = #{username}
      AND t.status != 'DONE'
      AND t.deadline BETWEEN #{now} AND #{deadline}
    ORDER BY t.deadline ASC
</select>
reminder.js（定期チェック&通知） 
const CHECK_INTERVAL = 15 * 60 * 1000; // 15分ごと
const notifiedTasks = new Set();        // 通知済みタスクID

/**
 * 期限が近いタスクをチェックして通知する
 */
async function checkDeadlineReminders() {
    if (Notification.permission !== 'granted') return;

    try {
        const response = await fetch('/api/reminders/upcoming?hoursAhead=24');
        const tasks = await response.json();

        tasks.forEach(task => {
            const key = task.id + '-' + task.hoursRemaining;

            // 同じタスク・同じ時間帯は重複通知しない
            if (notifiedTasks.has(key)) return;
            notifiedTasks.add(key);

            const urgency = task.hoursRemaining <= 1
                ? '1時間以内' : task.hoursRemaining + '時間以内';

            showNotification('期限リマインダー', {
                body: `「${task.title}」の期限が${urgency}です`,
                tag: 'deadline-' + task.id,
                url: '/todo/' + task.id
            });
        });
    } catch (error) {
        console.error('リマインダーチェックエラー:', error);
    }
}

// 定期チェック開始
setInterval(checkDeadlineReminders, CHECK_INTERVAL);
// 初回チェック
checkDeadlineReminders();
lightbulb通知の重複防止

notifiedTasksのSetでタスクID+残り時間をキーに管理し、同じ通知が何度も表示されるのを防いでいます。
ページをリロードするとリセットされるため、必要に応じてSessionStorageに保存することもできます。

notification-settings.js（設定管理） 
const SETTINGS_KEY = 'todo-notification-settings';

// デフォルト設定
const defaultSettings = {
    remind24h: true,
    remind1h: true,
    remindOverdue: false,
    checkInterval: 15
};

/**
 * 保存済み設定を読み込む
 */
function loadSettings() {
    const saved = localStorage.getItem(SETTINGS_KEY);
    return saved ? JSON.parse(saved) : { ...defaultSettings };
}

/**
 * 設定を保存する
 */
function saveSettings() {
    const settings = {
        remind24h: document.getElementById('remind-24h').checked,
        remind1h: document.getElementById('remind-1h').checked,
        remindOverdue:
            document.getElementById('remind-overdue').checked,
        checkInterval: parseInt(
            document.getElementById('check-interval').value
        )
    };

    localStorage.setItem(SETTINGS_KEY, JSON.stringify(settings));

    // リマインダーの再スケジュール
    restartReminderInterval(settings.checkInterval);

    alert('通知設定を保存しました');
}

/**
 * フォームに設定値を反映
 */
function applySettingsToForm(settings) {
    document.getElementById('remind-24h')
        .checked = settings.remind24h;
    document.getElementById('remind-1h')
        .checked = settings.remind1h;
    document.getElementById('remind-overdue')
        .checked = settings.remindOverdue;
    document.getElementById('check-interval')
        .value = settings.checkInterval;
}

// 初期化
document.addEventListener('DOMContentLoaded', () => {
    applySettingsToForm(loadSettings());

    document.getElementById('save-settings')
        .addEventListener('click', saveSettings);

    document.getElementById('test-notification')
        .addEventListener('click', () => {
            showNotification('テスト通知', {
                body: '通知は正常に動作しています'
            });
        });
});
