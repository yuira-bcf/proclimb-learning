WeeklyReportDto.java（週次レポートDTO） 
import java.time.LocalDate;
import java.util.List;

public class WeeklyReportDto {

    private LocalDate weekStart;
    private LocalDate weekEnd;
    private long totalCreated;       // 今週作成されたタスク数
    private long totalCompleted;     // 今週完了したタスク数
    private long totalPending;       // 未完了タスク数
    private double completionRate;   // 完了率（%）
    private List<CategoryCountDto> categoryBreakdown;  // カテゴリ別内訳

    // getter/setter
    public LocalDate getWeekStart() { return weekStart; }
    public void setWeekStart(LocalDate weekStart) { this.weekStart = weekStart; }
    public LocalDate getWeekEnd() { return weekEnd; }
    public void setWeekEnd(LocalDate weekEnd) { this.weekEnd = weekEnd; }
    public long getTotalCreated() { return totalCreated; }
    public void setTotalCreated(long totalCreated) { this.totalCreated = totalCreated; }
    public long getTotalCompleted() { return totalCompleted; }
    public void setTotalCompleted(long totalCompleted) { this.totalCompleted = totalCompleted; }
    public long getTotalPending() { return totalPending; }
    public void setTotalPending(long totalPending) { this.totalPending = totalPending; }
    public double getCompletionRate() { return completionRate; }
    public void setCompletionRate(double completionRate) { this.completionRate = completionRate; }
    public List<CategoryCountDto> getCategoryBreakdown() { return categoryBreakdown; }
    public void setCategoryBreakdown(List<CategoryCountDto> categoryBreakdown) { this.categoryBreakdown = categoryBreakdown; }
}
WeeklyReportService.java（集計ロジック） 
import org.springframework.stereotype.Service;
import java.time.DayOfWeek;
import java.time.LocalDate;
import java.time.temporal.TemporalAdjusters;

@Service
public class WeeklyReportService {

    private final ReportMapper reportMapper;

    public WeeklyReportService(ReportMapper reportMapper) {
        this.reportMapper = reportMapper;
    }

    /**
     * 指定週の集計データを生成
     */
    public WeeklyReportDto generateReport(LocalDate weekStart) {
        LocalDate weekEnd = weekStart.plusDays(6);

        WeeklyReportDto dto = new WeeklyReportDto();
        dto.setWeekStart(weekStart);
        dto.setWeekEnd(weekEnd);
        dto.setTotalCreated(
            reportMapper.countCreatedBetween(weekStart, weekEnd));
        dto.setTotalCompleted(
            reportMapper.countCompletedBetween(weekStart, weekEnd));
        dto.setTotalPending(reportMapper.countPending());

        long total = dto.getTotalCreated() + dto.getTotalPending();
        dto.setCompletionRate(total > 0
            ? (double) dto.getTotalCompleted() / total * 100 : 0);

        dto.setCategoryBreakdown(
            reportMapper.categoryBreakdown(weekStart, weekEnd));

        return dto;
    }

    /**
     * 今週の開始日（月曜日）を取得
     */
    public LocalDate getCurrentWeekStart() {
        return LocalDate.now()
            .with(TemporalAdjusters.previousOrSame(DayOfWeek.MONDAY));
    }
}
ReportMapper.xml（集計SQL） 
<select id="countCreatedBetween" resultType="long">
    SELECT COUNT(*) FROM todo
    WHERE DATE(created_at) BETWEEN #{start} AND #{end}
</select>

<select id="countCompletedBetween" resultType="long">
    SELECT COUNT(*) FROM todo
    WHERE done = TRUE
      AND DATE(completed_at) BETWEEN #{start} AND #{end}
</select>

<select id="countPending" resultType="long">
    SELECT COUNT(*) FROM todo WHERE done = FALSE
</select>

<select id="categoryBreakdown" resultType="CategoryCountDto">
    SELECT c.name AS categoryName, COUNT(t.id) AS count
    FROM todo t
    JOIN category c ON t.category_id = c.id
    WHERE DATE(t.completed_at) BETWEEN #{start} AND #{end}
      AND t.done = TRUE
    GROUP BY c.name
    ORDER BY count DESC
</select>
lightbulbDATE関数の注意

DATE(created_at)でタイムスタンプから日付部分だけを抽出しています。インデックスが効かなくなる場合があるため、大量データではcreated_at >= #{start} AND created_at < #{endPlusOne}の形式が推奨されます。

ReportScheduler.java（自動レポート生成） 
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;
import java.time.LocalDate;

@Component
public class ReportScheduler {

    private static final Logger log =
        LoggerFactory.getLogger(ReportScheduler.class);

    private final WeeklyReportService reportService;
    private final ReportMapper reportMapper;

    public ReportScheduler(WeeklyReportService reportService,
                           ReportMapper reportMapper) {
        this.reportService = reportService;
        this.reportMapper = reportMapper;
    }

    /**
     * 毎週月曜日の朝9時に前週のレポートを自動生成
     * cron = "秒 分 時 日 月 曜日"
     */
    @Scheduled(cron = "0 0 9 * * MON")
    public void generateWeeklyReport() {
        LocalDate lastWeekStart =
            reportService.getCurrentWeekStart().minusWeeks(1);

        log.info("週次レポート自動生成開始: {}", lastWeekStart);

        WeeklyReportDto report =
            reportService.generateReport(lastWeekStart);

        // DBに保存
        reportMapper.insertReport(
            lastWeekStart,
            report.getWeekEnd(),
            report.getTotalCreated(),
            report.getTotalCompleted(),
            report.getCompletionRate()
        );

        log.info("週次レポート自動生成完了: 完了数={}",
            report.getTotalCompleted());
    }
}
schema.sql（レポートテーブル） 
CREATE TABLE weekly_report (
    id INT AUTO_INCREMENT PRIMARY KEY,
    week_start DATE NOT NULL,
    week_end DATE NOT NULL,
    total_created INT DEFAULT 0,
    total_completed INT DEFAULT 0,
    completion_rate DECIMAL(5,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_week_start (week_start)
);
lightbulbcron式の読み方

"0 0 9 * * MON" = 「毎週月曜日の9時00分00秒」に実行。6つのフィールドは「秒 分 時 日 月 曜日」に対応します。*は「毎（全て）」を意味します。

ReportController.java（一覧・詳細・手動生成） 
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;
import java.time.LocalDate;

@Controller
@RequestMapping("/reports")
public class ReportController {

    private final WeeklyReportService reportService;
    private final ReportMapper reportMapper;

    public ReportController(WeeklyReportService reportService,
                            ReportMapper reportMapper) {
        this.reportService = reportService;
        this.reportMapper = reportMapper;
    }

    /**
     * レポート一覧
     */
    @GetMapping
    public String list(Model model) {
        model.addAttribute("reports", reportMapper.findAllReports());
        return "report/list";
    }

    /**
     * レポート詳細
     */
    @GetMapping("/{id}")
    public String detail(@PathVariable Integer id, Model model) {
        var saved = reportMapper.findReportById(id);
        if (saved == null) return "redirect:/reports";

        WeeklyReportDto report =
            reportService.generateReport(saved.getWeekStart());
        model.addAttribute("report", report);
        return "report/detail";
    }

    /**
     * 今週分を手動生成
     */
    @PostMapping("/generate")
    public String generate(RedirectAttributes redirectAttributes) {
        LocalDate weekStart = reportService.getCurrentWeekStart();
        WeeklyReportDto report = reportService.generateReport(weekStart);

        reportMapper.insertReport(
            weekStart, report.getWeekEnd(),
            report.getTotalCreated(), report.getTotalCompleted(),
            report.getCompletionRate());

        redirectAttributes.addFlashAttribute("message",
            "週次レポートを生成しました");
        return "redirect:/reports";
    }
}
