CountdownDto.java（カウントダウン情報） 
import java.time.LocalDateTime;

public class CountdownDto {

    private Long todoId;
    private String title;
    private LocalDateTime deadline;
    private long remainingMillis;   // 残りミリ秒
    private String remainingText;   // "2日 3時間 15分"
    private String urgencyLevel;    // "danger", "warning", "normal", "expired"

    // getter/setter
    public Long getTodoId() { return todoId; }
    public void setTodoId(Long todoId) { this.todoId = todoId; }
    public String getTitle() { return title; }
    public void setTitle(String title) { this.title = title; }
    public LocalDateTime getDeadline() { return deadline; }
    public void setDeadline(LocalDateTime deadline) { this.deadline = deadline; }
    public long getRemainingMillis() { return remainingMillis; }
    public void setRemainingMillis(long remainingMillis) { this.remainingMillis = remainingMillis; }
    public String getRemainingText() { return remainingText; }
    public void setRemainingText(String remainingText) { this.remainingText = remainingText; }
    public String getUrgencyLevel() { return urgencyLevel; }
    public void setUrgencyLevel(String urgencyLevel) { this.urgencyLevel = urgencyLevel; }
}
CountdownService.java（残り時間計算） 
import org.springframework.stereotype.Service;
import java.time.Duration;
import java.time.LocalDateTime;
import java.util.List;
import java.util.stream.Collectors;

@Service
public class CountdownService {

    private final TodoMapper todoMapper;

    public CountdownService(TodoMapper todoMapper) {
        this.todoMapper = todoMapper;
    }

    /**
     * 期限付き未完了タスクのカウントダウン情報を取得
     */
    public List<CountdownDto> getCountdowns() {
        List<Todo> todos = todoMapper.findWithDeadline();
        LocalDateTime now = LocalDateTime.now();

        return todos.stream()
            .map(todo -> toCountdownDto(todo, now))
            .sorted((a, b) -> Long.compare(
                a.getRemainingMillis(), b.getRemainingMillis()))
            .collect(Collectors.toList());
    }

    private CountdownDto toCountdownDto(Todo todo, LocalDateTime now) {
        CountdownDto dto = new CountdownDto();
        dto.setTodoId(todo.getId());
        dto.setTitle(todo.getTitle());
        dto.setDeadline(todo.getDeadline());

        Duration duration = Duration.between(now, todo.getDeadline());
        long millis = duration.toMillis();
        dto.setRemainingMillis(millis);

        if (millis < 0) {
            dto.setRemainingText("期限切れ");
            dto.setUrgencyLevel("expired");
        } else {
            long days = duration.toDays();
            long hours = duration.toHours() % 24;
            long minutes = duration.toMinutes() % 60;
            dto.setRemainingText(days + "日 " + hours + "時間 " + minutes + "分");
            dto.setUrgencyLevel(determineUrgency(millis));
        }
        return dto;
    }

    private String determineUrgency(long millis) {
        long hours = millis / (1000 * 60 * 60);
        if (hours < 24)  return "danger";   // 24時間以内
        if (hours < 72)  return "warning";  // 3日以内
        return "normal";
    }
}

Duration.between(start, end)は2つのLocalDateTime間の差をナノ秒精度で算出します。toDays()やtoHours()で日・時間単位に変換できます。

JavaScriptカウントダウンタイマー

setIntervalを使って1秒ごとにリアルタイム更新されるカウントダウンタイマーをフロントエンドに実装します。

countdown.js（リアルタイムカウントダウン） 
/**
 * カウントダウンタイマーを初期化
 * @param {string} elementId - 表示先のDOM要素ID
 * @param {string} deadlineISO - ISO8601形式の期限日時
 */
function initCountdown(elementId, deadlineISO) {
    const el = document.getElementById(elementId);
    const deadline = new Date(deadlineISO).getTime();

    const timer = setInterval(function() {
        const now = Date.now();
        const diff = deadline - now;

        if (diff <= 0) {
            clearInterval(timer);
            el.innerHTML = '<span class="countdown-expired">期限切れ</span>';
            el.closest('.countdown-card').classList.add('expired');
            return;
        }

        const days    = Math.floor(diff / (1000 * 60 * 60 * 24));
        const hours   = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
        const seconds = Math.floor((diff % (1000 * 60)) / 1000);

        el.querySelector('.cd-days').textContent    = days;
        el.querySelector('.cd-hours').textContent   = String(hours).padStart(2, '0');
        el.querySelector('.cd-minutes').textContent = String(minutes).padStart(2, '0');
        el.querySelector('.cd-seconds').textContent = String(seconds).padStart(2, '0');

        // 緊急度に応じてクラスを動的変更
        const card = el.closest('.countdown-card');
        card.classList.remove('danger', 'warning', 'normal');
        if (diff < 24 * 60 * 60 * 1000)      card.classList.add('danger');
        else if (diff < 72 * 60 * 60 * 1000)  card.classList.add('warning');
        else                                    card.classList.add('normal');
    }, 1000);
}

toast-notification.js（期限切れトースト） 
/**
 * トースト通知を表示
 */
function showToast(message, type) {
    const container = document.getElementById('toast-container')
        || createToastContainer();

    const toast = document.createElement('div');
    toast.className = 'toast toast-' + (type || 'info');
    toast.innerHTML =
        '<span class="material-icons toast-icon">alarm</span>' +
        '<span class="toast-message">' + message + '</span>' +
        '<button class="toast-close" onclick="this.parentElement.remove()">&times;</button>';

    container.appendChild(toast);

    // 5秒後に自動消去
    setTimeout(function() {
        toast.classList.add('toast-fade-out');
        setTimeout(function() { toast.remove(); }, 300);
    }, 5000);
}

function createToastContainer() {
    const container = document.createElement('div');
    container.id = 'toast-container';
    container.style.cssText =
        'position:fixed; top:20px; right:20px; z-index:9999; ' +
        'display:flex; flex-direction:column; gap:8px;';
    document.body.appendChild(container);
    return container;
}

# Step 1: カウントダウンDTO・サービスの作成
CountdownDto（todoId, title, deadline, remainingMillis, urgencyLevel）を作成し、期限付き未完了タスクの残り時間を計算するサービスを実装してください。

# Step 2: JavaScriptカウントダウンタイマー
setIntervalで1秒ごとに残り日数・時間・分・秒を再計算して表示を更新するJavaScriptを実装してください。padStartでゼロ埋めしてください。

# Step 3: 緊急度別の色分けCSS
残り24時間以内は赤、3日以内は黄色、それ以上は緑のスタイルを適用し、24時間以内にはpulseアニメーションを追加してください。

# Step 4: 期限切れトースト通知
カウントダウンがゼロになった瞬間に画面右上にスライドインするトースト通知を表示し、5秒後に自動消去する機能を実装してください。
