Section 66-1

SseEmitterの基礎

Server-Sent Events（SSE）はサーバーからクライアントへの一方向リアルタイム通信を実現する技術です。Spring BootのSseEmitterで簡単に実装できます。

cell_tower

SSE（Server-Sent Events）

HTTPベースの一方向リアルタイム通信プロトコル

compare_arrows

WebSocketとの違い

SSEは一方向で軽量、WebSocketは双方向だが実装が重い

memory

SseEmitter管理

接続中のEmitterをConcurrentHashMapで管理

SseService.java（Emitter管理サービス） 
import org.springframework.stereotype.Service;
import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;
import java.util.concurrent.ConcurrentHashMap;
import java.util.Map;

@Service
public class SseService {

    // ユーザーIDごとのEmitterを管理
    private final Map<Integer, SseEmitter> emitters
        = new ConcurrentHashMap<>();

    /**
     * 新しいSSE接続を作成
     */
    public SseEmitter createEmitter(Integer userId) {
        // タイムアウト: 30分
        SseEmitter emitter = new SseEmitter(30 * 60 * 1000L);

        emitter.onCompletion(() -> emitters.remove(userId));
        emitter.onTimeout(() -> emitters.remove(userId));
        emitter.onError(e -> emitters.remove(userId));

        emitters.put(userId, emitter);
        return emitter;
    }

    /**
     * 特定ユーザーにイベントを送信
     */
    public void sendEvent(Integer userId, String eventName, Object data) {
        SseEmitter emitter = emitters.get(userId);
        if (emitter != null) {
            try {
                emitter.send(SseEmitter.event()
                    .name(eventName)
                    .data(data));
            } catch (Exception e) {
                emitters.remove(userId);
            }
        }
    }

    /**
     * 全接続ユーザーにブロードキャスト
     */
    public void broadcast(String eventName, Object data) {
        emitters.forEach((userId, emitter) -> {
            try {
                emitter.send(SseEmitter.event()
                    .name(eventName)
                    .data(data));
            } catch (Exception e) {
                emitters.remove(userId);
            }
        });
    }
}
lightbulbSSE vs WebSocket

SSEはサーバー→クライアントの一方向通信に特化しています。チャットのような双方向通信にはWebSocketが適していますが、タスク更新通知のような用途ではSSEの方がシンプルで十分です。

SseController.java（SSE接続エンドポイント） 
import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;

@RestController
@RequestMapping("/api/sse")
public class SseController {

    private final SseService sseService;

    public SseController(SseService sseService) {
        this.sseService = sseService;
    }

    /**
     * SSE接続を確立
     * Content-Type: text/event-stream
     */
    @GetMapping(value = "/connect", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter connect(@AuthenticationPrincipal UserDetails user) {
        Integer userId = ((CustomUserDetails) user).getUserId();
        return sseService.createEmitter(userId);
    }
}
TodoService.java（タスク変更時にSSEイベント送信） 
@Service
public class TodoService {

    private final TodoMapper todoMapper;
    private final SseService sseService;

    public TodoService(TodoMapper todoMapper, SseService sseService) {
        this.todoMapper = todoMapper;
        this.sseService = sseService;
    }

    /**
     * タスク作成（SSEイベント付き）
     */
    public void create(Todo todo) {
        todoMapper.insert(todo);

        // 全ユーザーにタスク作成イベントをブロードキャスト
        sseService.broadcast("todo-created", Map.of(
            "id", todo.getId(),
            "title", todo.getTitle(),
            "message", "新しいタスクが追加されました"
        ));
    }

    /**
     * タスク更新（SSEイベント付き）
     */
    public void update(Todo todo) {
        todoMapper.update(todo);

        sseService.broadcast("todo-updated", Map.of(
            "id", todo.getId(),
            "title", todo.getTitle(),
            "completed", todo.isCompleted()
        ));
    }

    /**
     * タスク削除（SSEイベント付き）
     */
    public void delete(Integer id) {
        todoMapper.deleteById(id);

        sseService.broadcast("todo-deleted", Map.of("id", id));
    }
}
lightbulbproduces属性

produces = MediaType.TEXT_EVENT_STREAM_VALUEを指定することで、レスポンスのContent-Typeがtext/event-streamに設定され、ブラウザがSSE接続として認識します。

ブラウザのEventSource APIを使って、サーバーからのSSEイベントを受信します。自動再接続機能も備えています。

JavaScript（EventSource接続） 
/**
 * SSE接続を管理するクラス
 */
class TodoSSEClient {

    constructor() {
        this.eventSource = null;
        this.connect();
    }

    /**
     * SSEサーバーに接続
     */
    connect() {
        this.eventSource = new EventSource('/api/sse/connect');

        // 接続成功
        this.eventSource.onopen = () => {
            console.log('SSE接続を確立しました');
            this.updateStatus('connected');
        };

        // タスク作成イベント
        this.eventSource.addEventListener('todo-created', (event) => {
            const data = JSON.parse(event.data);
            this.onTodoCreated(data);
        });

        // タスク更新イベント
        this.eventSource.addEventListener('todo-updated', (event) => {
            const data = JSON.parse(event.data);
            this.onTodoUpdated(data);
        });

        // タスク削除イベント
        this.eventSource.addEventListener('todo-deleted', (event) => {
            const data = JSON.parse(event.data);
            this.onTodoDeleted(data);
        });

        // エラー時（自動再接続される）
        this.eventSource.onerror = () => {
            console.log('SSE接続エラー。再接続を試みます...');
            this.updateStatus('reconnecting');
        };
    }

    /**
     * 接続状態の表示更新
     */
    updateStatus(status) {
        const indicator = document.getElementById('sse-status');
        if (!indicator) return;
        indicator.className = 'sse-status ' + status;
    }

    // イベントハンドラー（後述のセクションで実装）
    onTodoCreated(data) { /* Section 66-4 */ }
    onTodoUpdated(data) { /* Section 66-4 */ }
    onTodoDeleted(data) { /* Section 66-4 */ }

    /**
     * 接続を閉じる
     */
    disconnect() {
        if (this.eventSource) {
            this.eventSource.close();
            this.updateStatus('disconnected');
        }
    }
}

// ページ読み込み時に接続
const sseClient = new TodoSSEClient();
lightbulbEventSourceの自動再接続

EventSourceはネットワーク切断時に自動でリトライします。デフォルトのリトライ間隔はブラウザ依存（通常3〜5秒）ですが、サーバー側でretry:フィールドを送ることで制御できます。

JavaScript（リアルタイム画面更新） 
/**
 * TodoSSEClient のイベントハンドラー実装
 */

onTodoCreated(data) {
    // 新しいタスク要素を作成してリストに追加
    const container = document.getElementById('todo-container');
    const item = document.createElement('div');
    item.className = 'todo-item';
    item.id = 'todo-' + data.id;
    item.innerHTML = `
        <div class="todo-checkbox">
            <input type="checkbox" data-id="${data.id}"
                   onchange="toggleStatus(this)" />
        </div>
        <div class="todo-title">${this.escapeHtml(data.title)}</div>
    `;

    // ハイライトアニメーション付きで追加
    item.style.animation = 'fadeInSlide 0.3s ease-out';
    container.prepend(item);

    this.showToast('新しいタスク「' + data.title + '」が追加されました');
}

onTodoUpdated(data) {
    const item = document.getElementById('todo-' + data.id);
    if (!item) return;

    // チェックボックスの状態を更新
    const checkbox = item.querySelector('input[type="checkbox"]');
    if (checkbox) checkbox.checked = data.completed;

    // 完了時のスタイルを切替
    item.classList.toggle('completed', data.completed);

    // ハイライト
    item.style.animation = 'none';
    item.offsetHeight; // リフロー
    item.style.animation = 'highlight 1s ease-out';
}

onTodoDeleted(data) {
    const item = document.getElementById('todo-' + data.id);
    if (!item) return;

    // フェードアウトして削除
    item.style.animation = 'fadeOut 0.3s ease-out';
    item.addEventListener('animationend', () => item.remove());
}

/**
 * トースト通知を表示
 */
showToast(message) {
    const toast = document.createElement('div');
    toast.className = 'sse-toast';
    toast.textContent = message;
    document.body.appendChild(toast);

    setTimeout(() => {
        toast.classList.add('fade-out');
        toast.addEventListener('animationend', () => toast.remove());
    }, 3000);
}

/**
 * XSS対策
 */
escapeHtml(str) {
    const div = document.createElement('div');
    div.textContent = str;
    return div.innerHTML;
}

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにSSE（Server-Sent Events）でリアルタイム更新機能を追加してください。

【要件】
- SseEmitterで接続管理（ConcurrentHashMap）
- タスクのCRUD操作時にSSEイベントをブロードキャスト
- JavaScriptのEventSourceでイベントを受信
- DOM操作でページ更新なしにリアルタイム反映

【技術スタック】
- Spring Boot 3.x SseEmitter
- JavaScript EventSource API
- CSSアニメーション（追加/更新/削除のフィードバック）
gps_fixed ステップ別プロンプト 
# Step 1: SseEmitter管理サービス
SseServiceクラスを作成してください。ユーザーIDごとのEmitterをConcurrentHashMapで管理し、sendEvent/broadcastメソッドを実装してください。

# Step 2: SSE接続エンドポイント
GET /api/sse/connectでSseEmitterを返すコントローラーを作成してください。Content-Typeはtext/event-streamにしてください。

# Step 3: EventSource接続
JavaScriptのEventSourceでSSEに接続し、todo-created/todo-updated/todo-deletedイベントをリスニングしてください。

# Step 4: リアルタイム画面更新
各イベントを受信したらDOM操作で画面を即座に更新してください。追加はprepend、削除はfadeOut、更新はhighlightアニメーションで。
