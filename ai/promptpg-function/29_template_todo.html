schema.sql（テンプレートテーブル） 
-- テンプレート本体
CREATE TABLE todo_template (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description VARCHAR(500),
    user_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- テンプレートに含まれるタスク項目
CREATE TABLE todo_template_item (
    id INT AUTO_INCREMENT PRIMARY KEY,
    template_id INT NOT NULL,
    title VARCHAR(200) NOT NULL,
    description VARCHAR(500),
    display_order INT DEFAULT 0,
    FOREIGN KEY (template_id) REFERENCES todo_template(id)
        ON DELETE CASCADE
);

TemplateController.java 
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;

@Controller
@RequestMapping("/templates")
public class TemplateController {

    private final TemplateService templateService;

    public TemplateController(TemplateService templateService) {
        this.templateService = templateService;
    }

    /** テンプレート作成フォーム */
    @GetMapping("/new")
    public String newForm(Model model) {
        model.addAttribute("template", new TodoTemplate());
        return "template/form";
    }

    /** テンプレート保存 */
    @PostMapping
    public String create(@ModelAttribute TodoTemplate template,
                         @RequestParam List<String> itemTitles,
                         @AuthenticationPrincipal UserDetails user,
                         RedirectAttributes ra) {
        templateService.create(template, itemTitles, user);
        ra.addFlashAttribute("message", "テンプレートを作成しました");
        return "redirect:/templates";
    }

    /** テンプレート削除 */
    @PostMapping("/{id}/delete")
    public String delete(@PathVariable Integer id,
                         RedirectAttributes ra) {
        templateService.delete(id);
        ra.addFlashAttribute("message", "テンプレートを削除しました");
        return "redirect:/templates";
    }
}
TemplateService.java 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.List;

@Service
public class TemplateService {

    private final TodoTemplateMapper templateMapper;
    private final TodoTemplateItemMapper itemMapper;

    public TemplateService(TodoTemplateMapper templateMapper,
                           TodoTemplateItemMapper itemMapper) {
        this.templateMapper = templateMapper;
        this.itemMapper = itemMapper;
    }

    /** テンプレートとタスク項目を一括作成 */
    @Transactional
    public void create(TodoTemplate template,
                       List<String> itemTitles,
                       UserDetails user) {
        template.setUserId(
            ((CustomUserDetails) user).getUserId());
        templateMapper.insert(template);

        for (int i = 0; i < itemTitles.size(); i++) {
            String title = itemTitles.get(i).trim();
            if (!title.isEmpty()) {
                TodoTemplateItem item = new TodoTemplateItem();
                item.setTemplateId(template.getId());
                item.setTitle(title);
                item.setDisplayOrder(i + 1);
                itemMapper.insert(item);
            }
        }
    }

    /** テンプレート削除（CASCADE） */
    @Transactional
    public void delete(Integer templateId) {
        templateMapper.deleteById(templateId);
    }
}

/** テンプレートからタスクを一括作成 */
@PostMapping("/{id}/apply")
public String applyTemplate(@PathVariable Integer id,
                            @AuthenticationPrincipal UserDetails user,
                            RedirectAttributes ra) {
    int count = templateService.applyTemplate(id, user);
    ra.addFlashAttribute("message",
        count + "件のタスクを作成しました");
    return "redirect:/todo";
}
TemplateService.java（一括作成ロジック） 
/**
 * テンプレートからToDoタスクを一括作成
 * @return 作成したタスク数
 */
@Transactional
public int applyTemplate(Integer templateId,
                         UserDetails user) {
    Integer userId = ((CustomUserDetails) user).getUserId();

    // テンプレート項目を取得
    List<TodoTemplateItem> items =
        itemMapper.findByTemplateId(templateId);

    // 各項目をtodoテーブルにINSERT
    for (TodoTemplateItem item : items) {
        Todo todo = new Todo();
        todo.setTitle(item.getTitle());
        todo.setDescription(item.getDescription());
        todo.setCompleted(false);
        todo.setUserId(userId);
        todoMapper.insert(todo);
    }

    return items.size();
}
warning大量データの注意

テンプレートに100件以上の項目がある場合、1件ずつINSERTすると遅くなります。MyBatisの<foreach>を使ったバッチINSERTを検討しましょう。

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにテンプレートToDo機能を追加してください。

【要件】

- テンプレートを選択してタスクを一括作成する機能
- テンプレート作成画面でJavaScriptで項目を動的追加

【技術スタック】
- Spring Boot 3.x + MyBatis
- Thymeleaf
- @Transactionalで一括作成の整合性を保証
gps_fixed ステップ別プロンプト 
# Step 1: テーブル設計
todo_templateテーブル（名前・説明・ユーザーID）とtodo_template_itemテーブル（タイトル・表示順）を作成してください。

# Step 2: エンティティとMapper
TodoTemplate・TodoTemplateItemのエンティティとMyBatis Mapperを作成してください。ResultMapのcollectionで項目一覧を自動ロードしてください。

# Step 3: テンプレート登録
テンプレート名と複数のタスク項目を一度に保存するフォームとコントローラーを作成してください。

# Step 4: タスク一括作成
テンプレートを選択してtodoテーブルにタスクを一括INSERTするapply機能を実装してください。

# Step 5: 一覧画面
テンプレート一覧をカード型レイアウトで表示し、「タスク作成」「削除」ボタンを配置してください。
