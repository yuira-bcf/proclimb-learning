StatsMapper.xml（集計SQL） 
<!-- ステータス別集計 -->
<select id="countByStatus" resultType="map">
    SELECT status, COUNT(*) AS cnt
    FROM todo
    WHERE user_id = #{userId}
    GROUP BY status
</select>

<!-- カテゴリ別集計 -->
<select id="countByCategory" resultType="map">
    SELECT c.name AS category_name, COUNT(t.id) AS cnt
    FROM todo t
    JOIN category c ON t.category_id = c.id
    WHERE t.user_id = #{userId}
    GROUP BY c.name
    ORDER BY cnt DESC
</select>

<!-- 直近7日間の日別完了数 -->
<select id="countDailyCompleted" resultType="map">
    SELECT DATE(updated_at) AS completed_date,
           COUNT(*) AS cnt
    FROM todo
    WHERE user_id = #{userId}
      AND status = 'COMPLETED'
      AND updated_at >= DATE_SUB(CURDATE(), INTERVAL 7 DAY)
    GROUP BY DATE(updated_at)
    ORDER BY completed_date
</select>

<!-- 期限切れタスク数 -->
<select id="countOverdue" resultType="long">
    SELECT COUNT(*)
    FROM todo
    WHERE user_id = #{userId}
      AND status != 'COMPLETED'
      AND deadline < CURDATE()
</select>
DashboardService.java（集計サービス） 
import org.springframework.stereotype.Service;
import java.util.*;
import java.util.stream.Collectors;

@Service
public class DashboardService {

    private final StatsMapper statsMapper;

    public DashboardService(StatsMapper statsMapper) {
        this.statsMapper = statsMapper;
    }

    /**
     * ダッシュボード用の統計データを集計
     */
    public DashboardStats getStats(Integer userId) {
        DashboardStats stats = new DashboardStats();

        // ステータス別集計
        List<Map<String, Object>> statusList =
            statsMapper.countByStatus(userId);
        Map<String, Long> statusCounts = statusList.stream()
            .collect(Collectors.toMap(
                m -> (String) m.get("status"),
                m -> ((Number) m.get("cnt")).longValue()
            ));
        stats.setStatusCounts(statusCounts);

        long total = statusCounts.values().stream()
            .mapToLong(Long::longValue).sum();
        long completed = statusCounts.getOrDefault("COMPLETED", 0L);
        stats.setTotalTasks(total);
        stats.setCompletedTasks(completed);
        stats.setPendingTasks(total - completed);

        // カテゴリ別集計
        List<Map<String, Object>> catList =
            statsMapper.countByCategory(userId);
        Map<String, Long> catCounts = catList.stream()
            .collect(Collectors.toMap(
                m -> (String) m.get("category_name"),
                m -> ((Number) m.get("cnt")).longValue()
            ));
        stats.setCategoryCounts(catCounts);

        // 日別完了数
        List<Map<String, Object>> dailyList =
            statsMapper.countDailyCompleted(userId);
        Map<String, Long> dailyCounts = new LinkedHashMap<>();
        for (Map<String, Object> row : dailyList) {
            dailyCounts.put(
                row.get("completed_date").toString(),
                ((Number) row.get("cnt")).longValue()
            );
        }
        stats.setDailyCompletedCounts(dailyCounts);

        // 期限切れ
        stats.setOverdueTasks(statsMapper.countOverdue(userId));

        return stats;
    }
}
lightbulbLinkedHashMapで順序を保持

日別データは日付順に表示する必要があるため、LinkedHashMapを使って挿入順序を保持しています。通常のHashMapでは順序が保証されません。

account_tree 処理フロー: ダッシュボード機能

1
DashboardController.index() で統計データを集計
DashboardStats stats = dashboardService.getStats(loginUser.getUserId());
2
ステータス別カウント（Map<String, Long>）を集計
SELECT status, COUNT(*) AS cnt FROM todo WHERE user_id = #{userId} GROUP BY status
3
優先度別・カテゴリ別の集計を実行
SELECT c.name, COUNT(t.id) AS cnt FROM todo t JOIN category c ON t.category_id = c.id GROUP BY c.name
4
Thymeleaf で統計カード + Chart.js グラフを描画
model.addAttribute("stats", stats); → th:text="${stats.totalTasks}" + Chart.js描画

ポイント: GROUP BYで集計したMapデータをThymeleafのth:inline="javascript"でChart.jsに渡し、棒グラフ・円グラフを描画します。LinkedHashMapで日付順序を保持します。

DashboardController.java 
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;

@Controller
public class DashboardController {

    private final DashboardService dashboardService;

    public DashboardController(DashboardService dashboardService) {
        this.dashboardService = dashboardService;
    }

    @GetMapping("/dashboard")
    public String dashboard(
            @AuthenticationPrincipal LoginUser loginUser,
            Model model) {
        DashboardStats stats =
            dashboardService.getStats(loginUser.getUserId());
        model.addAttribute("stats", stats);
        return "dashboard";
    }
}
lightbulbChart.js v4の変更点

Chart.js v4ではTree-shakingに対応し、chart.umd.min.jsを使えば全機能がバンドルされます。v3以前のようにChart.register()で個別登録する必要はありません。

warningThymeleafインラインの注意点

th:inline="javascript"を使う場合、Thymeleafは自動的にデータをJSON形式にシリアライズします。ただし、XSS対策のため出力されるデータは自動エスケープされます。/*[[...]]*/構文を使うことで安全にデータを埋め込めます。
