Section 34-2

fetch APIでタスク追加

フォーム送信をfetch API（非同期通信）で処理し、ページリロードなしでタスクを追加します。サーバーにはJSON形式でデータを送信します。

TodoApiController.java（タスク追加API） 
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.BindingResult;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/todo")
public class TodoApiController {

    private final TodoService todoService;

    public TodoApiController(TodoService todoService) {
        this.todoService = todoService;
    }

    /**
     * タスクを追加（クイック入力用）
     */
    @PostMapping
    public ResponseEntity<?> create(
            @Valid @RequestBody TodoCreateRequest req,
            BindingResult result) {

        if (result.hasErrors()) {
            String msg = result.getFieldError().getDefaultMessage();
            return ResponseEntity.badRequest()
                .body(Map.of("error", msg));
        }

        Todo todo = todoService.create(req.getTitle());
        return ResponseEntity.status(HttpStatus.CREATED)
            .body(Map.of(
                "id", todo.getId(),
                "title", todo.getTitle()
            ));
    }
}

// リクエストDTO
public class TodoCreateRequest {
    @NotBlank(message = "タイトルは必須です")
    @Size(max = 200, message = "200文字以内で入力してください")
    private String title;
    // getter/setter
}
quick-input.js（fetch APIでタスク送信） 
document.addEventListener('DOMContentLoaded', () => {
    const input = document.getElementById('quickTitle');
    const btn = document.getElementById('quickSubmitBtn');
    const errorDiv = document.getElementById('quickError');

    // --- ボタンクリックで送信 ---
    btn.addEventListener('click', submitTask);

    // --- Enterキーで送信 ---
    input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
            e.preventDefault();
            submitTask();
        }
    });

    async function submitTask() {
        const title = input.value.trim();
        if (!title) {
            showError('タイトルを入力してください');
            return;
        }

        btn.disabled = true;

        try {
            const res = await fetch('/api/todo', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ title })
            });

            const data = await res.json();

            if (!res.ok) {
                showError(data.error || '追加に失敗しました');
                return;
            }

            // 成功：入力クリア＋リスト先頭に追加
            input.value = '';
            hideError();
            prependToList(data);
            input.focus();

        } catch (err) {
            showError('通信エラーが発生しました');
        } finally {
            btn.disabled = false;
        }
    }

    function showError(msg) {
        errorDiv.textContent = msg;
        errorDiv.style.display = 'block';
    }

    function hideError() {
        errorDiv.style.display = 'none';
    }
});
warningCSRF対策

Spring SecurityのCSRF保護が有効な場合、fetch APIリクエストにもCSRFトークンを含める必要があります。
Thymeleafの<meta name="_csrf" th:content="${_csrf.token}">から取得してヘッダーに付与しましょう。

quick-input.js（リスト即時反映） 
/** タスクリストの先頭に新しい要素を追加 */
function prependToList(todo) {
    const list = document.getElementById('todo-list');

    const item = document.createElement('li');
    item.className = 'todo-item new-item';
    item.dataset.id = todo.id;
    item.innerHTML = `
        <span class="todo-title">${escapeHtml(todo.title)}</span>
        <div class="todo-actions">
            <a href="/todo/${todo.id}" class="btn btn-sm">詳細</a>
        </div>
    `;

    // リスト先頭に挿入
    list.prepend(item);

    // アニメーション後にハイライトを解除
    setTimeout(() => item.classList.remove('new-item'), 2000);
}

/** XSS対策：HTMLエスケープ */
function escapeHtml(str) {
    const div = document.createElement('div');
    div.textContent = str;
    return div.innerHTML;
}

lightbulbXSS対策を忘れずに

ユーザー入力をそのままinnerHTMLに埋め込むとXSS（クロスサイトスクリプティング）の脆弱性になります。
必ずtextContentで設定するか、escapeHtml()関数でエスケープしましょう。

/** エラー表示 */
function showError(msg) {
    const errorDiv = document.getElementById('quickError');
    const input = document.getElementById('quickTitle');

    errorDiv.textContent = msg;
    errorDiv.style.display = 'block';
    input.classList.add('input-error');
    input.focus();
}

/** エラー非表示 */
function hideError() {
    const errorDiv = document.getElementById('quickError');
    const input = document.getElementById('quickTitle');

    errorDiv.style.display = 'none';
    input.classList.remove('input-error');
}

// 入力開始時にエラーをクリア
document.getElementById('quickTitle')

quick-input.js（CSRF対応版fetch） 
/**
 * CSRFトークン付きfetch（Spring Security対応）
 */
function getCsrfToken() {
    const meta = document.querySelector('meta[name="_csrf"]');
    const header = document.querySelector('meta[name="_csrf_header"]');
    return {
        token: meta ? meta.content : '',
        headerName: header ? header.content : 'X-CSRF-TOKEN'
    };
}

async function submitTask() {
    const title = input.value.trim();
    if (!title) {
        showError('タイトルを入力してください');
        return;
    }

    const csrf = getCsrfToken();
    btn.disabled = true;

    try {
        const res = await fetch('/api/todo', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                [csrf.headerName]: csrf.token
            },
            body: JSON.stringify({ title })
        });

        const data = await res.json();

        if (!res.ok) {
            showError(data.error || '追加に失敗しました');
            return;
        }

        input.value = '';
        hideError();
        prependToList(data);
        input.focus();
    } catch (err) {
        showError('通信エラーが発生しました');
    } finally {
        btn.disabled = false;
    }
}

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにクイック入力バー機能を追加してください。

【要件】
- 画面上部にposition:stickyで固定されたタスク入力バー
- タイトル入力してEnterキーまたはボタンクリックで送信
- fetch APIでPOST /api/todoにJSON送信（ページリロードなし）
- 追加成功後にリスト先頭にDOM挿入（スライドインアニメーション付き）
- バリデーションエラーはインライン表示、入力開始で自動クリア
- Spring Security CSRF対応

【技術スタック】
- Spring Boot 3.x + MyBatis
- @RestController + @Valid
- Thymeleaf + Vanilla JavaScript（fetch API）
gps_fixed ステップ別プロンプト 
# Step 1: 固定入力バーUI
画面上部にposition:stickyで固定される入力バー（テキスト入力+送信ボタン）をHTMLとCSSで作成してください。

# Step 2: REST API
POST /api/todoでJSONリクエスト（title）を受け取り、バリデーション（@NotBlank, @Size）付きでタスクを作成するAPIを実装してください。

# Step 3: fetch送信
Enterキーとボタンクリックでfetch APIを呼び出し、成功時に入力欄クリア、エラー時にメッセージ表示するJavaScriptを作成してください。CSRFトークンも含めてください。

# Step 4: DOM即時反映
追加成功後にリスト先頭にDOM要素を挿入し、スライドインアニメーションとハイライト効果を付けてください。XSSエスケープも実装してください。
lightbulb プロンプトのコツ

「ページリロードなしで即時反映してください」とだけ書くと、SPA的な実装を提案されることがあります。
「Thymeleafのサーバーサイドレンダリングと、クイック入力のみfetch APIで非同期にしてください」のように、範囲を限定すると正確な実装が得られます。
