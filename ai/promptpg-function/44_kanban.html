KanbanController.java 
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import java.util.List;

@Controller
public class KanbanController {

    private final TodoMapper todoMapper;

    public KanbanController(TodoMapper todoMapper) {
        this.todoMapper = todoMapper;
    }

    @GetMapping("/kanban")
    public String kanban(
            @AuthenticationPrincipal LoginUser loginUser,
            Model model) {

        Integer userId = loginUser.getUserId();

        List<Todo> todoList =
            todoMapper.findByUserIdAndStatus(userId, "TODO");
        List<Todo> inProgressList =
            todoMapper.findByUserIdAndStatus(userId, "IN_PROGRESS");
        List<Todo> completedList =
            todoMapper.findByUserIdAndStatus(userId, "COMPLETED");

        model.addAttribute("todoList", todoList);
        model.addAttribute("inProgressList", inProgressList);
        model.addAttribute("completedList", completedList);

        return "kanban";
    }
}
TodoMapper.xml（ステータス別取得） 
<select id="findByUserIdAndStatus" resultType="Todo">
    SELECT id, title, status, deadline, priority, category_id
    FROM todo
    WHERE user_id = #{userId}
      AND status = #{status}
    ORDER BY priority DESC, deadline ASC
</select>
warningステータス値の統一

ステータスの値（TODO、IN_PROGRESS、COMPLETED）はDB・Java・HTMLで統一してください。値が不一致だとカラムにタスクが表示されません。enum型での管理を推奨します。

kanban.js（ドラッグ&ドロップ実装） 
document.addEventListener('DOMContentLoaded', function() {
    const cards = document.querySelectorAll('.kanban-card');
    const columns = document.querySelectorAll('.kanban-column');

    // ---- カードのドラッグイベント ----
    cards.forEach(card => {
        card.addEventListener('dragstart', function(e) {
            e.dataTransfer.setData('text/plain', this.dataset.id);
            this.classList.add('dragging');
        });

        card.addEventListener('dragend', function() {
            this.classList.remove('dragging');
            columns.forEach(col =>
                col.classList.remove('drag-over'));
        });
    });

    // ---- カラムのドロップイベント ----
    columns.forEach(column => {
        column.addEventListener('dragover', function(e) {
            e.preventDefault(); // ドロップを許可
            this.classList.add('drag-over');
        });

        column.addEventListener('dragleave', function() {
            this.classList.remove('drag-over');
        });

        column.addEventListener('drop', function(e) {
            e.preventDefault();
            this.classList.remove('drag-over');

            const taskId = e.dataTransfer.getData('text/plain');
            const newStatus = this.dataset.status;

            // カードをDOMで移動
            const card = document.querySelector(
                `.kanban-card[data-id="${taskId}"]`
            );
            this.appendChild(card);

            // APIでステータスを更新
            updateTaskStatus(taskId, newStatus);

            // カラムのカウントを更新
            updateColumnCounts();
        });
    });
});

function updateColumnCounts() {
    document.querySelectorAll('.kanban-column').forEach(col => {
        const count = col.querySelectorAll('.kanban-card').length;
        col.querySelector('.kanban-column-count').textContent = count;
    });
}
lightbulbe.preventDefault()の重要性

dragoverイベントではe.preventDefault()を呼ばないとドロップが発生しません。ブラウザのデフォルト動作ではドロップが禁止されているためです。

REST APIでステータス更新

ドラッグ&ドロップ完了時にfetch APIでREST APIを呼び出し、サーバー側のタスクステータスを更新します。エラー時のロールバック処理も実装します。

kanban.js（API呼び出し） 
/**
 * タスクのステータスをAPIで更新
 */
async function updateTaskStatus(taskId, newStatus) {
    try {
        const response = await fetch(`/api/todo/${taskId}/status`, {
            method: 'PATCH',
            headers: {
                'Content-Type': 'application/json',
                // CSRFトークン（Spring Securityが有効な場合）
                [csrfHeader]: csrfToken
            },
            body: JSON.stringify({ status: newStatus })
        });

        if (!response.ok) {
            throw new Error('ステータス更新に失敗しました');
        }

        // 成功時のフィードバック
        showToast('ステータスを更新しました');

    } catch (error) {
        // 失敗時はページをリロードして元に戻す
        alert(error.message);
        location.reload();
    }
}

// CSRFトークンの取得
const csrfToken = document.querySelector(
    'meta[name="_csrf"]')?.content;
const csrfHeader = document.querySelector(
    'meta[name="_csrf_header"]')?.content;
TodoApiController.java（ステータス更新API） 
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

@RestController
@RequestMapping("/api/todo")
public class TodoApiController {

    private final TodoMapper todoMapper;

    public TodoApiController(TodoMapper todoMapper) {
        this.todoMapper = todoMapper;
    }

    /**
     * タスクのステータスを更新（カンバン用）
     */
    @PatchMapping("/{id}/status")
    public ResponseEntity<?> updateStatus(
            @PathVariable Integer id,
            @RequestBody Map<String, String> body) {

        String newStatus = body.get("status");

        // バリデーション
        if (!List.of("TODO", "IN_PROGRESS", "COMPLETED")
                .contains(newStatus)) {
            return ResponseEntity.badRequest()
                .body(Map.of("error", "無効なステータスです"));
        }

        todoMapper.updateStatus(id, newStatus);
        return ResponseEntity.ok(
            Map.of("message", "ステータスを更新しました"));
    }
}
TodoMapper.xml（ステータス更新SQL） 
<update id="updateStatus">
    UPDATE todo
    SET status = #{status},
        updated_at = NOW()
    WHERE id = #{id}
</update>
warningCSRFトークンを忘れずに

Spring SecurityのCSRF保護が有効な場合、fetch APIのヘッダーにCSRFトークンを含める必要があります。Thymeleafでは<meta name="_csrf" th:content="${_csrf.token}">でトークンをHTMLに埋め込みます。

# Step 1: カンバンレイアウト
CSS Gridで3カラム（TODO/IN_PROGRESS/COMPLETED）のカンバンボードレイアウトを作成してください。レスポンシブ対応で、モバイルでは縦積みにしてください。
