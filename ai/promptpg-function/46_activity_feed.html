CREATE TABLE activity (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    action_type VARCHAR(50) NOT NULL,  -- CREATE, UPDATE, DELETE, COMPLETE
    target_type VARCHAR(50) NOT NULL,  -- TODO, CATEGORY, USER
    target_id INT,
    target_title VARCHAR(255),
    description VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_activity_user_date ON activity(user_id, created_at DESC);

Section 46-2

AOP/イベントで自動記録

Spring AOPを使って、タスクの作成・更新・削除時に自動的にアクティビティを記録します。既存のサービスコードを変更せずに横断的関心事として実装します。

ユーザー情報取得

SecurityContextからログインユーザーを特定

ActivityAspect.java（AOP実装）
import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.annotation.*;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.stereotype.Component;

@Aspect
@Component
public class ActivityAspect {

    private final ActivityMapper activityMapper;

    public ActivityAspect(ActivityMapper activityMapper) {
        this.activityMapper = activityMapper;
    }

    /**
     * タスク作成後にアクティビティを記録
     */
    @AfterReturning(
        pointcut = "execution(* com.example.todo.service.TodoService.create(..))",
        returning = "result"
    )
    public void afterCreate(JoinPoint jp, Object result) {
        Todo todo = (Todo) result;
        Activity activity = new Activity();
        activity.setUserId(getCurrentUserId());
        activity.setActionType("CREATE");
        activity.setTargetType("TODO");
        activity.setTargetId(todo.getId());
        activity.setTargetTitle(todo.getTitle());
        activity.setDescription("タスク「" + todo.getTitle() + "」を作成しました");
        activityMapper.insert(activity);
    }

    /**
     * タスク完了時にアクティビティを記録
     */
    @AfterReturning(
        pointcut = "execution(* com.example.todo.service.TodoService.complete(..))"
    )
    public void afterComplete(JoinPoint jp) {
        Integer todoId = (Integer) jp.getArgs()[0];
        Activity activity = new Activity();
        activity.setUserId(getCurrentUserId());
        activity.setActionType("COMPLETE");
        activity.setTargetType("TODO");
        activity.setTargetId(todoId);
        activity.setDescription("タスクを完了しました");
        activityMapper.insert(activity);
    }

    /**
     * タスク削除時にアクティビティを記録
     */
    @AfterReturning(
        pointcut = "execution(* com.example.todo.service.TodoService.delete(..))"
    )
    public void afterDelete(JoinPoint jp) {
        Integer todoId = (Integer) jp.getArgs()[0];
        Activity activity = new Activity();
        activity.setUserId(getCurrentUserId());
        activity.setActionType("DELETE");
        activity.setTargetType("TODO");
        activity.setTargetId(todoId);
        activity.setDescription("タスクを削除しました");
        activityMapper.insert(activity);
    }

    private Integer getCurrentUserId() {
        LoginUser user = (LoginUser) SecurityContextHolder
            .getContext().getAuthentication().getPrincipal();
        return user.getUserId();
    }
}
warningAOPの適用順序

@AfterReturningはメソッドが正常終了した場合のみ実行されます。例外発生時はアクティビティが記録されないため、失敗した操作が記録されないのは意図的な設計です。

ActivityController.java
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
public class ActivityController {

    private final ActivityMapper activityMapper;

    public ActivityController(ActivityMapper activityMapper) {
        this.activityMapper = activityMapper;
    }

    @GetMapping("/activity")
    public String activity(
            @RequestParam(required = false) String type,
            @AuthenticationPrincipal LoginUser loginUser,
            Model model) {

        Integer userId = loginUser.getUserId();
        List<Activity> activities;

        if (type != null && !type.isEmpty()) {
            activities = activityMapper.findByUserIdAndType(
                userId, type, 50);
        } else {
            activities = activityMapper.findByUserId(
                userId, 50, 0);
        }

        model.addAttribute("activities", activities);
        model.addAttribute("currentType", type);
        return "activity";
    }
}

Spring Boot 3.xのToDoアプリにアクティビティフィード機能を追加してください。

【技術スタック】
- Spring Boot 3.x + MyBatis + Spring AOP
- Thymeleaf
- CSS Timeline UI
gps_fixed ステップ別プロンプト
# Step 1: activityテーブル作成
操作種別(action_type)、対象(target_type/target_id/target_title)、説明(description)を持つactivityテーブルを設計してください。

# Step 2: AOP自動記録
@Aspect + @AfterReturningでTodoServiceのcreate/complete/deleteメソッド実行後にアクティビティを自動記録してください。

# Step 3: タイムラインUI
CSS ::beforeで左側に縦線を表示し、操作種別ごとに色分けしたアイコン付きタイムラインUIを作成してください。

# Step 4: フィルタリング
操作種別（すべて/作成/完了/更新/削除）でアクティビティを絞り込むフィルタボタンを追加してください。
