build.gradle（依存関係）
dependencies {

    implementation 'org.commonmark:commonmark:0.21.0'
}
MarkdownUtil.java（変換ユーティリティ）
import org.commonmark.node.Node;
import org.commonmark.parser.Parser;
import org.commonmark.renderer.html.HtmlRenderer;

public class MarkdownUtil {

    private static final Parser PARSER = Parser.builder().build();
    private static final HtmlRenderer RENDERER =
        HtmlRenderer.builder().build();

    /**
     * MarkdownテキストをHTMLに変換
     */
    public static String toHtml(String markdown) {
        if (markdown == null || markdown.isEmpty()) {
            return "";
        }
        Node document = PARSER.parse(markdown);
        return RENDERER.render(document);
    }
}
lightbulbcommonmarkライブラリ

org.commonmarkはCommonMark仕様に準拠した軽量なMarkdownパーサーです。Spring Bootとの相性が良く、拡張機能（テーブル、取り消し線等）もプラグインで追加できます。

<script>
function insertMd(before, after) {
    var ta = document.getElementById('memoInput');
    var start = ta.selectionStart;
    var end = ta.selectionEnd;
    var selected = ta.value.substring(start, end);
    ta.value = ta.value.substring(0, start)
        + before + selected + after
        + ta.value.substring(end);
    ta.focus();
    ta.selectionStart = start + before.length;
    ta.selectionEnd = start + before.length + selected.length;
}
</script>

// タスク詳細画面でMarkdownをHTMLに変換して表示
@GetMapping("/todo/{id}")
public String detail(@PathVariable Integer id, Model model) {
    Todo todo = todoMapper.findById(id);
    model.addAttribute("todo", todo);

    // Markdown → HTML変換
    String memoHtml = MarkdownUtil.toHtml(todo.getMemo());
    model.addAttribute("memoHtml", memoHtml);

    return "todo/detail";
}

lightbulbth:utext の注意点

th:utextはHTMLをそのまま出力するため、XSS攻撃のリスクがあります。必ずサニタイズ処理を通したHTMLのみを渡してください。次のセクションで詳しく扱います。

Section 55-4

XSS対策サニタイズ

ユーザー入力のMarkdownから生成されたHTMLに含まれる危険なスクリプトを除去するサニタイズ処理を実装します。

security

OWASP HTML Sanitizer

Java側でHTMLタグのホワイトリスト制御

shield

DOMPurify

クライアント側でJavaScript実行を防止

block

scriptタグ除去

script, iframe, onXXX属性などを完全に除去

HtmlSanitizer.java（Java側サニタイズ）
// build.gradle に追加:
// implementation 'com.googlecode.owasp-java-html-sanitizer:owasp-java-html-sanitizer:20220608.1'

import org.owasp.html.HtmlPolicyBuilder;
import org.owasp.html.PolicyFactory;

public class HtmlSanitizer {

    /** Markdown由来のHTMLに許可するタグ */
    private static final PolicyFactory POLICY = new HtmlPolicyBuilder()
        .allowElements(
            "p", "br", "h1", "h2", "h3", "h4", "h5", "h6",
            "ul", "ol", "li", "blockquote", "pre", "code",
            "strong", "em", "a", "img", "table", "thead",
            "tbody", "tr", "th", "td", "hr")
        .allowUrlProtocols("http", "https")
        .allowAttributes("href").onElements("a")
        .allowAttributes("src", "alt").onElements("img")
        .allowAttributes("class").globally()
        .requireRelNofollowOnLinks()
        .toFactory();

    /**
     * HTMLをサニタイズ（危険なタグ・属性を除去）
     */
    public static String sanitize(String html) {
        if (html == null) return "";
        return POLICY.sanitize(html);
    }
}
DOMPurify（クライアント側サニタイズ）
<!-- DOMPurify CDN -->
<script src="https://cdn.jsdelivr.net/npm/dompurify/dist/purify.min.js"></script>

<script>
function updatePreview() {
    var md = document.getElementById('memoInput').value;

    // marked.js でMarkdown→HTML変換
    var rawHtml = marked.parse(md);

    // DOMPurify でサニタイズ（XSS対策）
    var cleanHtml = DOMPurify.sanitize(rawHtml, {
        ALLOWED_TAGS: [
            'p', 'br', 'h1', 'h2', 'h3', 'h4', 'strong', 'em',
            'ul', 'ol', 'li', 'a', 'code', 'pre', 'blockquote',
            'table', 'thead', 'tbody', 'tr', 'th', 'td', 'hr', 'img'
        ],
        ALLOWED_ATTR: ['href', 'src', 'alt', 'class']
    });

    document.getElementById('memoPreview').innerHTML = cleanHtml;
}
</script>
warningXSS対策は必須

ユーザー入力をHTMLとして表示する場合、必ずサニタイズ処理を通してください。サニタイズなしでth:utextやinnerHTMLを使うと、悪意のある<script>タグが実行される危険があります。サーバー側（OWASP）とクライアント側（DOMPurify）の二重防御が理想的です。

XSS攻撃パターンの例

入力サニタイズなしサニタイズあり
<script>alert('XSS')</script>スクリプト実行タグ除去
<img onerror="alert(1)" src="x">スクリプト実行onerror属性除去
<a href="javascript:alert(1)">スクリプト実行href除去
lightbulb二重防御のメリット

サーバー側でサニタイズすればDB保存時点で安全になりますが、クライアント側のリアルタイムプレビューではDOMPurifyでも防御が必要です。両方を組み合わせることで防御の漏れを防ぎます。
