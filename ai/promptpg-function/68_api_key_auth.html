CREATE TABLE api_keys (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    key_name VARCHAR(100) NOT NULL,
    key_value VARCHAR(64) NOT NULL UNIQUE,
    key_prefix VARCHAR(8) NOT NULL,         -- 表示用プレフィックス
    enabled BOOLEAN DEFAULT TRUE,
    expires_at TIMESTAMP NULL,
    last_used_at TIMESTAMP NULL,
    request_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

Section 68-2

キー生成・発行ロジック

SecureRandomを使って暗号学的に安全なAPIキーを生成します。発行時にのみフルキーを表示します。

vpn_key

SecureRandom生成

暗号学的に安全な乱数でキーを生成

enhanced_encryption

SHA-256ハッシュ保存

キーのハッシュ値のみをDBに保存

visibility_off

発行時1回限り表示

フルキーは発行レスポンスでのみ返却

ApiKeyService.java
import java.security.SecureRandom;
import java.security.MessageDigest;
import java.util.HexFormat;

@Service
public class ApiKeyService {

    private final ApiKeyMapper apiKeyMapper;

    public ApiKeyService(ApiKeyMapper apiKeyMapper) {
        this.apiKeyMapper = apiKeyMapper;
    }

    /**
     * 新しいAPIキーを生成して発行
     * @return 生のキー文字列（発行時のみ返す）
     */
    public String generateApiKey(Integer userId, String keyName) {
        // 1. SecureRandomで32バイトのランダム値を生成
        SecureRandom random = new SecureRandom();
        byte[] bytes = new byte[32];
        random.nextBytes(bytes);
        String rawKey = "ak_" + HexFormat.of().formatHex(bytes);

        // 2. SHA-256ハッシュ値を計算（DB保存用）
        String hashedKey = hashKey(rawKey);

        // 3. DBに保存
        ApiKey apiKey = new ApiKey();
        apiKey.setUserId(userId);
        apiKey.setKeyName(keyName);
        apiKey.setKeyValue(hashedKey);
        apiKey.setKeyPrefix(rawKey.substring(0, 11)); // "ak_1a2b3c4d"
        apiKeyMapper.insert(apiKey);

        // 4. 生のキーを返す（これが最初で最後の表示）
        return rawKey;
    }

    /**
     * APIキーを検証
     */
    public ApiKey validateKey(String rawKey) {
        String hashed = hashKey(rawKey);
        ApiKey key = apiKeyMapper.findByKeyValue(hashed);

        if (key == null || !key.isEnabled()) return null;
        if (key.getExpiresAt() != null
            && key.getExpiresAt().isBefore(LocalDateTime.now())) {
            return null;
        }

        // 最終使用日時とリクエスト数を更新
        apiKeyMapper.updateUsage(key.getId());
        return key;
    }

    private String hashKey(String rawKey) {
        try {
            MessageDigest md = MessageDigest.getInstance("SHA-256");
            byte[] hash = md.digest(rawKey.getBytes());
            return HexFormat.of().formatHex(hash);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}
ApiKeyController.java（キー管理API）
@Controller
@RequestMapping("/settings/api-keys")
public class ApiKeyController {

    private final ApiKeyService apiKeyService;

    @GetMapping
    public String list(@AuthenticationPrincipal UserDetails user,
                       Model model) {
        Integer userId = ((CustomUserDetails) user).getUserId();
        model.addAttribute("apiKeys",
            apiKeyService.findByUserId(userId));
        return "settings/api-keys";
    }

    @PostMapping("/generate")
    public String generate(@AuthenticationPrincipal UserDetails user,
                           @RequestParam String keyName,
                           RedirectAttributes redirectAttributes) {
        Integer userId = ((CustomUserDetails) user).getUserId();
        String rawKey = apiKeyService.generateApiKey(userId, keyName);

        // 発行された生キーをフラッシュメッセージで1回だけ表示
        redirectAttributes.addFlashAttribute("newKey", rawKey);
        redirectAttributes.addFlashAttribute("message",
            "APIキーを発行しました。このキーは再表示できません。");
        return "redirect:/settings/api-keys";
    }

    @PostMapping("/{id}/revoke")
    public String revoke(@PathVariable Integer id,
                         RedirectAttributes redirectAttributes) {
        apiKeyService.revokeKey(id);
        redirectAttributes.addFlashAttribute("message",
            "APIキーを無効化しました");
        return "redirect:/settings/api-keys";
    }
}
lightbulbハッシュ保存の理由

APIキーの生の値はDBに保存しません。SHA-256ハッシュのみ保存し、認証時に入力キーをハッシュ化して比較します。これはパスワードのBCryptハッシュ保存と同じ考え方です。

Section 68-3

認証フィルター実装

REST APIへのリクエストに含まれるX-API-Keyヘッダーを検証する認証フィルターを実装します。

filter_alt

OncePerRequestFilter

Spring SecurityのフィルターチェーンにAPIキー認証を追加

http

X-API-Keyヘッダー

標準的なAPIキーヘッダーで認証情報を送信

error_outline

401レスポンス

無効なキーには401 Unauthorizedを返却

ApiKeyAuthFilter.java
import jakarta.servlet.FilterChain;
import jakarta.servlet.http.*;
import org.springframework.web.filter.OncePerRequestFilter;

@Component
public class ApiKeyAuthFilter extends OncePerRequestFilter {

    private final ApiKeyService apiKeyService;
    private final UserMapper userMapper;

    public ApiKeyAuthFilter(ApiKeyService apiKeyService,
                            UserMapper userMapper) {
        this.apiKeyService = apiKeyService;
        this.userMapper = userMapper;
    }

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain filterChain) throws Exception {

        // /api/ パス以外はスキップ
        if (!request.getRequestURI().startsWith("/api/")) {
            filterChain.doFilter(request, response);
            return;
        }

        String apiKeyHeader = request.getHeader("X-API-Key");

        // ヘッダーがなければ通常のセッション認証に委譲
        if (apiKeyHeader == null || apiKeyHeader.isEmpty()) {
            filterChain.doFilter(request, response);
            return;
        }

        // APIキーを検証
        ApiKey key = apiKeyService.validateKey(apiKeyHeader);
        if (key == null) {
            response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
            response.setContentType("application/json");
            response.getWriter().write(
                "{\"error\":\"Invalid or expired API key\"}");
            return;
        }

        // 認証成功: リクエスト属性にユーザー情報を設定
        request.setAttribute("apiKeyUserId", key.getUserId());
        filterChain.doFilter(request, response);
    }
}
SecurityConfig.java（フィルター登録）
@Bean
public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
    http
        // APIキーフィルターをUsernamePasswordAuthenticationFilterの前に追加
        .addFilterBefore(apiKeyAuthFilter,
            UsernamePasswordAuthenticationFilter.class)
        .authorizeHttpRequests(auth -> auth
            .requestMatchers("/api/**").permitAll()
            .anyRequest().authenticated()
        )
        // ... 他の設定
    ;
    return http.build();
}
lightbulbフィルター順序

addFilterBeforeでAPIキーフィルターをSpring Securityのフィルターチェーン前方に配置します。APIキーが指定されていればキー認証、なければ通常のセッション認証にフォールバックします。

Section 68-4

レート制限

APIの過度な呼び出しを防ぐため、APIキーごとのレート制限（リクエスト数上限）を実装します。

speed

スライディングウィンドウ

直近1時間のリクエスト数をカウント

memory

ConcurrentHashMap管理

インメモリでリクエストカウントを高速管理

warning

429レスポンス

制限超過時にToo Many Requestsを返却

RateLimiter.java
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.atomic.AtomicInteger;

@Component
public class RateLimiter {

    private static final int MAX_REQUESTS_PER_HOUR = 1000;

    // APIキーID -> [カウント, ウィンドウ開始時刻]
    private final ConcurrentHashMap<Integer, long[]> counters
        = new ConcurrentHashMap<>();

    /**
     * リクエストを許可するか判定
     * @return true=許可, false=制限超過
     */
    public boolean allowRequest(Integer apiKeyId) {
        long now = System.currentTimeMillis();
        long windowStart = now - 3600_000; // 1時間前

        counters.compute(apiKeyId, (id, data) -> {
            if (data == null || data[1] < windowStart) {
                // 新しいウィンドウ開始
                return new long[]{ 1, now };
            }
            data[0]++;
            return data;
        });

        long[] data = counters.get(apiKeyId);
        return data[0] <= MAX_REQUESTS_PER_HOUR;
    }

    /**
     * 残りリクエスト数を取得
     */
    public int getRemainingRequests(Integer apiKeyId) {
        long[] data = counters.get(apiKeyId);
        if (data == null) return MAX_REQUESTS_PER_HOUR;
        return Math.max(0, MAX_REQUESTS_PER_HOUR - (int) data[0]);
    }
}
ApiKeyAuthFilter.java（レート制限追加版）
// APIキー検証成功後にレート制限チェックを追加
ApiKey key = apiKeyService.validateKey(apiKeyHeader);
if (key == null) {
    response.setStatus(401);
    response.getWriter().write("{\"error\":\"Invalid API key\"}");
    return;
}

// レート制限チェック
if (!rateLimiter.allowRequest(key.getId())) {
    response.setStatus(429); // Too Many Requests
    response.setContentType("application/json");
    response.setHeader("Retry-After", "3600");
    response.getWriter().write(
        "{\"error\":\"Rate limit exceeded. Try again later.\"}");
    return;
}

// レスポンスヘッダーにレート制限情報を追加
response.setHeader("X-RateLimit-Limit", "1000");
response.setHeader("X-RateLimit-Remaining",
    String.valueOf(rateLimiter.getRemainingRequests(key.getId())));

edit_note 基本プロンプト
Spring Boot 3.xのToDoアプリにAPIキー認証機能を追加してください。

【要件】
- api_keysテーブルでAPIキーを管理（SHA-256ハッシュ保存）
- SecureRandomでキーを生成、発行時のみフルキーを表示
- OncePerRequestFilterでX-API-Keyヘッダーを検証
- インメモリのレート制限（1時間あたり1000リクエスト）

【技術スタック】
- Spring Boot 3.x + Spring Security
- SHA-256（キーハッシュ化）
- ConcurrentHashMap（レート制限カウンター）
- OncePerRequestFilter
gps_fixed ステップ別プロンプト
# Step 1: APIキーテーブル
api_keysテーブルを作成してください。key_value(ハッシュ), key_prefix(表示用), user_id, expires_at, request_count列を含めてください。

# Step 2: キー生成・発行
SecureRandomで32バイトのキーを生成し、SHA-256ハッシュ値のみDBに保存してください。生キーは発行時のフラッシュメッセージでのみ表示してください。

# Step 3: 認証フィルター
OncePerRequestFilterでX-API-Keyヘッダーを検証してください。/api/パスのみ対象とし、キーがなければセッション認証にフォールバックしてください。

# Step 4: レート制限
ConcurrentHashMapで1時間あたりのリクエスト数を管理し、上限超過時は429レスポンスを返してください。X-RateLimit-Remainingヘッダーも付与してください。
