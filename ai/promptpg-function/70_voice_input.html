Section 70-1

Web Speech API基礎

ブラウザ標準のWeb Speech API（SpeechRecognition）を使い、音声でタスクを入力できる機能を実装します。サーバーサイドの変更は不要で、JavaScriptのみで実現できます。

言語設定

日本語（ja-JP）を指定して認識精度を向上

settings_voice

認識パラメータ

連続認識・中間結果の取得設定

speechRecognition.js（API初期化） 
/**
 * Web Speech API の初期化
 * SpeechRecognition はブラウザによってプレフィックスが異なる
 */
const SpeechRecognition =
    window.SpeechRecognition || window.webkitSpeechRecognition;

const recognition = new SpeechRecognition();

// --- 基本設定 ---
recognition.lang = 'ja-JP';           // 日本語
recognition.continuous = false;        // 一度の発話で停止
recognition.interimResults = true;     // 中間結果を取得
recognition.maxAlternatives = 1;       // 候補数

console.log('SpeechRecognition initialized:', recognition);
speechRecognition.js（イベントハンドラ） 
/**
 * 音声認識の各種イベント
 */
recognition.onresult = (event) => {
    const result = event.results[event.resultIndex];
    const transcript = result[0].transcript;

    if (result.isFinal) {
        console.log('確定テキスト:', transcript);
        // フォームに反映する処理をここに追加
    } else {
        console.log('中間結果:', transcript);
    }
};

recognition.onerror = (event) => {
    console.error('音声認識エラー:', event.error);
    // 'no-speech', 'audio-capture', 'not-allowed' など
};

recognition.onend = () => {
    console.log('音声認識が終了しました');
};

recognition.onstart = () => {
    console.log('音声認識を開始しました');
};

voiceToggle.js（開始/停止制御） 
let isRecording = false;

/**
 * 音声認識の開始/停止をトグル
 */
function toggleVoice() {
    const btn = document.getElementById('voiceBtn');
    const status = document.getElementById('voiceStatus');

    if (isRecording) {
        // --- 停止 ---
        recognition.stop();
        btn.classList.remove('recording');
        btn.querySelector('.material-icons').textContent = 'mic';
        status.textContent = '';
        isRecording = false;
    } else {
        // --- 開始 ---
        try {
            recognition.start();
            btn.classList.add('recording');
            btn.querySelector('.material-icons').textContent = 'mic_off';
            status.textContent = '音声を聞いています...';
            isRecording = true;
        } catch (e) {
            status.textContent = 'エラー: ' + e.message;
        }
    }
}

// 認識終了時にUIをリセット
recognition.onend = () => {
    const btn = document.getElementById('voiceBtn');
    btn.classList.remove('recording');
    btn.querySelector('.material-icons').textContent = 'mic';
    document.getElementById('voiceStatus').textContent = '';
    isRecording = false;
};
lightbulbマイク権限について

voiceFormReflect.js（フォーム反映） 
/**
 * 音声認識結果をフォームに反映する
 */
recognition.onresult = (event) => {
    const titleInput = document.getElementById('title');
    const status = document.getElementById('voiceStatus');
    const result = event.results[event.resultIndex];
    const transcript = result[0].transcript;

    if (result.isFinal) {
        // --- 確定テキスト → フォームに設定 ---
        titleInput.value = transcript;
        status.textContent = '入力完了: 「' + transcript + '」';
        status.style.color = '#22c55e';

        // 3秒後にステータスをクリア
        setTimeout(() => {
            status.textContent = '';
            status.style.color = '#64748b';
        }, 3000);
    } else {
        // --- 中間結果 → プレビュー表示 ---
        status.textContent = '認識中: ' + transcript;
        status.style.color = '#f59e0b';
    }
};
voiceAppendMode.js（追記モード対応） 
/**
 * 追記モード: 既存テキストの後ろに音声認識結果を追加
 * メモ欄など長いテキストフィールド向け
 */
function setupVoiceForTextarea(textareaId, appendMode = true) {
    const textarea = document.getElementById(textareaId);

    recognition.onresult = (event) => {
        const result = event.results[event.resultIndex];
        const transcript = result[0].transcript;

        if (result.isFinal) {
            if (appendMode && textarea.value.length > 0) {
                // 既存テキストの末尾に追記
                textarea.value += ' ' + transcript;
            } else {
                // 上書きモード
                textarea.value = transcript;
            }
        }
    };
}

voiceMultiField.js（複数フィールド対応） 
/**
 * フォーカス中のフィールドに音声入力する
 * どのinput/textareaにフォーカスがあるかを自動判定
 */
let activeField = null;

// フォーカスされたフィールドを追跡
document.querySelectorAll('input[type="text"], textarea')
    .forEach(field => {
        field.addEventListener('focus', () => {
            activeField = field;
        });
    });

recognition.onresult = (event) => {
    if (!activeField) return;

    const result = event.results[event.resultIndex];
    const transcript = result[0].transcript;

    if (result.isFinal) {
        activeField.value = transcript;
        // inputイベントを発火（フレームワーク連携用）
        activeField.dispatchEvent(new Event('input', { bubbles: true }));
    }
};
lightbulbinputイベントの発火

JavaScriptでvalueを直接書き換えた場合、inputイベントは自動で発火しません。
Thymeleafやフロントエンドフレームワークと連携するため、dispatchEventで明示的にイベントを発火させます。

Section 70-4

対応ブラウザ判定

Web Speech APIは全てのブラウザで利用できるわけではありません。対応ブラウザの判定を行い、非対応ブラウザではマイクボタンを非表示にするフォールバックを実装します。

browserDetect.js（対応ブラウザ判定） 
/**
 * Web Speech API の対応状況をチェックし、
 * 非対応の場合はマイクボタンを非表示にする
 */
document.addEventListener('DOMContentLoaded', () => {
    const voiceBtn = document.getElementById('voiceBtn');

    const isSupported =
        'SpeechRecognition' in window ||
        'webkitSpeechRecognition' in window;

    if (!isSupported) {
        // --- 非対応ブラウザ ---
        if (voiceBtn) {
            voiceBtn.style.display = 'none';
        }
        console.warn('このブラウザはWeb Speech APIに対応していません');
        return;
    }

    // --- 対応ブラウザ → 初期化 ---
    initSpeechRecognition();
});
voiceInput.js（統合モジュール） 
/**
 * 音声入力モジュール（統合版）
 * ブラウザ判定 → 初期化 → 開始/停止 → フォーム反映
 */
const VoiceInput = (() => {
    let recognition = null;
    let isRecording = false;

    function init(targetInputId) {
        const SR = window.SpeechRecognition
                || window.webkitSpeechRecognition;
        if (!SR) {
            console.warn('Web Speech API is not supported');
            return false;
        }

        recognition = new SR();
        recognition.lang = 'ja-JP';
        recognition.continuous = false;
        recognition.interimResults = true;

        const input = document.getElementById(targetInputId);
        const status = document.getElementById('voiceStatus');

        recognition.onresult = (event) => {
            const result = event.results[event.resultIndex];
            const text = result[0].transcript;
            if (result.isFinal) {
                input.value = text;
                input.dispatchEvent(new Event('input', { bubbles: true }));
                if (status) status.textContent = '完了: 「' + text + '」';
            } else {
                if (status) status.textContent = '認識中: ' + text;
            }
        };

        recognition.onend = () => { isRecording = false; updateUI(); };
        recognition.onerror = (e) => {
            console.error('Voice error:', e.error);
            isRecording = false;
            updateUI();
        };

        return true;
    }

    function toggle() {
        if (!recognition) return;
        if (isRecording) {
            recognition.stop();
        } else {
            recognition.start();
        }
        isRecording = !isRecording;
        updateUI();
    }

    function updateUI() {
        const btn = document.getElementById('voiceBtn');
        if (!btn) return;
        btn.classList.toggle('recording', isRecording);
        btn.querySelector('.material-icons').textContent =
            isRecording ? 'mic_off' : 'mic';
    }

    return { init, toggle };
})();

// 使用例
// VoiceInput.init('title');
// ボタンの onclick="VoiceInput.toggle()"

# Step 1: ブラウザ判定
Web Speech API の対応ブラウザを判定し、非対応ブラウザではマイクボタンを非表示にするJavaScriptを作成してください。

# Step 2: 音声認識の初期化
SpeechRecognition を日本語（ja-JP）で初期化し、中間結果を取得できるよう設定してください。

# Step 3: 開始/停止UI
マイクボタンをクリックで音声認識の開始/停止をトグルし、録音中はパルスアニメーションを表示するUIを作成してください。

# Step 4: フォーム反映
音声認識の確定結果をタスク名のinputフィールドに反映し、中間結果はステータス表示欄にリアルタイム表示してください。
