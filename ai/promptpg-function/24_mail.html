Section 23-2

メール送信の実装

MailService.java（シンプルなメール送信） 
import org.springframework.beans.factory.annotation.Value;
import org.springframework.mail.SimpleMailMessage;
import org.springframework.mail.javamail.JavaMailSender;
import org.springframework.stereotype.Service;

@Service
public class MailService {

    private final JavaMailSender mailSender;

    @Value("${app.mail.from}")
    private String fromAddress;

    public MailService(JavaMailSender mailSender) {
        this.mailSender = mailSender;
    }

    /**
     * テキストメールを送信
     */
    public void sendSimpleMail(String to, String subject, String text) {
        SimpleMailMessage message = new SimpleMailMessage();
        message.setFrom(fromAddress);
        message.setTo(to);
        message.setSubject(subject);
        message.setText(text);

        mailSender.send(message);
    }

    /**
     * ToDo作成通知
     */
    public void sendTodoCreatedNotification(String to, Todo todo) {
        String subject = "【ToDo作成】" + todo.getTitle();
        String text = String.format(
            "ToDoが作成されました。\n\n" +
            "タイトル: %s\n" +
            "説明: %s\n" +
            "期限: %s\n\n" +
            "ToDoアプリでご確認ください。",
            todo.getTitle(),
            todo.getDescription(),
            todo.getDeadline() != null ? todo.getDeadline().toString() : "未設定"
        );

        sendSimpleMail(to, subject, text);
    }
}
MailService.java（HTMLメール送信） 
import org.springframework.mail.javamail.MimeMessageHelper;
import jakarta.mail.MessagingException;
import jakarta.mail.internet.MimeMessage;

@Service
public class MailService {

    /**
     * HTMLメールを送信
     */
    public void sendHtmlMail(String to, String subject, String htmlContent)
            throws MessagingException {

        MimeMessage message = mailSender.createMimeMessage();
        MimeMessageHelper helper = new MimeMessageHelper(message, true, "UTF-8");

        helper.setFrom(fromAddress);
        helper.setTo(to);
        helper.setSubject(subject);
        helper.setText(htmlContent, true);  // true = HTML形式

        mailSender.send(message);
    }

    /**
     * 期限リマインダーメール（HTML）
     */
    public void sendDeadlineReminder(String to, Todo todo) throws MessagingException {
        String subject = "【期限間近】" + todo.getTitle();

        String html = """
            <html>
            <body style="font-family: sans-serif;">
                <h2 style="color: #e74c3c;">期限が近づいています</h2>
                <div style="background: #f5f5f5; padding: 20px; border-radius: 8px;">
                    <p><strong>タイトル:</strong> %s</p>
                    <p><strong>期限:</strong> %s</p>
                    <p><strong>残り:</strong> あと1日</p>
                </div>
                <p style="margin-top: 20px;">
                    <a href="http://localhost:8080/todo/%d"
                       style="background: #3498db; color: white; padding: 10px 20px;
                              text-decoration: none; border-radius: 4px;">
                        ToDoを確認する
                    </a>
                </p>
            </body>
            </html>
            """.formatted(todo.getTitle(), todo.getDeadline(), todo.getId());

        sendHtmlMail(to, subject, html);
    }
}
Thymeleafテンプレートでメール生成 
import org.thymeleaf.TemplateEngine;
import org.thymeleaf.context.Context;

@Service
public class MailService {

    private final TemplateEngine templateEngine;

    /**
     * Thymeleafテンプレートを使用したメール送信
     */
    public void sendTemplatedMail(String to, String subject,
            String templateName, Map<String, Object> variables)
            throws MessagingException {

        // Thymeleafコンテキストに変数をセット
        Context context = new Context();
        context.setVariables(variables);

        // テンプレートからHTMLを生成
        String htmlContent = templateEngine.process(templateName, context);

        sendHtmlMail(to, subject, htmlContent);
    }
}

// 使用例
Map<String, Object> vars = new HashMap<>();
vars.put("todo", todo);
vars.put("user", user);
mailService.sendTemplatedMail(
    user.getEmail(),
    "期限リマインダー",
    "mail/reminder",  // templates/mail/reminder.html
    vars
);
Section 23-3

期限リマインダー機能

ReminderScheduler.java（定期実行） 
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

@Component
public class ReminderScheduler {

    private final TodoMapper todoMapper;
    private final UserMapper userMapper;
    private final MailService mailService;

    /**
     * 毎日朝9時に期限リマインダーを送信
     * cron: 秒 分 時 日 月 曜日
     */
    @Scheduled(cron = "0 0 9 * * *")
    public void sendDailyReminders() {
        System.out.println("期限リマインダー処理開始: " + LocalDateTime.now());

        // 明日が期限のToDoを取得
        LocalDate tomorrow = LocalDate.now().plusDays(1);
        List<Todo> upcomingTodos = todoMapper.findByDeadline(tomorrow);

        for (Todo todo : upcomingTodos) {
            try {
                User user = userMapper.findById(todo.getUserId());
                mailService.sendDeadlineReminder(user.getEmail(), todo);
                System.out.println("リマインダー送信: " + todo.getTitle());
            } catch (Exception e) {
                System.err.println("送信失敗: " + todo.getId());
            }
        }

        System.out.println("期限リマインダー処理完了");
    }

    /**
     * 1時間ごとに期限切れをチェック
     */
    @Scheduled(fixedRate = 3600000)  // 1時間 = 3600000ミリ秒
    public void checkOverdueTodos() {
        List<Todo> overdueTodos = todoMapper.findOverdue(LocalDate.now());

        for (Todo todo : overdueTodos) {
            if (!todo.isNotified()) {
                // 期限切れ通知を送信
                sendOverdueNotification(todo);
                // 通知済みフラグを更新
                todoMapper.updateNotified(todo.getId(), true);
            }
        }
    }
}

オプション説明例
cronCron式で指定"0 0 9 * * *"（毎日9時）
fixedRate固定間隔（ミリ秒）60000（1分ごと）
fixedDelay前回完了からの間隔60000

非同期メール送信 
import org.springframework.scheduling.annotation.Async;

@Service
public class MailService {

    /**
     * 非同期でメール送信（ユーザーを待たせない）
     */
    @Async
    public void sendMailAsync(String to, String subject, String text) {
        try {
            sendSimpleMail(to, subject, text);
            System.out.println("メール送信完了: " + to);
        } catch (Exception e) {
            System.err.println("メール送信失敗: " + e.getMessage());
            // 失敗時はリトライキューに入れるなど
        }
    }
}

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにメール送信通知機能を追加してください。

【要件】
- spring-boot-starter-mailの依存関係を追加
- application.propertiesでSMTPサーバー設定（Gmail対応）
- JavaMailSenderでテキストメール・HTMLメール送信
- MimeMessageHelperでHTMLメールとマルチパートメール対応
- ThymeleafテンプレートでHTMLメール本文を生成
- @Scheduledで定期実行（毎日9時に期限リマインダー送信）
- @Asyncで非同期メール送信（ユーザーを待たせない）

【技術スタック】
- Spring Boot 3.x
- JavaMailSender
- Thymeleaf（メールテンプレート）
- Spring Scheduling

# Step 1: メール設定
pom.xmlにspring-boot-starter-mailを追加し、

# Step 2: MailServiceの作成

# Step 3: HTMLメール対応

# Step 4: 定期実行リマインダー
@EnableSchedulingを追加し、@Scheduledで毎日9時に
