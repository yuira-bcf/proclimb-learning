Section 37-1

依存関係テーブル設計

タスク間の依存関係を管理するための中間テーブルを設計します。「タスクAが完了しないとタスクBに着手できない」という前後関係を表現します。

storage

中間テーブル

task_dependenciesテーブルで多対多の依存関係を管理

account_tree

前提・後続の関係

predecessor（前提タスク）とsuccessor（後続タスク）

key

複合主キー

predecessor_id + successor_idで一意制約

schema.sql（依存関係テーブル） 
CREATE TABLE task_dependencies (
    predecessor_id INT NOT NULL,
    successor_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (predecessor_id, successor_id),
    FOREIGN KEY (predecessor_id) REFERENCES todo(id) ON DELETE CASCADE,
    FOREIGN KEY (successor_id) REFERENCES todo(id) ON DELETE CASCADE,
    -- 自己参照の防止
    CHECK (predecessor_id != successor_id)
);

@Mapper
public interface TaskDependencyMapper {

    void insert(@Param("predecessorId") Integer predecessorId,
                @Param("successorId") Integer successorId);

    void delete(@Param("predecessorId") Integer predecessorId,
                @Param("successorId") Integer successorId);

    /** 指定タスクの前提タスク一覧を取得 */
    List<Todo> findPredecessors(@Param("taskId") Integer taskId);

    /** 指定タスクの後続タスク一覧を取得 */
    List<Todo> findSuccessors(@Param("taskId") Integer taskId);

    /** 全ての後続タスクIDを再帰的に取得（循環チェック用） */
    List<Integer> findAllSuccessorIds(@Param("taskId") Integer taskId);
}

Section 37-2

循環依存チェックロジック

A→B→C→Aのような循環依存を防ぐチェックロジックを実装します。依存関係を追加する前に、グラフ探索で循環が発生しないことを検証します。

loop

循環検出アルゴリズム

BFS/DFSで依存グラフを走査して循環を検出

verified

事前バリデーション

依存追加前に循環チェックを実行

error

エラーハンドリング

循環検出時は分かりやすいエラーメッセージを返す

TaskDependencyService.java（循環チェック付きサービス） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.*;

@Service
public class TaskDependencyService {

    private final TaskDependencyMapper dependencyMapper;

    public TaskDependencyService(TaskDependencyMapper dependencyMapper) {
        this.dependencyMapper = dependencyMapper;
    }

    /**
     * 依存関係を追加（循環チェック付き）
     * @param predecessorId 前提タスクID
     * @param successorId   後続タスクID
     */
    @Transactional
    public void addDependency(Integer predecessorId,
                              Integer successorId) {
        // 自己参照チェック
        if (predecessorId.equals(successorId)) {
            throw new IllegalArgumentException(
                "タスクは自分自身に依存できません");
        }

        // 循環依存チェック
        if (wouldCreateCycle(predecessorId, successorId)) {
            throw new IllegalArgumentException(
                "循環依存が発生するため追加できません");
        }

        dependencyMapper.insert(predecessorId, successorId);
    }

    /**
     * BFS（幅優先探索）で循環依存を検出
     * predecessorの後続チェーンにsuccessorが含まれる場合は循環
     */
    private boolean wouldCreateCycle(Integer predecessorId,
                                     Integer successorId) {
        // successorの後続を辿ってpredecessorに到達するか確認
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        queue.add(successorId);

        while (!queue.isEmpty()) {
            Integer current = queue.poll();
            if (current.equals(predecessorId)) {
                return true;  // 循環検出
            }
            if (visited.contains(current)) continue;
            visited.add(current);

            // currentの後続タスクをキューに追加
            List<Todo> successors =
                dependencyMapper.findSuccessors(current);
            for (Todo s : successors) {
                queue.add(s.getId());
            }
        }
        return false;
    }

    /**
     * 依存関係を削除
     */
    @Transactional
    public void removeDependency(Integer predecessorId,
                                 Integer successorId) {
        dependencyMapper.delete(predecessorId, successorId);
    }

    /**
     * タスクが着手可能か判定（全前提タスクが完了済み）
     */
    public boolean isReadyToStart(Integer taskId) {
        List<Todo> predecessors =
            dependencyMapper.findPredecessors(taskId);
        return predecessors.stream()
            .allMatch(t -> t.isCompleted());
    }
}
TaskDependencyController.java 
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

@RestController
@RequestMapping("/api/tasks/{taskId}/dependencies")
public class TaskDependencyController {

    private final TaskDependencyService service;

    public TaskDependencyController(TaskDependencyService service) {
        this.service = service;
    }

    @PostMapping
    public ResponseEntity<?> addDependency(
            @PathVariable Integer taskId,
            @RequestBody Map<String, Integer> body) {
        try {
            Integer predecessorId = body.get("predecessorId");
            service.addDependency(predecessorId, taskId);
            return ResponseEntity.ok(Map.of(
                "message", "依存関係を追加しました"));
        } catch (IllegalArgumentException e) {
            return ResponseEntity.badRequest()
                .body(Map.of("error", e.getMessage()));
        }
    }

    @DeleteMapping("/{predecessorId}")
    public ResponseEntity<?> removeDependency(
            @PathVariable Integer taskId,
            @PathVariable Integer predecessorId) {
        service.removeDependency(predecessorId, taskId);
        return ResponseEntity.ok(Map.of(
            "message", "依存関係を解除しました"));
    }
}
warningパフォーマンスに注意

BFS循環チェックはタスク数が多い場合にクエリ回数が増加します。大規模なプロジェクトでは、依存グラフをメモリ上にキャッシュする、またはSQL再帰CTEを使った一括取得を検討してください。

const taskId = document.getElementById('task-id').value;

async function addDependency() {
    const select = document.getElementById('predecessor-select');
    const predecessorId = select.value;
    if (!predecessorId) return;

    try {
        const res = await fetch(`/api/tasks/${taskId}/dependencies`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ predecessorId: parseInt(predecessorId) })
        });
        const data = await res.json();

        if (res.ok) {
            location.reload();
        } else {
            alert(data.error);
        }
    } catch (err) {
        alert('通信エラーが発生しました');
    }
}

DependencyGraphController.java（グラフデータAPI） 
import org.springframework.web.bind.annotation.*;
import java.util.*;

@RestController
@RequestMapping("/api/dependency-graph")
public class DependencyGraphController {

    private final TodoMapper todoMapper;
    private final TaskDependencyMapper dependencyMapper;

    public DependencyGraphController(TodoMapper todoMapper,
            TaskDependencyMapper dependencyMapper) {
        this.todoMapper = todoMapper;
        this.dependencyMapper = dependencyMapper;
    }

    @GetMapping
    public Map<String, Object> getGraph() {
        List<Todo> tasks = todoMapper.findAll();

        // ノード
        List<Map<String, Object>> nodes = tasks.stream()
            .map(t -> Map.<String, Object>of(
                "id", t.getId(),
                "title", t.getTitle(),
                "completed", t.isCompleted()
            )).toList();

        // エッジ
        List<Map<String, Integer>> edges = new ArrayList<>();
        for (Todo task : tasks) {
            List<Todo> successors =
                dependencyMapper.findSuccessors(task.getId());
            for (Todo s : successors) {
                edges.add(Map.of(
                    "from", task.getId(),
                    "to", s.getId()
                ));
            }
        }

        return Map.of("nodes", nodes, "edges", edges);
    }
}
dependency-graph.js（SVGグラフ描画） 
document.addEventListener('DOMContentLoaded', async () => {
    const container = document.getElementById('dependency-graph');
    if (!container) return;

    const res = await fetch('/api/dependency-graph');
    const { nodes, edges } = await res.json();

    const svg = document.createElementNS(
        'http://www.w3.org/2000/svg', 'svg');
    svg.setAttribute('viewBox', '0 0 800 400');
    svg.style.width = '100%';
    svg.style.maxHeight = '400px';

    // ノード配置（簡易レイアウト）
    const nodePositions = {};
    const cols = Math.ceil(Math.sqrt(nodes.length));
    nodes.forEach((node, i) => {
        const x = 80 + (i % cols) * 160;
        const y = 60 + Math.floor(i / cols) * 100;
        nodePositions[node.id] = { x, y };

        // 矩形ノード
        const rect = document.createElementNS(
            'http://www.w3.org/2000/svg', 'rect');
        rect.setAttribute('x', x - 60);
        rect.setAttribute('y', y - 20);
        rect.setAttribute('width', 120);
        rect.setAttribute('height', 40);
        rect.setAttribute('rx', 8);
        rect.setAttribute('fill',
            node.completed ? '#dcfce7' : '#fef3c7');
        rect.setAttribute('stroke',
            node.completed ? '#22c55e' : '#f59e0b');
        rect.setAttribute('stroke-width', 2);
        svg.appendChild(rect);

        // タイトルテキスト
        const text = document.createElementNS(
            'http://www.w3.org/2000/svg', 'text');
        text.setAttribute('x', x);
        text.setAttribute('y', y + 5);
        text.setAttribute('text-anchor', 'middle');
        text.setAttribute('font-size', '12');
        text.textContent = node.title.length > 10
            ? node.title.substring(0, 10) + '...'
            : node.title;
        svg.appendChild(text);
    });

    // エッジ（矢印線）
    edges.forEach(edge => {
        const from = nodePositions[edge.from];
        const to = nodePositions[edge.to];
        if (!from || !to) return;

        const line = document.createElementNS(
            'http://www.w3.org/2000/svg', 'line');
        line.setAttribute('x1', from.x + 60);
        line.setAttribute('y1', from.y);
        line.setAttribute('x2', to.x - 60);
        line.setAttribute('y2', to.y);
        line.setAttribute('stroke', '#94a3b8');
        line.setAttribute('stroke-width', 2);
        line.setAttribute('marker-end', 'url(#arrowhead)');
        svg.appendChild(line);
    });

    // 矢印マーカー定義
    const defs = document.createElementNS(
        'http://www.w3.org/2000/svg', 'defs');
    defs.innerHTML = `
        <marker id="arrowhead" markerWidth="10" markerHeight="7"
                refX="10" refY="3.5" orient="auto">
            <polygon points="0 0, 10 3.5, 0 7"
                     fill="#94a3b8"/>
        </marker>`;
    svg.prepend(defs);

    container.appendChild(svg);
});
