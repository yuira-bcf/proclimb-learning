<select id="getRanking" resultType="UserRankingDto">
    SELECT
        RANK() OVER (ORDER BY COUNT(t.id) DESC) AS rank,
        u.id AS userId,
        u.username,
        COUNT(t.id) AS completedCount
    FROM users u
    LEFT JOIN todo t ON t.user_id = u.id
        AND t.done = TRUE
        <if test="startDate != null">
            AND t.completed_at >= #{startDate}
        </if>
        <if test="endDate != null">
            AND t.completed_at < #{endDate}
        </if>
    GROUP BY u.id, u.username
    ORDER BY completedCount DESC
    LIMIT 10
</select>
lightbulbRANK() vs ROW_NUMBER()

RANK()は同数の場合に同じ順位を付けます（1位, 1位, 3位...）。ROW_NUMBER()は必ず連番になるため、同率順位が発生しません。ランキングにはRANK()が適切です。

RankingService.java 
import org.springframework.stereotype.Service;
import java.time.DayOfWeek;
import java.time.LocalDate;
import java.time.temporal.TemporalAdjusters;
import java.util.List;

@Service
public class RankingService {

    private final RankingMapper rankingMapper;

    public RankingService(RankingMapper rankingMapper) {
        this.rankingMapper = rankingMapper;
    }

    /**
     * 指定期間のランキングを取得
     * @param period "week", "month", or "all"
     */
    public List<UserRankingDto> getRanking(String period) {
        LocalDate startDate = null;
        LocalDate endDate = null;

        switch (period != null ? period : "all") {
            case "week":
                startDate = LocalDate.now()
                    .with(TemporalAdjusters.previousOrSame(DayOfWeek.MONDAY));
                endDate = startDate.plusWeeks(1);
                break;
            case "month":
                startDate = LocalDate.now()
                    .with(TemporalAdjusters.firstDayOfMonth());
                endDate = startDate.plusMonths(1);
                break;
            default: // "all"
                break;
        }

        return rankingMapper.getRanking(startDate, endDate);
    }
}
RankingController.java 
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;

@Controller
public class RankingController {

    private final RankingService rankingService;

    public RankingController(RankingService rankingService) {
        this.rankingService = rankingService;
    }

    @GetMapping("/ranking")
    public String ranking(
            @RequestParam(defaultValue = "all") String period,
            Model model) {

        model.addAttribute("rankings", rankingService.getRanking(period));
        model.addAttribute("currentPeriod", period);
        return "ranking";
    }
}
lightbulbTemporalAdjusters

TemporalAdjusters.previousOrSame(DayOfWeek.MONDAY)で「今週の月曜日」を取得できます。firstDayOfMonth()で月初を取得するなど、日付計算に便利なユーティリティです。

<!-- MyBatisの動的SQLで期間条件を切り替え -->
<select id="getRanking" resultType="UserRankingDto">
    SELECT
        RANK() OVER (ORDER BY COUNT(t.id) DESC) AS rank,
        u.id AS userId,
        u.username,
        COUNT(t.id) AS completedCount
    FROM users u
    LEFT JOIN todo t ON t.user_id = u.id AND t.done = TRUE
    <where>
        <if test="startDate != null">
            AND t.completed_at >= #{startDate}
        </if>
        <if test="endDate != null">

Ajax版タブ切り替え（SPA風） 
// ページ遷移なしでランキングを切り替え（REST API版）
document.querySelectorAll('.tab').forEach(function(tab) {
    tab.addEventListener('click', function(e) {
        e.preventDefault();
        const period = this.dataset.period;

        // アクティブタブの切替
        document.querySelectorAll('.tab').forEach(function(t) {
            t.classList.remove('active');
        });
        this.classList.add('active');

        // APIからランキングデータを取得
        fetch('/api/ranking?period=' + period)
            .then(function(res) { return res.json(); })
            .then(function(data) {
                renderRanking(data);
            });
    });
});

function renderRanking(rankings) {
    var container = document.querySelector('.ranking-list');
    var maxCount = rankings.length > 0 ? rankings[0].completedCount : 1;

    container.innerHTML = rankings.map(function(r) {
        var barWidth = (r.completedCount / maxCount * 100);
        var medalHtml = '';
        if (r.rank === 1) medalHtml = '<span class="medal gold"><span class="material-icons">emoji_events</span></span>';
        else if (r.rank === 2) medalHtml = '<span class="medal silver"><span class="material-icons">emoji_events</span></span>';
        else if (r.rank === 3) medalHtml = '<span class="medal bronze"><span class="material-icons">emoji_events</span></span>';
        else medalHtml = '<span class="rank-number">' + r.rank + '</span>';

        return '<div class="ranking-item ' + (r.rank <= 3 ? 'top-' + r.rank : '') + '">' +
            '<div class="ranking-rank">' + medalHtml + '</div>' +
            '<div class="ranking-user"><div class="avatar">' +
                r.username.charAt(0).toUpperCase() +
            '</div><span>' + r.username + '</span></div>' +
            '<div class="ranking-bar-wrapper">' +
                '<div class="ranking-bar" style="width:' + barWidth + '%"></div>' +
                '<span class="ranking-count">' + r.completedCount + '件</span>' +
            '</div></div>';
    }).join('');
}

# Step 1: ランキング集計SQL
RANK() OVER (ORDER BY COUNT DESC)を使って、ユーザーごとのタスク完了数ランキングを取得するMyBatis SQLを作成してください。期間指定のための動的WHERE句も追加してください。

# Step 2: ランキングサービス
期間パラメータ（week/month/all）に応じて開始日・終了日を計算し、Mapperを呼び出すサービスクラスを作成してください。

# Step 3: ランキング表示UI
Thymeleafで1〜3位にメダル表示、プログレスバーでスコア視覚化、ユーザー名頭文字のアバターを含むランキング画面を作成してください。

# Step 4: 期間フィルタタブ
