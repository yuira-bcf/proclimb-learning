Section 60-1

Webhook設定テーブル

外部サービスと連携するためのWebhook設定をDBに保存します。URL、イベント種別、有効/無効の状態を管理します。

Webhookとは

イベント発生時に外部URLへHTTPリクエストを送信する仕組み

schema.sql（Webhookテーブル）
CREATE TABLE webhooks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    url VARCHAR(500) NOT NULL,
    event_type VARCHAR(50) NOT NULL,  -- TASK_CREATED, TASK_UPDATED, TASK_COMPLETED, TASK_DELETED
    is_active BOOLEAN DEFAULT TRUE,
    secret VARCHAR(100),              -- HMAC署名用シークレット
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE webhook_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    webhook_id INT NOT NULL,
    event_type VARCHAR(50) NOT NULL,
    payload TEXT,
    response_status INT,
    response_body TEXT,
    sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (webhook_id) REFERENCES webhooks(id)
);

public class Webhook {

lightbulbWebhookログの重要性

webhook_logsテーブルにリクエスト/レスポンスを記録することで、デバッグや障害調査が容易になります。外部サービスが応答しない場合のリトライ判断にも使えます。

// --- TodoEvent.java ---
public class TodoEvent {
    private final WebhookEvent eventType;
    private final Todo todo;
    private final Integer userId;

    public TodoEvent(WebhookEvent eventType, Todo todo, Integer userId) {
        this.eventType = eventType;
        this.todo = todo;
        this.userId = userId;
    }
    // getter
}

// --- TodoService.java（イベント発行を追加）---
@Service
public class TodoService {

    private final ApplicationEventPublisher eventPublisher;
    // ... 他のフィールド

    public void createTodo(Todo todo, Integer userId) {
        todoMapper.insert(todo);
        // Webhookイベントを発行
        eventPublisher.publishEvent(
            new TodoEvent(WebhookEvent.TASK_CREATED, todo, userId));
    }

    public void updateTodo(Todo todo, Integer userId) {
        todoMapper.update(todo);
        eventPublisher.publishEvent(
            new TodoEvent(WebhookEvent.TASK_UPDATED, todo, userId));
    }

    public void completeTodo(Integer id, Integer userId) {
        Todo todo = todoMapper.findById(id);
        todo.setStatus("DONE");
        todoMapper.update(todo);
        eventPublisher.publishEvent(
            new TodoEvent(WebhookEvent.TASK_COMPLETED, todo, userId));
    }
}
WebhookEventListener.java（イベントリスナー）
import org.springframework.context.event.EventListener;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Component;
import java.util.List;

@Component
public class WebhookEventListener {

    private final WebhookMapper webhookMapper;
    private final WebhookSender webhookSender;

    public WebhookEventListener(WebhookMapper webhookMapper,
                                 WebhookSender webhookSender) {
        this.webhookMapper = webhookMapper;
        this.webhookSender = webhookSender;
    }

    @Async
    @EventListener
    public void handleTodoEvent(TodoEvent event) {
        // このユーザーの有効なWebhook設定を取得
        List<Webhook> hooks = webhookMapper.findByUserAndEvent(
            event.getUserId(),
            event.getEventType().name());

        for (Webhook hook : hooks) {
            webhookSender.send(hook, event);
        }
    }
}
lightbulb非同期処理の重要性

Webhook送信は@Asyncで非同期化しています。外部URLへのHTTPリクエストは遅延やタイムアウトの可能性があるため、メインのビジネスロジックをブロックしないことが重要です。

RestTemplateでHTTP送信

RestTemplateを使って登録されたURLにJSON形式のイベントデータをPOST送信します。HMAC署名による検証機能も実装します。

send

HTTP POST送信

RestTemplateでJSONペイロードを外部URLに送信

lock

HMAC署名

シークレットキーでペイロードに署名して真正性を保証

history

送信ログ記録

リクエスト/レスポンスをwebhook_logsに記録

WebhookSender.java（HTTP送信）
import com.fasterxml.jackson.databind.ObjectMapper;
import org.springframework.http.*;
import org.springframework.stereotype.Component;
import org.springframework.web.client.RestTemplate;
import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;
import java.util.*;

@Component
public class WebhookSender {

    private final RestTemplate restTemplate;
    private final ObjectMapper objectMapper;
    private final WebhookLogMapper logMapper;

    public WebhookSender(RestTemplate restTemplate,
                          ObjectMapper objectMapper,
                          WebhookLogMapper logMapper) {
        this.restTemplate = restTemplate;
        this.objectMapper = objectMapper;
        this.logMapper = logMapper;
    }

    public void send(Webhook hook, TodoEvent event) {
        try {
            // ペイロードを構築
            Map<String, Object> payload = new HashMap<>();
            payload.put("event", event.getEventType().name());
            payload.put("timestamp", System.currentTimeMillis());
            payload.put("data", Map.of(
                "id", event.getTodo().getId(),
                "title", event.getTodo().getTitle(),
                "status", event.getTodo().getStatus()
            ));

            String json = objectMapper.writeValueAsString(payload);

            // HTTPヘッダー
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_JSON);

            // HMAC署名（シークレットが設定されている場合）
            if (hook.getSecret() != null) {
                String signature = hmacSha256(json, hook.getSecret());
                headers.set("X-Webhook-Signature", signature);
            }

            HttpEntity<String> request = new HttpEntity<>(json, headers);

            // 送信（タイムアウト5秒）
            ResponseEntity<String> response = restTemplate.exchange(
                hook.getUrl(), HttpMethod.POST, request, String.class);

            // ログ記録
            saveLog(hook.getId(), event.getEventType().name(),
                json, response.getStatusCodeValue(),
                response.getBody());

        } catch (Exception e) {
            saveLog(hook.getId(), event.getEventType().name(),
                null, 0, e.getMessage());
        }
    }

    private String hmacSha256(String data, String secret)
            throws Exception {
        Mac mac = Mac.getInstance("HmacSHA256");
        mac.init(new SecretKeySpec(
            secret.getBytes(), "HmacSHA256"));
        byte[] hash = mac.doFinal(data.getBytes());
        return Base64.getEncoder().encodeToString(hash);
    }

    private void saveLog(Integer webhookId, String event,
                          String payload, int status, String body) {
        WebhookLog log = new WebhookLog();
        log.setWebhookId(webhookId);
        log.setEventType(event);
        log.setPayload(payload);
        log.setResponseStatus(status);
        log.setResponseBody(body);
        logMapper.insert(log);
    }
}
lightbulbHMAC署名とは

受信側がWebhookリクエストの送信元を検証するための仕組みです。X-Webhook-SignatureヘッダーにHMAC-SHA256署名を含め、受信側は同じシークレットで署名を再計算して一致を確認します。

WebhookSettingsController.java
@Controller
@RequestMapping("/settings/webhooks")
public class WebhookSettingsController {

    private final WebhookMapper webhookMapper;

    public WebhookSettingsController(WebhookMapper webhookMapper) {
        this.webhookMapper = webhookMapper;
    }

    @GetMapping
    public String list(@AuthenticationPrincipal UserDetails user,
                       Model model) {
        model.addAttribute("webhooks",
            webhookMapper.findByUserId(getUserId(user)));
        return "settings/webhooks";
    }

    @PostMapping
    public String create(@RequestParam String url,
                         @RequestParam String eventType,
                         @RequestParam(required = false) String secret,
                         @AuthenticationPrincipal UserDetails user,
                         RedirectAttributes ra) {
        Webhook hook = new Webhook();
        hook.setUserId(getUserId(user));
        hook.setUrl(url);
        hook.setEventType(eventType);
        hook.setSecret(secret);
        hook.setActive(true);
        webhookMapper.insert(hook);
        ra.addFlashAttribute("message", "Webhookを追加しました");
        return "redirect:/settings/webhooks";
    }

    @PostMapping("/{id}/delete")
    public String delete(@PathVariable Integer id, RedirectAttributes ra) {
        webhookMapper.deleteById(id);
        ra.addFlashAttribute("message", "Webhookを削除しました");
        return "redirect:/settings/webhooks";
    }
}

# Step 1: テーブル設計
