RegisterForm.java（登録フォーム） 
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Size;

public class RegisterForm {

    @NotBlank(message = "ユーザー名は必須です")
    @Size(min = 3, max = 50, message = "3〜50文字で入力してください")
    private String username;

    @NotBlank(message = "パスワードは必須です")
    @Size(min = 8, message = "8文字以上で入力してください")
    private String password;

    @NotBlank(message = "確認パスワードは必須です")
    private String confirmPassword;

    // getter/setter
}
RegisterController.java 
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.web.bind.annotation.*;

@Controller
public class RegisterController {

    private final UserService userService;

    public RegisterController(UserService userService) {
        this.userService = userService;
    }

    /**
     * 登録フォーム表示
     */
    @GetMapping("/register")
    public String showForm(Model model) {
        model.addAttribute("registerForm", new RegisterForm());
        return "register";
    }

    /**
     * ユーザー登録処理
     */
    @PostMapping("/register")
    public String register(@Valid @ModelAttribute RegisterForm form,
                           BindingResult result, Model model) {

        // パスワード一致チェック
        if (!form.getPassword().equals(form.getConfirmPassword())) {
            result.rejectValue("confirmPassword", "error.mismatch",
                "パスワードが一致しません");
        }

        // ユーザー名の重複チェック
        if (userService.existsByUsername(form.getUsername())) {
            result.rejectValue("username", "error.duplicate",
                "このユーザー名は既に使用されています");
        }

        if (result.hasErrors()) {
            return "register";
        }

        userService.register(form.getUsername(), form.getPassword());
        return "redirect:/login?registered=true";
    }
}
UserService.java（登録ロジック） 
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;

@Service
public class UserService {

    private final UserMapper userMapper;
    private final PasswordEncoder passwordEncoder;

    public UserService(UserMapper userMapper,
                       PasswordEncoder passwordEncoder) {
        this.userMapper = userMapper;
        this.passwordEncoder = passwordEncoder;
    }

    /**
     * 新規ユーザーを登録（パスワードはBCryptでハッシュ化）
     */
    public void register(String username, String rawPassword) {
        User user = new User();
        user.setUsername(username);
        user.setPassword(passwordEncoder.encode(rawPassword));
        user.setRole("USER");       // デフォルトは一般ユーザー
        user.setEnabled(true);       // デフォルトは有効
        userMapper.insert(user);
    }

    /**
     * ユーザー名の重複チェック
     */
    public boolean existsByUsername(String username) {
        return userMapper.findByUsername(username) != null;
    }
}

ユーザー一覧・詳細表示（Read）

管理者（ADMIN）専用のユーザー一覧画面と詳細画面を実装します。一般ユーザーはアクセスできないよう@PreAuthorizeで保護します。

AdminUserController.java（一覧・詳細） 
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
@RequestMapping("/admin/users")
@PreAuthorize("hasRole('ADMIN')")
public class AdminUserController {

    private final UserService userService;

    public AdminUserController(UserService userService) {
        this.userService = userService;
    }

    /**
     * ユーザー一覧
     */
    @GetMapping
    public String list(Model model) {
        model.addAttribute("users", userService.findAll());
        return "admin/user-list";
    }

    /**
     * ユーザー詳細
     */
    @GetMapping("/{id}")
    public String detail(@PathVariable Integer id, Model model) {
        User user = userService.findById(id);
        if (user == null) {
            return "redirect:/admin/users";
        }
        model.addAttribute("user", user);
        return "admin/user-detail";
    }
}

AdminUserController.java（更新処理） 
/**
 * 編集フォーム表示
 */
@GetMapping("/{id}/edit")
public String editForm(@PathVariable Integer id, Model model) {
    User user = userService.findById(id);
    if (user == null) {
        return "redirect:/admin/users";
    }

    UserEditForm form = new UserEditForm();
    form.setId(user.getId());
    form.setUsername(user.getUsername());
    form.setRole(user.getRole());
    form.setEnabled(user.isEnabled());

    model.addAttribute("form", form);
    return "admin/user-edit";
}

/**
 * 更新処理
 */
@PostMapping("/{id}/edit")
public String update(@PathVariable Integer id,
                     @Valid @ModelAttribute("form") UserEditForm form,
                     BindingResult result,
                     RedirectAttributes redirectAttributes) {

    if (result.hasErrors()) {
        return "admin/user-edit";
    }

    userService.updateByAdmin(id, form);
    redirectAttributes.addFlashAttribute("message", "ユーザー情報を更新しました");

    return "redirect:/admin/users";
}
UserService.java（更新ロジック） 
/**
 * 管理者によるユーザー情報更新
 */
public void updateByAdmin(Integer id, UserEditForm form) {
    User user = userMapper.findById(id);

    user.setUsername(form.getUsername());
    user.setRole(form.getRole());
    user.setEnabled(form.isEnabled());

    // 新しいパスワードが入力された場合のみ更新
    if (form.getNewPassword() != null
            && !form.getNewPassword().isEmpty()) {
        user.setPassword(
            passwordEncoder.encode(form.getNewPassword())
        );
    }

    userMapper.update(user);
}

Section 16-4

ユーザー削除（Delete）

管理者がユーザーアカウントを削除する機能を実装します。関連するToDoデータの扱いについても考慮します。

AdminUserController.java（削除処理） 
/**
 * ユーザー削除
 */
@PostMapping("/{id}/delete")
public String delete(@PathVariable Integer id,
                     @AuthenticationPrincipal UserDetails currentUser,
                     RedirectAttributes redirectAttributes) {

    User target = userService.findById(id);

    // 自分自身は削除不可
    if (target.getUsername().equals(currentUser.getUsername())) {
        redirectAttributes.addFlashAttribute("error",
            "自分自身のアカウントは削除できません");
        return "redirect:/admin/users";
    }

    userService.deleteUser(id);
    redirectAttributes.addFlashAttribute("message",
        "ユーザー「" + target.getUsername() + "」を削除しました");

    return "redirect:/admin/users";
}
UserService.java（削除ロジック） 
/**
 * ユーザーを削除（関連ToDoも削除）
 */
@Transactional
public void deleteUser(Integer userId) {
    // 1. ユーザーのToDoを先に削除（外部キー制約対策）
    todoMapper.deleteByUserId(userId);
    // 2. ユーザーを削除
    userMapper.deleteById(userId);
}

warning削除時の注意点

自分自身の削除を禁止しています。管理者が自分を削除するとシステムにログインできるユーザーがいなくなる可能性があります。
また、@Transactionalで関連ToDo → ユーザーの順に削除し、途中でエラーが起きた場合は全体をロールバックします。

削除方式の比較

方式説明メリットデメリット
物理削除DBからレコードを完全に削除データがクリーン復元不可
論理削除enabled = falseに設定復元可能・履歴が残るデータが増え続ける
lightbulb本番環境では論理削除を推奨

この教材では物理削除で実装していますが、本番環境ではenabled = falseにする論理削除が一般的です。
監査やデータ復旧の観点から、レコードを残しておく方が安全です。

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリに管理者用のユーザー管理CRUD機能を追加してください。

【要件】
- 新規ユーザー登録：ユーザー名・パスワード入力、BCryptでハッシュ化、重複チェック
- ユーザー一覧：管理者のみアクセス可能、全ユーザーの一覧表示
- ユーザー編集：ロール変更（USER/ADMIN）、有効/無効切替、パスワードリセット
- ユーザー削除：確認ダイアログ付き、関連ToDoも削除、自分自身は削除不可

【技術スタック】
- Spring Security (@PreAuthorize, BCryptPasswordEncoder)
- Spring Boot 3.x + MyBatis
- Thymeleaf
- @Transactional（削除時のデータ整合性）
gps_fixed ステップ別プロンプト 
# Step 1: usersテーブルの拡張
usersテーブルにenabled（有効/無効）列とupdated_at列を追加してください。

# Step 4: ユーザー編集機能
/admin/users/{id}/editでユーザーのロール変更、有効/無効切替、パスワードリセットができる編集画面を作成してください。

# Step 5: ユーザー削除機能
/admin/users/{id}/deleteでユーザーを削除する機能を追加してください。確認ダイアログ、自分自身の削除禁止、@Transactionalで関連ToDoの同時削除を実装してください。

管理者特有の操作フローを明示すると、適切な実装が得られます。
