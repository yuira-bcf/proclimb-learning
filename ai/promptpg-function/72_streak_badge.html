StreakService.java（ストリーク計算） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.LocalDate;

@Service
public class StreakService {

    private final StreakMapper streakMapper;

    public StreakService(StreakMapper streakMapper) {
        this.streakMapper = streakMapper;
    }

    /**
     * タスク完了時にストリークを更新
     */
    @Transactional
    public int updateStreak(Integer userId) {
        UserStreak streak = getOrCreateStreak(userId);
        LocalDate today = LocalDate.now();
        LocalDate lastActive = streak.getLastActiveDate();

        if (lastActive == null) {
            // 初回アクティビティ
            streak.setCurrentStreak(1);
        } else if (lastActive.equals(today)) {
            // 同日の2回目以降 → ストリーク変更なし
            return streak.getCurrentStreak();
        } else if (lastActive.equals(today.minusDays(1))) {
            // 連続日 → ストリーク+1
            streak.setCurrentStreak(streak.getCurrentStreak() + 1);
        } else {
            // 途切れた → リセット
            streak.setCurrentStreak(1);
        }

        // 最大ストリークの更新
        if (streak.getCurrentStreak() > streak.getMaxStreak()) {
            streak.setMaxStreak(streak.getCurrentStreak());
        }

        streak.setLastActiveDate(today);
        streakMapper.update(streak);

        return streak.getCurrentStreak();
    }

    private UserStreak getOrCreateStreak(Integer userId) {
        UserStreak streak = streakMapper.findByUserId(userId);
        if (streak == null) {
            streak = new UserStreak();
            streak.setUserId(userId);
            streak.setCurrentStreak(0);
            streak.setMaxStreak(0);
            streakMapper.insert(streak);
        }
        return streak;
    }
}

BadgeService.java（バッジ授与判定） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.ArrayList;
import java.util.List;

@Service
public class BadgeService {

    private final BadgeMapper badgeMapper;
    private final StreakMapper streakMapper;
    private final PointMapper pointMapper;
    private final TodoMapper todoMapper;

    public BadgeService(BadgeMapper badgeMapper,
                        StreakMapper streakMapper,
                        PointMapper pointMapper,
                        TodoMapper todoMapper) {
        this.badgeMapper = badgeMapper;
        this.streakMapper = streakMapper;
        this.pointMapper = pointMapper;
        this.todoMapper = todoMapper;
    }

    /**
     * バッジ獲得条件をチェックし、新規獲得バッジを返す
     */
    @Transactional
    public List<Badge> checkAndAwardBadges(Integer userId) {
        List<Badge> newBadges = new ArrayList<>();

        // 未獲得のバッジ一覧を取得
        List<Badge> unearnedBadges =
            badgeMapper.findUnearnedByUserId(userId);

        for (Badge badge : unearnedBadges) {
            boolean earned = evaluateCondition(
                userId, badge.getConditionType(),
                badge.getConditionValue());

            if (earned) {
                badgeMapper.insertUserBadge(userId, badge.getId());
                newBadges.add(badge);
            }
        }

        return newBadges;
    }

    /**
     * 条件タイプに応じて達成判定
     */
    private boolean evaluateCondition(Integer userId,
                                       String type, int value) {
        switch (type) {
            case "STREAK":
                UserStreak streak = streakMapper.findByUserId(userId);
                return streak != null
                    && streak.getCurrentStreak() >= value;

            case "TOTAL_COMPLETE":
                int count = todoMapper.countCompletedByUserId(userId);
                return count >= value;

            case "LEVEL":
                UserPoint point = pointMapper.findByUserId(userId);
                return point != null
                    && point.getCurrentLevel() >= value;

            default:
                return false;
        }
    }
}

public TaskCompleteResult completeTodo(Integer todoId, Integer userId) {
    // 1. タスクを完了に更新
    Todo todo = todoMapper.findById(todoId);

lightbulbパフォーマンスの考慮

findUnearnedByUserIdで未獲得バッジのみを取得し、チェック対象を最小限にしています。
バッジ数が少ない（数十件程度）ため、タスク完了の度に全チェックしてもパフォーマンス上問題ありません。

@Controller
@RequestMapping("/badges")
public class BadgeController {

    private final BadgeMapper badgeMapper;

    public BadgeController(BadgeMapper badgeMapper) {
        this.badgeMapper = badgeMapper;
    }

    @GetMapping
    public String list(@AuthenticationPrincipal UserDetails user,
                       Model model) {
        Integer userId = ((CustomUserDetails) user).getUserId();

        List<Badge> allBadges = badgeMapper.findAll();
        List<UserBadge> earned = badgeMapper.findEarnedByUserId(userId);

        // 獲得済みバッジIDのセット
        Set<Integer> earnedIds = earned.stream()
            .map(UserBadge::getBadgeId)
            .collect(Collectors.toSet());

        model.addAttribute("allBadges", allBadges);
        model.addAttribute("earnedIds", earnedIds);
        model.addAttribute("earnedCount", earnedIds.size());
        model.addAttribute("totalCount", allBadges.size());

        return "badges/list";
    }
}

# Step 1: ストリーク計算
user_streaksテーブルを作成し、タスク完了時に連続日数を計算・更新するStreakServiceを実装してください。

# Step 2: バッジ定義
badgesテーブル（条件タイプ + 閾値）とuser_badgesテーブルを作成し、初期バッジデータをINSERTしてください。

# Step 3: バッジ授与判定
タスク完了時に未獲得バッジの条件をチェックし、条件達成時に自動授与するBadgeServiceを作成してください。

# Step 4: バッジ一覧UI
獲得済みバッジをカラフルに、未獲得バッジをグレーアウトで表示するグリッドレイアウトのUIを作成してください。獲得時のポップアップ通知も含めてください。
lightbulb プロンプトのコツ

「バッジ獲得時にconfettiアニメーションを表示し、SNSでシェアできるボタンも追加してください」のように、
ゲーミフィケーションの演出を具体的に指定すると、より魅力的なユーザー体験が実現できます。
