TodoServiceTest.java（基本的なテスト） 
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.BeforeEach;
import static org.junit.jupiter.api.Assertions.*;

class TodoServiceTest {

    private TodoService todoService;

    @BeforeEach
    void setUp() {
        // 各テストの前に実行される
        todoService = new TodoService();
    }

    @Test
    @DisplayName("ToDoを正常に作成できる")
    void testCreateTodo() {
        // Arrange（準備）
        Todo todo = new Todo();
        todo.setTitle("テストタスク");
        todo.setDescription("テスト説明");

        // Act（実行）
        Todo result = todoService.create(todo);

        // Assert（検証）
        assertNotNull(result.getId());
        assertEquals("テストタスク", result.getTitle());
        assertFalse(result.isCompleted());
    }

    @Test
    @DisplayName("タイトルが空の場合は例外が発生する")
    void testCreateTodoWithEmptyTitle() {
        Todo todo = new Todo();
        todo.setTitle("");

        // 例外がスローされることを検証
        assertThrows(IllegalArgumentException.class, () -> {
            todoService.create(todo);
        });
    }

    @Test
    @DisplayName("ToDoを完了状態に更新できる")
    void testCompleteTodo() {
        Todo todo = new Todo();
        todo.setId(1);
        todo.setCompleted(false);

        todoService.complete(1);

        assertTrue(todo.isCompleted());
    }
}

Section 21-2

Mockitoでモック化

TodoServiceMockTest.java（Mockitoを使用） 
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import static org.mockito.Mockito.*;
import static org.junit.jupiter.api.Assertions.*;

@ExtendWith(MockitoExtension.class)
class TodoServiceMockTest {

    @Mock
    private TodoMapper todoMapper;  // モック化するオブジェクト

    @InjectMocks
    private TodoService todoService;  // テスト対象（モックが注入される）

    @Test
    @DisplayName("IDでToDoを取得できる")
    void testFindById() {
        // モックの振る舞いを定義
        Todo expected = new Todo();
        expected.setId(1);
        expected.setTitle("テストタスク");

        when(todoMapper.findById(1)).thenReturn(expected);

        // 実行
        Todo result = todoService.findById(1);

        // 検証
        assertEquals(1, result.getId());
        assertEquals("テストタスク", result.getTitle());

        // メソッドが呼ばれたことを検証
        verify(todoMapper, times(1)).findById(1);
    }

    @Test
    @DisplayName("存在しないIDの場合はnullを返す")
    void testFindByIdNotFound() {
        when(todoMapper.findById(999)).thenReturn(null);

        Todo result = todoService.findById(999);

        assertNull(result);
    }

    @Test
    @DisplayName("ToDoを保存するとMapperのinsertが呼ばれる")
    void testCreate() {
        Todo todo = new Todo();
        todo.setTitle("新規タスク");

        // insertが呼ばれたら引数をそのまま返す
        when(todoMapper.insert(any(Todo.class))).thenReturn(1);

        todoService.create(todo);

        // insertが1回呼ばれたことを検証
        verify(todoMapper, times(1)).insert(todo);
    }

    @Test
    @DisplayName("削除時にMapperのdeleteが呼ばれる")
    void testDelete() {
        todoService.delete(1);

        verify(todoMapper).delete(1);
        verifyNoMoreInteractions(todoMapper);
    }
}
lightbulbMockitoの主要メソッド
  • when().thenReturn()：モックの戻り値を設定
  • verify()：メソッドが呼ばれたか検証
  • any()：任意の引数にマッチ
  • times(n)：n回呼ばれたか検証
  • never()：呼ばれていないことを検証
Section 21-3

@SpringBootTestによる統合テスト

TodoControllerIntegrationTest.java 
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.web.servlet.MockMvc;
import org.springframework.transaction.annotation.Transactional;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.*;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;

@SpringBootTest
@AutoConfigureMockMvc
@Transactional  // テスト後にロールバック
class TodoControllerIntegrationTest {

    @Autowired
    private MockMvc mockMvc;

    @Autowired
    private TodoMapper todoMapper;

    @Test
    @DisplayName("ToDo一覧ページが表示される")
    void testListPage() throws Exception {
        mockMvc.perform(get("/todo"))
            .andExpect(status().isOk())
            .andExpect(view().name("todo/list"))
            .andExpect(model().attributeExists("todoList"));
    }

    @Test
    @DisplayName("新規作成フォームが表示される")
    void testCreateForm() throws Exception {
        mockMvc.perform(get("/todo/new"))
            .andExpect(status().isOk())
            .andExpect(view().name("todo/form"))
            .andExpect(model().attributeExists("todoForm"));
    }

    @Test
    @DisplayName("ToDoを作成できる")
    void testCreate() throws Exception {
        mockMvc.perform(post("/todo")
                .param("title", "テストタスク")
                .param("description", "テスト説明"))
            .andExpect(status().is3xxRedirection())
            .andExpect(redirectedUrl("/todo"));
    }

    @Test
    @DisplayName("バリデーションエラー時はフォームに戻る")
    void testCreateValidationError() throws Exception {
        mockMvc.perform(post("/todo")
                .param("title", ""))  // タイトルが空
            .andExpect(status().isOk())
            .andExpect(view().name("todo/form"))
            .andExpect(model().hasErrors());
    }

    @Test
    @DisplayName("ToDoの詳細が表示される")
    void testDetail() throws Exception {
        // テストデータを準備
        Todo todo = new Todo();
        todo.setTitle("テストタスク");
        todoMapper.insert(todo);

        mockMvc.perform(get("/todo/{id}", todo.getId()))
            .andExpect(status().isOk())
            .andExpect(view().name("todo/detail"))
            .andExpect(model().attribute("todo", todo));
    }
}
TodoMapperTest.java（MyBatisテスト） 
import org.junit.jupiter.api.Test;
import org.mybatis.spring.boot.test.autoconfigure.MybatisTest;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.jdbc.AutoConfigureTestDatabase;
import static org.junit.jupiter.api.Assertions.*;

@MybatisTest
@AutoConfigureTestDatabase(replace = AutoConfigureTestDatabase.Replace.NONE)
class TodoMapperTest {

    @Autowired
    private TodoMapper todoMapper;

    @Test
    @DisplayName("ToDoを挿入して取得できる")
    void testInsertAndFindById() {
        Todo todo = new Todo();
        todo.setTitle("テストタスク");
        todo.setDescription("テスト説明");

        todoMapper.insert(todo);
        assertNotNull(todo.getId());

        Todo found = todoMapper.findById(todo.getId());
        assertEquals("テストタスク", found.getTitle());
    }

    @Test
    @DisplayName("全件取得できる")
    void testFindAll() {
        // テストデータを挿入
        Todo todo1 = new Todo();
        todo1.setTitle("タスク1");
        todoMapper.insert(todo1);

        Todo todo2 = new Todo();
        todo2.setTitle("タスク2");
        todoMapper.insert(todo2);

        List<Todo> todos = todoMapper.findAll();
        assertTrue(todos.size() >= 2);
    }
}

テストの種類と使い分け

種類アノテーション用途
単体テスト@ExtendWith(MockitoExtension.class)1クラスを独立してテスト
統合テスト@SpringBootTestアプリケーション全体をテスト
Webレイヤーテスト@WebMvcTestControllerのみテスト
データレイヤーテスト@MybatisTestMapperのみテスト
warningテストのベストプラクティス
  • テストは独立して実行できるようにする（他のテストに依存しない）
  • @Transactionalでテスト後にデータをロールバック
  • テスト名は「何をテストしているか」が分かるように
  • AAAパターン（Arrange, Act, Assert）で整理

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリに単体テストを追加してください。

【要件】
- JUnit5とMockitoを使用したTodoServiceの単体テスト
- @Mock でTodoMapperをモック化し、@InjectMocksでサービスに注入
- when().thenReturn()でモックの振る舞いを定義
- verify()でメソッド呼び出しを検証
- @SpringBootTestとMockMvcを使用したControllerの統合テスト
- @MybatisTestを使用したMapperのテスト
- @Transactionalでテスト後のロールバック

【技術スタック】
- JUnit5
- Mockito
- Spring Boot Test
- MockMvc
- MyBatis Spring Boot Starter Test
gps_fixed ステップ別プロンプト 
# Step 1: 基本的な単体テスト
TodoServiceTestクラスを作成し、create、findById、deleteメソッドの
基本的なテストケースを@Test、@DisplayNameを使って実装してください。

# Step 2: Mockitoによるモックテスト
@ExtendWith(MockitoExtension.class)を使い、TodoMapperをモック化した
TodoServiceのテストを作成してください。when/verify を使用。

# Step 3: MockMvcによる統合テスト
@SpringBootTest、@AutoConfigureMockMvcを使ったControllerの統合テストを作成し、
GET/POSTリクエストの検証、ステータスコード、ビュー名の確認を行ってください。

# Step 4: Mapperテスト
@MybatisTestを使ってTodoMapperのCRUD操作をテストしてください。
テスト後は@Transactionalでロールバック。
lightbulb プロンプトのコツ

「既存のTodoService.javaを読み込んで、全メソッドのテストケースを自動生成して」のように、
既存コードを指定すると、カバレッジの高いテストが得られます。
