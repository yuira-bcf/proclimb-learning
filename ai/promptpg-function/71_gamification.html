CREATE TABLE user_points (
    id INT AUTO_INCREMENT PRIMARY KEY,

    total_points INT DEFAULT 0,
    current_level INT DEFAULT 1,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- ポイント付与履歴
CREATE TABLE point_history (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    points INT NOT NULL,
    reason VARCHAR(100) NOT NULL,
    todo_id INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (todo_id) REFERENCES todo(id)
        ON DELETE SET NULL
);

Section 71-2

ポイント付与ロジック

タスク完了時にポイントを自動付与するサービスを実装します。タスクの優先度や期限内完了などの条件でボーナスポイントも設定します。

stars

基本ポイント

タスク完了で10ポイント付与

speed

優先度ボーナス

高優先度タスクほど追加ポイント

timer

期限内ボーナス

期限前に完了すると追加ポイント

PointService.java（ポイント付与ロジック） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.LocalDate;

@Service
public class PointService {

    private static final int BASE_POINTS = 10;
    private static final int PRIORITY_HIGH_BONUS = 10;
    private static final int PRIORITY_MEDIUM_BONUS = 5;
    private static final int ON_TIME_BONUS = 5;

    private final PointMapper pointMapper;

    public PointService(PointMapper pointMapper) {
        this.pointMapper = pointMapper;
    }

    /**
     * タスク完了時にポイントを付与
     */
    @Transactional
    public int awardPoints(Integer userId, Todo completedTodo) {
        // 1. 基本ポイント
        int points = BASE_POINTS;

        // 2. 優先度ボーナス
        if ("HIGH".equals(completedTodo.getPriority())) {
            points += PRIORITY_HIGH_BONUS;
        } else if ("MEDIUM".equals(completedTodo.getPriority())) {
            points += PRIORITY_MEDIUM_BONUS;
        }

        // 3. 期限内完了ボーナス
        if (completedTodo.getDeadline() != null
                && !LocalDate.now().isAfter(completedTodo.getDeadline())) {
            points += ON_TIME_BONUS;
        }

        // 4. 履歴に記録
        PointHistory history = new PointHistory();
        history.setUserId(userId);
        history.setPoints(points);
        history.setReason("TASK_COMPLETE");
        history.setTodoId(completedTodo.getId());
        pointMapper.insertHistory(history);

        // 5. 累計ポイントを更新
        UserPoint userPoint = getOrCreateUserPoint(userId);
        int newTotal = userPoint.getTotalPoints() + points;
        int newLevel = calculateLevel(newTotal);
        pointMapper.updateTotalPoints(userId, newTotal, newLevel);

        return points;
    }

    private UserPoint getOrCreateUserPoint(Integer userId) {
        UserPoint up = pointMapper.findByUserId(userId);
        if (up == null) {
            up = new UserPoint();
            up.setUserId(userId);
            up.setTotalPoints(0);
            up.setCurrentLevel(1);
            pointMapper.insertUserPoint(up);
        }
        return up;
    }
}
TodoService.java（ポイント付与の統合） 
/**
 * タスク完了処理（ポイント付与付き）
 */
@Transactional
public String completeTodo(Integer todoId, Integer userId) {
    Todo todo = todoMapper.findById(todoId);
    todo.setCompleted(true);
    todoMapper.update(todo);

    // ポイント付与
    int earned = pointService.awardPoints(userId, todo);

    return earned + "ポイント獲得！";
}

lightbulb@Transactionalの重要性

ポイント付与は「履歴記録」と「累計更新」の2操作で構成されます。@Transactionalで一つのトランザクションにまとめ、途中でエラーが発生した場合に両方ロールバックされるようにします。

PointService.java（レベル計算） 
/**
 * 累計ポイントからレベルを算出
 * レベルn に必要な累計ポイント = n * (n - 1) * 50 / 2
 * Lv1: 0pt, Lv2: 50pt, Lv3: 150pt, Lv4: 300pt, ...
 */
public static int calculateLevel(int totalPoints) {
    int level = 1;
    int threshold = 0;
    while (true) {
        int nextThreshold = threshold + level * 50;
        if (totalPoints < nextThreshold) {
            break;
        }
        threshold = nextThreshold;
        level++;
    }
    return level;
}

/**
 * 現在のレベルの進捗率を算出（0.0 〜 1.0）
 */
public static double calculateProgress(int totalPoints) {
    int level = calculateLevel(totalPoints);
    int currentThreshold = getLevelThreshold(level);
    int nextThreshold = getLevelThreshold(level + 1);
    int range = nextThreshold - currentThreshold;

    if (range == 0) return 1.0;
    return (double)(totalPoints - currentThreshold) / range;
}

/**
 * 指定レベルに必要な累計ポイントを算出
 */
public static int getLevelThreshold(int level) {
    int total = 0;
    for (int i = 1; i < level; i++) {
        total += i * 50;
    }
    return total;
}
ProfileController.java（レベル表示） 
@Controller
public class ProfileController {

    private final PointService pointService;
    private final PointMapper pointMapper;

    public ProfileController(PointService pointService,
                             PointMapper pointMapper) {
        this.pointService = pointService;
        this.pointMapper = pointMapper;
    }

    @GetMapping("/profile")
    public String profile(@AuthenticationPrincipal UserDetails user,
                          Model model) {
        Integer userId = ((CustomUserDetails) user).getUserId();
        UserPoint up = pointMapper.findByUserId(userId);

        if (up != null) {
            model.addAttribute("totalPoints", up.getTotalPoints());
            model.addAttribute("level", up.getCurrentLevel());
            model.addAttribute("progress",
                PointService.calculateProgress(up.getTotalPoints()));
            model.addAttribute("nextLevelPoints",
                PointService.getLevelThreshold(up.getCurrentLevel() + 1));
        } else {
            model.addAttribute("totalPoints", 0);
            model.addAttribute("level", 1);
            model.addAttribute("progress", 0.0);
            model.addAttribute("nextLevelPoints", 50);
        }

        return "profile";
    }
}

lightbulbレベル設計の考え方

レベルアップに必要なポイントを段階的に増加させることで、初期は素早くレベルアップし、上位レベルほど達成感が増す設計になります。
Lv1→2は50pt、Lv2→3は100ptと、各レベルでレベル数 * 50ptが必要です。

public class PointHistoryController {

    @GetMapping("/history")
    public String history(@AuthenticationPrincipal UserDetails user,
                          Model model) {
        Integer userId = ((CustomUserDetails) user).getUserId();

        List<PointHistory> histories =
            pointMapper.findHistoryByUserId(userId, 50);
        UserPoint userPoint = pointMapper.findByUserId(userId);

        model.addAttribute("histories", histories);
        model.addAttribute("userPoint", userPoint);
        return "points/history";
    }
}

PointMapper.xml（履歴取得SQL） 
<select id="findHistoryByUserId" resultType="PointHistory">
    SELECT ph.id, ph.user_id, ph.points, ph.reason,
           ph.todo_id, ph.created_at,
           t.title AS todo_title
    FROM point_history ph
    LEFT JOIN todo t ON ph.todo_id = t.id
    WHERE ph.user_id = #{userId}
    ORDER BY ph.created_at DESC
    LIMIT #{limit}
</select>

<select id="findByUserId" resultType="UserPoint">
    SELECT id, user_id, total_points, current_level, updated_at
    FROM user_points
    WHERE user_id = #{userId}
</select>

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにゲーミフィケーション（ポイント/レベルシステム）を追加してください。

【要件】
- タスク完了時にポイントを自動付与（基本10pt）
- 優先度ボーナス（HIGH: +10pt, MEDIUM: +5pt）と期限内完了ボーナス（+5pt）
- 累計ポイントからレベルを算出（段階的に必要ポイント増加）
- プロフィール画面にレベル・プログレスバーを表示
- ポイント履歴一覧（日時・アクション・ポイント数）

【技術スタック】
- Spring Boot 3.x + MyBatis
- Thymeleaf
- @Transactional（ポイント付与の原子性）
gps_fixed ステップ別プロンプト 
# Step 1: テーブル設計
user_pointsテーブル（累計ポイント・現在レベル）とpoint_historyテーブル（付与履歴）を作成してください。

# Step 2: ポイント付与サービス
タスク完了時にポイントを計算・付与するPointServiceを作成してください。優先度ボーナスと期限内ボーナスを含めてください。

# Step 3: レベル計算
累計ポイントからレベルを算出するロジックと、次のレベルまでの進捗率を計算するメソッドを実装してください。

# Step 4: 表示UI
プロフィール画面にレベル・プログレスバー・ポイント履歴一覧を表示するThymeleafテンプレートを作成してください。

ゲーミフィケーションの演出を具体的に指定すると、ユーザー体験が向上する実装が得られます。
