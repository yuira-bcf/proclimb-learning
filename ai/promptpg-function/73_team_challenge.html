schema.sql（チーム・チャレンジテーブル） 
-- チーム
CREATE TABLE teams (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    created_by INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- チームメンバー
CREATE TABLE team_members (
    id INT AUTO_INCREMENT PRIMARY KEY,
    team_id INT NOT NULL,
    user_id INT NOT NULL,
    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (team_id) REFERENCES teams(id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    UNIQUE KEY uk_team_user (team_id, user_id)
);

-- チャレンジ定義
CREATE TABLE challenges (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    description VARCHAR(500),
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    goal_type VARCHAR(30) NOT NULL,   -- 'TASK_COUNT', 'POINT_SUM'
    goal_value INT NOT NULL,           -- 目標値
    status VARCHAR(20) DEFAULT 'ACTIVE', -- ACTIVE, COMPLETED, CANCELLED
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- チャレンジ参加（チーム単位）
CREATE TABLE challenge_participants (
    id INT AUTO_INCREMENT PRIMARY KEY,
    challenge_id INT NOT NULL,
    team_id INT NOT NULL,
    score INT DEFAULT 0,
    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (challenge_id) REFERENCES challenges(id),
    FOREIGN KEY (team_id) REFERENCES teams(id),
    UNIQUE KEY uk_challenge_team (challenge_id, team_id)
);

TeamService.java（チーム管理） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
public class TeamService {

    private final TeamMapper teamMapper;

    public TeamService(TeamMapper teamMapper) {
        this.teamMapper = teamMapper;
    }

    /**
     * チーム作成（作成者は自動的にメンバーに追加）
     */
    @Transactional
    public Team createTeam(String name, Integer createdByUserId) {
        Team team = new Team();
        team.setName(name);
        team.setCreatedBy(createdByUserId);
        teamMapper.insertTeam(team);

        // 作成者をメンバーに追加
        teamMapper.insertMember(team.getId(), createdByUserId);
        return team;
    }

    /**
     * メンバーをチームに追加
     */
    @Transactional
    public void addMember(Integer teamId, Integer userId) {
        if (teamMapper.isMember(teamId, userId)) {
            throw new IllegalStateException("既にチームメンバーです");
        }
        teamMapper.insertMember(teamId, userId);
    }

    /**
     * ユーザーが所属するチーム一覧
     */
    public List<Team> findTeamsByUserId(Integer userId) {
        return teamMapper.findByUserId(userId);
    }
}
ChallengeService.java（チャレンジ参加） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.LocalDate;

@Service
public class ChallengeService {

    private final ChallengeMapper challengeMapper;

    public ChallengeService(ChallengeMapper challengeMapper) {
        this.challengeMapper = challengeMapper;
    }

    /**
     * チームをチャレンジに参加させる
     */
    @Transactional
    public void joinChallenge(Integer challengeId, Integer teamId) {
        Challenge challenge = challengeMapper.findById(challengeId);

        // チャレンジが有効か確認
        if (!"ACTIVE".equals(challenge.getStatus())) {
            throw new IllegalStateException("このチャレンジは終了しています");
        }
        if (LocalDate.now().isAfter(challenge.getEndDate())) {
            throw new IllegalStateException("チャレンジ期限を過ぎています");
        }

        // 重複参加チェック
        if (challengeMapper.isParticipant(challengeId, teamId)) {
            throw new IllegalStateException("既に参加済みです");
        }

        challengeMapper.insertParticipant(challengeId, teamId);
    }

    /**
     * アクティブなチャレンジ一覧
     */
    public List<Challenge> findActiveChallenges() {
        return challengeMapper.findByStatus("ACTIVE");
    }
}

@Controller
@RequestMapping("/challenges")
public class ChallengeController {

    private final ChallengeService challengeService;
    private final TeamService teamService;

    public ChallengeController(ChallengeService challengeService,
                                TeamService teamService) {
        this.challengeService = challengeService;
        this.teamService = teamService;
    }

    @GetMapping
    public String list(Model model) {
        model.addAttribute("challenges",
            challengeService.findActiveChallenges());
        return "challenges/list";
    }

    @PostMapping("/{id}/join")
    public String join(@PathVariable Integer id,
                       @RequestParam Integer teamId,
                       @AuthenticationPrincipal UserDetails user,
                       RedirectAttributes redirectAttributes) {
        try {
            challengeService.joinChallenge(id, teamId);
            redirectAttributes.addFlashAttribute("message",
                "チャレンジに参加しました！");
        } catch (IllegalStateException e) {
            redirectAttributes.addFlashAttribute("error", e.getMessage());
        }
        return "redirect:/challenges/" + id;
    }
}

チーム別スコア集計

チャレンジ期間中のタスク完了をチームのスコアに自動加算するロジックを実装します。タスク完了時にユーザーが参加中のチャレンジを探してスコアを更新します。

ChallengeScoreService.java（スコア集計） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.LocalDate;
import java.util.List;

@Service
public class ChallengeScoreService {

    private final ChallengeMapper challengeMapper;
    private final TeamMapper teamMapper;

    public ChallengeScoreService(ChallengeMapper challengeMapper,
                                  TeamMapper teamMapper) {
        this.challengeMapper = challengeMapper;
        this.teamMapper = teamMapper;
    }

    /**
     * タスク完了時にチャレンジスコアを更新
     */
    @Transactional
    public void onTaskCompleted(Integer userId) {
        LocalDate today = LocalDate.now();

        // ユーザーが所属するチームを取得
        List<Team> teams = teamMapper.findByUserId(userId);

        for (Team team : teams) {
            // チームが参加中のアクティブなチャレンジを取得
            List<ChallengeParticipant> participants =
                challengeMapper.findActiveParticipants(
                    team.getId(), today);

            for (ChallengeParticipant cp : participants) {
                // スコアを+1
                challengeMapper.incrementScore(
                    cp.getChallengeId(), team.getId());
            }
        }
    }

    /**
     * チャレンジのランキング（スコア降順）
     */
    public List<ChallengeParticipant> getRanking(Integer challengeId) {
        return challengeMapper.findParticipantsByScore(challengeId);
    }
}
ChallengeMapper.xml（スコア集計SQL） 
<!-- スコア加算 -->
<update id="incrementScore">
    UPDATE challenge_participants
    SET score = score + 1
    WHERE challenge_id = #{challengeId}
      AND team_id = #{teamId}
</update>

<!-- チームが参加中のアクティブなチャレンジ -->
<select id="findActiveParticipants" resultType="ChallengeParticipant">
    SELECT cp.*
    FROM challenge_participants cp
    JOIN challenges c ON cp.challenge_id = c.id
    WHERE cp.team_id = #{teamId}
      AND c.status = 'ACTIVE'
      AND c.start_date <= #{today}
      AND c.end_date >= #{today}
</select>

<!-- ランキング（スコア降順） -->
<select id="findParticipantsByScore" resultType="ChallengeParticipant">
    SELECT cp.*, t.name AS team_name,
           (SELECT COUNT(*) FROM team_members
            WHERE team_id = cp.team_id) AS member_count
    FROM challenge_participants cp
    JOIN teams t ON cp.team_id = t.id
    WHERE cp.challenge_id = #{challengeId}
    ORDER BY cp.score DESC
</select>
ChallengeScheduler.java（自動完了判定） 
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;
import java.time.LocalDate;

@Component
public class ChallengeScheduler {

    private final ChallengeMapper challengeMapper;

    public ChallengeScheduler(ChallengeMapper challengeMapper) {
        this.challengeMapper = challengeMapper;
    }

    /**
     * 毎日0:00に期限切れチャレンジを完了に更新
     */
    @Scheduled(cron = "0 0 0 * * *")
    public void closeExpiredChallenges() {
        LocalDate yesterday = LocalDate.now().minusDays(1);
        challengeMapper.updateStatusByEndDate(yesterday, "COMPLETED");
    }
}

@GetMapping("/{id}")
public String detail(@PathVariable Integer id, Model model) {
    Challenge challenge = challengeMapper.findById(id);
    List<ChallengeParticipant> ranking =
        challengeScoreService.getRanking(id);

    // 残り日数を計算
    long daysRemaining = java.time.temporal.ChronoUnit.DAYS
        .between(LocalDate.now(), challenge.getEndDate());

    model.addAttribute("challenge", challenge);
    model.addAttribute("ranking", ranking);
    model.addAttribute("daysRemaining", Math.max(0, daysRemaining));

    return "challenges/detail";
}

- 自動終了: @Scheduledで期限切れチャレンジを自動完了
