CsvImportController.java（アップロード受付） 
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;

@Controller
@RequestMapping("/todo/import")
public class CsvImportController {

    private final CsvImportService importService;

    public CsvImportController(CsvImportService importService) {
        this.importService = importService;
    }

    @GetMapping
    public String showForm() {
        return "import";
    }

    @PostMapping
    public String importCsv(@RequestParam("file") MultipartFile file,
                            Model model,
                            RedirectAttributes redirectAttributes) {
        // ファイル形式チェック
        String filename = file.getOriginalFilename();
        if (filename == null || !filename.endsWith(".csv")) {
            model.addAttribute("errors",
                java.util.List.of("CSVファイルを選択してください"));
            return "import";
        }

        // インポート実行
        CsvImportResult result = importService.importCsv(file);

        if (result.hasErrors()) {
            model.addAttribute("errors", result.getErrors());
            return "import";
        }

        redirectAttributes.addFlashAttribute("message",
            result.getSuccessCount() + "件のタスクをインポートしました");
        return "redirect:/todo";
    }

    @GetMapping("/template")
    public void downloadTemplate(
            jakarta.servlet.http.HttpServletResponse response)
            throws Exception {
        response.setContentType("text/csv; charset=UTF-8");
        response.setHeader("Content-Disposition",
            "attachment; filename=todo_template.csv");
        response.getWriter().write(
            "タイトル,説明,優先度,期限,カテゴリ\n"
            + "サンプルタスク,タスクの説明文,HIGH,2026-04-01,仕事\n"
        );
    }
}

アップロードされたCSVファイルを1行ずつ解析し、Todoオブジェクトに変換するパーサーを実装します。BufferedReaderでストリーム処理することで大容量ファイルにも対応します。

CsvParser.java（CSVパーサー） 
import org.springframework.stereotype.Component;
import org.springframework.web.multipart.MultipartFile;
import java.io.*;
import java.nio.charset.StandardCharsets;
import java.util.*;

@Component
public class CsvParser {

    /**
     * CSVファイルをパースしてMap<列名, 値>のリストに変換
     */
    public List<Map<String, String>> parse(MultipartFile file)
            throws IOException {

        List<Map<String, String>> rows = new ArrayList<>();

        try (BufferedReader reader = new BufferedReader(
                new InputStreamReader(file.getInputStream(),
                    StandardCharsets.UTF_8))) {

            // ヘッダー行の読み取り
            String headerLine = reader.readLine();
            if (headerLine == null) {
                throw new IOException("空のCSVファイルです");
            }

            // BOM除去（Excel出力CSV対策）
            if (headerLine.startsWith("\uFEFF")) {
                headerLine = headerLine.substring(1);
            }

            String[] headers = splitCsvLine(headerLine);

            // データ行の読み取り
            String line;
            while ((line = reader.readLine()) != null) {
                if (line.isBlank()) continue;
                String[] values = splitCsvLine(line);
                Map<String, String> row = new LinkedHashMap<>();
                for (int i = 0; i < headers.length; i++) {
                    String value = (i < values.length)
                        ? values[i].trim() : "";
                    row.put(headers[i].trim(), value);
                }
                rows.add(row);
            }
        }
        return rows;
    }

    /**
     * CSV行をダブルクォート考慮で分割
     */
    private String[] splitCsvLine(String line) {
        List<String> fields = new ArrayList<>();
        StringBuilder sb = new StringBuilder();
        boolean inQuotes = false;

        for (char c : line.toCharArray()) {
            if (c == '"') {
                inQuotes = !inQuotes;
            } else if (c == ',' && !inQuotes) {
                fields.add(sb.toString());
                sb.setLength(0);
            } else {
                sb.append(c);
            }
        }
        fields.add(sb.toString());
        return fields.toArray(new String[0]);
    }
}
warningBOM（バイトオーダーマーク）に注意

Excelで保存したCSVファイルには先頭にBOM（\uFEFF）が付くことがあります。これを除去しないとヘッダー行の最初の列名が正しく認識されません。

バリデーションとエラーレポート

CsvImportService.java（バリデーション付きインポート） 
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.web.multipart.MultipartFile;
import java.time.LocalDate;
import java.time.format.DateTimeParseException;
import java.util.*;

@Service
public class CsvImportService {

    private final CsvParser csvParser;
    private final TodoMapper todoMapper;

    private static final Set<String> VALID_PRIORITIES =
        Set.of("HIGH", "MEDIUM", "LOW");

    public CsvImportService(CsvParser csvParser,
                            TodoMapper todoMapper) {
        this.csvParser = csvParser;
        this.todoMapper = todoMapper;
    }

    @Transactional
    public CsvImportResult importCsv(MultipartFile file) {
        CsvImportResult result = new CsvImportResult();

        try {
            List<Map<String, String>> rows = csvParser.parse(file);
            List<Todo> validTodos = new ArrayList<>();

            for (int i = 0; i < rows.size(); i++) {
                int lineNum = i + 2; // ヘッダー行=1なのでデータは2行目から
                Map<String, String> row = rows.get(i);
                List<String> rowErrors = validateRow(row, lineNum);

                if (rowErrors.isEmpty()) {
                    validTodos.add(mapToTodo(row));
                } else {
                    result.addErrors(rowErrors);
                }
            }

            // 正常行を一括INSERT
            if (!validTodos.isEmpty()) {
                todoMapper.insertBatch(validTodos);
                result.setSuccessCount(validTodos.size());
            }
        } catch (Exception e) {
            result.addError("ファイルの読み込みに失敗: " + e.getMessage());
        }
        return result;
    }

    private List<String> validateRow(Map<String, String> row,
                                      int lineNum) {
        List<String> errors = new ArrayList<>();
        String title = row.getOrDefault("タイトル", "");
        String priority = row.getOrDefault("優先度", "");
        String deadline = row.getOrDefault("期限", "");

        if (title.isBlank()) {
            errors.add(lineNum + "行目: タイトルは必須です");
        }
        if (!priority.isBlank()
                && !VALID_PRIORITIES.contains(priority)) {
            errors.add(lineNum + "行目: 優先度は HIGH/MEDIUM/LOW");
        }
        if (!deadline.isBlank()) {
            try { LocalDate.parse(deadline); }
            catch (DateTimeParseException e) {
                errors.add(lineNum
                    + "行目: 期限はyyyy-MM-dd形式で入力");
            }
        }
        return errors;
    }

    private Todo mapToTodo(Map<String, String> row) {
        Todo todo = new Todo();
        todo.setTitle(row.get("タイトル"));
        todo.setDescription(row.getOrDefault("説明", ""));
        todo.setPriority(row.getOrDefault("優先度", "MEDIUM"));
        String deadline = row.getOrDefault("期限", "");
        if (!deadline.isBlank()) {
            todo.setDeadline(LocalDate.parse(deadline));
        }
        todo.setCategory(row.getOrDefault("カテゴリ", ""));
        return todo;
    }
}
lightbulb部分インポート方式

エラーのある行だけスキップし、正常な行はインポートする「部分インポート」方式を採用しています。100行中95行が正常なら95件を登録し、5件のエラー行だけレポートします。

CsvImportResult.java（インポート結果クラス） 
import java.util.ArrayList;
import java.util.List;

public class CsvImportResult {

    private int successCount = 0;
    private final List<String> errors = new ArrayList<>();

    public void setSuccessCount(int count) {
        this.successCount = count;
    }
    public int getSuccessCount() {
        return successCount;
    }

    public void addError(String error) {
        this.errors.add(error);
    }
    public void addErrors(List<String> errors) {
        this.errors.addAll(errors);
    }
    public List<String> getErrors() {
        return errors;
    }

    public boolean hasErrors() {
        return !errors.isEmpty();
    }
}
warning大量データのバッチサイズ

1回のINSERT文に含められる行数にはDB側の制限があります。1000行を超える場合は、500行ずつに分割してバッチINSERTを実行するようにしましょう。
