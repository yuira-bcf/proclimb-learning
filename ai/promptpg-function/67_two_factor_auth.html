TotpService.java（TOTP生成・検証）
import org.apache.commons.codec.binary.Base32;
import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;
import java.security.SecureRandom;

@Service
public class TotpService {

    private static final int CODE_DIGITS = 6;
    private static final int TIME_STEP = 30; // 秒

    /**
     * シークレットキーを生成（Base32）
     */
    public String generateSecret() {
        SecureRandom random = new SecureRandom();
        byte[] bytes = new byte[20];
        random.nextBytes(bytes);
        return new Base32().encodeToString(bytes);
    }

    /**
     * TOTPコードを検証
     */
    public boolean verifyCode(String secret, String code) {
        Base32 base32 = new Base32();
        byte[] decodedKey = base32.decode(secret);
        long timeIndex = System.currentTimeMillis() / 1000 / TIME_STEP;

        // 前後1ステップの許容（時刻ズレ対策）
        for (int i = -1; i <= 1; i++) {
            String generated = generateTOTP(decodedKey, timeIndex + i);
            if (generated.equals(code)) {
                return true;
            }
        }
        return false;
    }

    private String generateTOTP(byte[] key, long timeIndex) {
        try {
            byte[] data = new byte[8];
            for (int i = 7; i >= 0; i--) {
                data[i] = (byte) (timeIndex & 0xFF);
                timeIndex >>= 8;
            }
            Mac mac = Mac.getInstance("HmacSHA1");
            mac.init(new SecretKeySpec(key, "HmacSHA1"));
            byte[] hash = mac.doFinal(data);
            int offset = hash[hash.length - 1] & 0x0F;
            int otp = ((hash[offset] & 0x7F) << 24)
                    | ((hash[offset + 1] & 0xFF) << 16)
                    | ((hash[offset + 2] & 0xFF) << 8)
                    | (hash[offset + 3] & 0xFF);
            return String.format("%0" + CODE_DIGITS + "d",
                otp % (int) Math.pow(10, CODE_DIGITS));
        } catch (Exception e) {
            throw new RuntimeException("TOTP生成エラー", e);
        }
    }
}
lightbulb時刻ズレの許容

-1 <= i <= 1で前後30秒の時刻ズレを許容しています。サーバーとスマートフォンの時計が多少ずれていても認証が通ります。

QrCodeGenerator.java
import com.google.zxing.BarcodeFormat;
import com.google.zxing.client.j2se.MatrixToImageWriter;
import com.google.zxing.qrcode.QRCodeWriter;
import java.io.ByteArrayOutputStream;
import java.util.Base64;

@Component
public class QrCodeGenerator {

    /**
     * TOTP用のotpauth URIからQRコードをBase64画像で返す
     */
    public String generateQrBase64(String username, String secret) {
        String issuer = "ToDoApp";
        String otpauthUrl = String.format(
            "otpauth://totp/%s:%s?secret=%s&issuer=%s&digits=6",
            issuer, username, secret, issuer
        );

        try {
            QRCodeWriter writer = new QRCodeWriter();
            var matrix = writer.encode(otpauthUrl,
                BarcodeFormat.QR_CODE, 200, 200);

            ByteArrayOutputStream os = new ByteArrayOutputStream();
            MatrixToImageWriter.writeToStream(matrix, "PNG", os);

            return "data:image/png;base64,"
                + Base64.getEncoder().encodeToString(os.toByteArray());
        } catch (Exception e) {
            throw new RuntimeException("QRコード生成エラー", e);
        }
    }
}

Section 67-3

認証フロー実装

ログイン後に2FA認証コードの入力を求めるフローを実装します。Spring Securityと連携させます。

login

2段階ログインフロー

ID/PW認証後 → 2FAコード確認 → ログイン完了

verified_user

セッションに中間状態保存

2FA未完了の状態をセッションで管理

block

2FA未認証のアクセス制限

2FA完了前は他のページにアクセスできないよう制御

TwoFactorAuthController.java
@Controller
@RequestMapping("/2fa")
public class TwoFactorAuthController {

    private final TotpService totpService;
    private final UserMapper userMapper;

    public TwoFactorAuthController(TotpService totpService,
                                    UserMapper userMapper) {
        this.totpService = totpService;
        this.userMapper = userMapper;
    }

    /**
     * 2FA認証コード入力画面を表示
     */
    @GetMapping("/verify")
    public String showVerifyForm(HttpSession session) {
        // セッションに2FA待ち状態がなければログインへ
        if (session.getAttribute("2fa_user_id") == null) {
            return "redirect:/login";
        }
        return "2fa/verify";
    }

    /**
     * 2FA認証コードを検証
     */
    @PostMapping("/verify")
    public String verify(@RequestParam String code,
                         HttpSession session,
                         RedirectAttributes redirectAttributes) {

        Integer userId = (Integer) session.getAttribute("2fa_user_id");
        if (userId == null) return "redirect:/login";

        User user = userMapper.findById(userId);
        if (totpService.verifyCode(user.getTotpSecret(), code)) {
            // 2FA成功: セッションに認証済みフラグを設定
            session.setAttribute("2fa_verified", true);
            session.removeAttribute("2fa_user_id");
            return "redirect:/todo";
        }

        // 2FA失敗
        redirectAttributes.addFlashAttribute("error",
            "認証コードが正しくありません");
        return "redirect:/2fa/verify";
    }
}
TwoFactorAuthFilter.java（2FA未認証を制限）
import jakarta.servlet.*;
import jakarta.servlet.http.*;
import org.springframework.stereotype.Component;

@Component
public class TwoFactorAuthFilter implements Filter {

    @Override
    public void doFilter(ServletRequest request, ServletResponse response,
                         FilterChain chain) throws Exception {

        HttpServletRequest req = (HttpServletRequest) request;
        HttpServletResponse res = (HttpServletResponse) response;
        HttpSession session = req.getSession(false);

        String path = req.getRequestURI();

        // 2FA関連ページ・静的リソースはフィルタ対象外
        if (path.startsWith("/2fa/") || path.startsWith("/login")
            || path.startsWith("/css/") || path.startsWith("/js/")) {
            chain.doFilter(request, response);
            return;
        }

        // 2FA待ち状態で他ページにアクセスした場合、2FAページへリダイレクト
        if (session != null
            && session.getAttribute("2fa_user_id") != null
            && session.getAttribute("2fa_verified") == null) {
            res.sendRedirect("/2fa/verify");
            return;
        }

        chain.doFilter(request, response);
    }
}

TwoFactorSettingsController.java
@Controller
@RequestMapping("/settings/2fa")
public class TwoFactorSettingsController {

    private final TotpService totpService;
    private final QrCodeGenerator qrCodeGenerator;
    private final UserMapper userMapper;

    // コンストラクタ省略

    /**
     * 2FA設定画面を表示
     */
    @GetMapping
    public String showSettings(@AuthenticationPrincipal UserDetails user,
                               Model model) {
        User u = userMapper.findByUsername(user.getUsername());
        model.addAttribute("twoFactorEnabled",
            u.getTotpSecret() != null);
        return "settings/2fa-settings";
    }

    /**
     * 2FAセットアップ開始（QRコード表示）
     */
    @GetMapping("/setup")
    public String setup(@AuthenticationPrincipal UserDetails user,
                        Model model) {
        String secret = totpService.generateSecret();
        String qrImage = qrCodeGenerator.generateQrBase64(
            user.getUsername(), secret);

        model.addAttribute("secret", secret);
        model.addAttribute("qrCodeImage", qrImage);
        return "settings/2fa-setup";
    }

    /**
     * 2FAを有効化（確認コード検証後）
     */
    @PostMapping("/verify")
    public String enableTwoFactor(
            @AuthenticationPrincipal UserDetails user,
            @RequestParam String secret,
            @RequestParam String code,
            RedirectAttributes redirectAttributes) {

        if (totpService.verifyCode(secret, code)) {
            User u = userMapper.findByUsername(user.getUsername());
            u.setTotpSecret(secret);
            userMapper.update(u);
            redirectAttributes.addFlashAttribute("message",
                "二要素認証を有効にしました");
        } else {
            redirectAttributes.addFlashAttribute("error",
                "認証コードが正しくありません");
            return "redirect:/settings/2fa/setup";
        }
        return "redirect:/settings/2fa";
    }

    /**
     * 2FAを無効化
     */
    @PostMapping("/disable")
    public String disableTwoFactor(
            @AuthenticationPrincipal UserDetails user,
            RedirectAttributes redirectAttributes) {
        User u = userMapper.findByUsername(user.getUsername());
        u.setTotpSecret(null);
        userMapper.update(u);
        redirectAttributes.addFlashAttribute("message",
            "二要素認証を無効にしました");
        return "redirect:/settings/2fa";
    }
}
