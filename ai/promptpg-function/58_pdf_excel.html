Section 58-1

Apache POIでExcel出力

Apache POIライブラリを使って、タスク一覧をExcelファイル（.xlsx）としてエクスポートする機能を実装します。ヘッダー行のスタイルやセル幅の自動調整も行います。

build.gradle（依存関係） 
dependencies {
    // Apache POI（Excel出力）
    implementation 'org.apache.poi:poi-ooxml:5.2.5'

    // iText（PDF出力）
    implementation 'com.itextpdf:itext7-core:8.0.3'
}
ExcelExportService.java（Excel生成） 
import org.apache.poi.ss.usermodel.*;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;
import org.springframework.stereotype.Service;
import java.io.ByteArrayOutputStream;
import java.util.List;

@Service
public class ExcelExportService {

    /**
     * タスク一覧をExcelファイルとして生成
     */
    public byte[] exportTodos(List<Todo> todos) throws Exception {
        try (Workbook workbook = new XSSFWorkbook()) {
            Sheet sheet = workbook.createSheet("タスク一覧");

            // ヘッダースタイル
            CellStyle headerStyle = workbook.createCellStyle();
            headerStyle.setFillForegroundColor(
                IndexedColors.ROYAL_BLUE.getIndex());
            headerStyle.setFillPattern(FillPatternType.SOLID_FOREGROUND);
            Font headerFont = workbook.createFont();
            headerFont.setColor(IndexedColors.WHITE.getIndex());
            headerFont.setBold(true);
            headerStyle.setFont(headerFont);

            // ヘッダー行
            String[] headers = {"ID", "タイトル", "ステータス",
                "優先度", "期限", "カテゴリ"};
            Row headerRow = sheet.createRow(0);
            for (int i = 0; i < headers.length; i++) {
                Cell cell = headerRow.createCell(i);
                cell.setCellValue(headers[i]);
                cell.setCellStyle(headerStyle);
            }

            // データ行
            for (int i = 0; i < todos.size(); i++) {
                Todo todo = todos.get(i);
                Row row = sheet.createRow(i + 1);
                row.createCell(0).setCellValue(todo.getId());
                row.createCell(1).setCellValue(todo.getTitle());
                row.createCell(2).setCellValue(todo.getStatus());
                row.createCell(3).setCellValue(todo.getPriority());
                row.createCell(4).setCellValue(
                    todo.getDeadline() != null
                        ? todo.getDeadline().toString() : "");
                row.createCell(5).setCellValue(
                    todo.getCategory() != null
                        ? todo.getCategory() : "");
            }

            // 列幅を自動調整
            for (int i = 0; i < headers.length; i++) {
                sheet.autoSizeColumn(i);
            }

            ByteArrayOutputStream out = new ByteArrayOutputStream();
            workbook.write(out);
            return out.toByteArray();
        }
    }
}
lightbulbXSSFWorkbook vs SXSSFWorkbook

XSSFWorkbookは全データをメモリに保持するため、大量データにはメモリ効率が良いSXSSFWorkbook（ストリーミング版）を使います。
数百件程度ならXSSFWorkbookで十分です。

PdfExportService.java（PDF生成） 
import com.itextpdf.kernel.colors.ColorConstants;
import com.itextpdf.kernel.font.PdfFont;
import com.itextpdf.kernel.font.PdfFontFactory;
import com.itextpdf.kernel.pdf.PdfDocument;
import com.itextpdf.kernel.pdf.PdfWriter;
import com.itextpdf.layout.Document;
import com.itextpdf.layout.element.*;
import com.itextpdf.layout.properties.TextAlignment;
import com.itextpdf.layout.properties.UnitValue;
import org.springframework.stereotype.Service;
import java.io.ByteArrayOutputStream;
import java.util.List;

@Service
public class PdfExportService {

    /**
     * タスク一覧をPDFとして生成
     */
    public byte[] exportTodos(List<Todo> todos) throws Exception {
        ByteArrayOutputStream out = new ByteArrayOutputStream();

        PdfWriter writer = new PdfWriter(out);
        PdfDocument pdfDoc = new PdfDocument(writer);
        Document document = new Document(pdfDoc);

        // 日本語フォント設定
        PdfFont font = PdfFontFactory.createFont(
            "src/main/resources/fonts/NotoSansJP-Regular.ttf",
            "Identity-H");
        document.setFont(font);

        // タイトル
        document.add(new Paragraph("タスク一覧")
            .setFontSize(18)
            .setBold()
            .setTextAlignment(TextAlignment.CENTER));

        document.add(new Paragraph(" ")); // 余白

        // テーブル作成
        float[] columnWidths = {1, 4, 2, 2, 3};
        Table table = new Table(
            UnitValue.createPercentArray(columnWidths));
        table.setWidth(UnitValue.createPercentValue(100));

        // ヘッダー行
        String[] headers = {"ID", "タイトル", "ステータス",
            "優先度", "期限"};
        for (String header : headers) {
            table.addHeaderCell(new Cell()
                .add(new Paragraph(header).setBold())
                .setBackgroundColor(ColorConstants.LIGHT_GRAY)
                .setTextAlignment(TextAlignment.CENTER));
        }

        // データ行
        for (Todo todo : todos) {
            table.addCell(
                String.valueOf(todo.getId()));
            table.addCell(todo.getTitle());
            table.addCell(todo.getStatus());
            table.addCell(todo.getPriority());
            table.addCell(todo.getDeadline() != null
                ? todo.getDeadline().toString() : "-");
        }

        document.add(table);
        document.close();

        return out.toByteArray();
    }
}
lightbulb日本語フォントの配置

iTextで日本語を表示するには、src/main/resources/fonts/にNoto Sans JPなどのTTFフォントファイルを配置する必要があります。
フォントを埋め込まないと日本語が文字化けします。

エクスポートコントローラー

Excel/PDFの生成サービスを呼び出し、適切なContent-TypeとContent-Dispositionヘッダーを設定してファイルをダウンロードさせるコントローラーを実装します。

text_fields

ファイル名

Content-Dispositionでダウンロードファイル名を指定

ExportController.java 
import org.springframework.http.*;
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import java.time.LocalDate;
import java.util.List;

@Controller
@RequestMapping("/export")
public class ExportController {

    private final TodoService todoService;
    private final ExcelExportService excelService;
    private final PdfExportService pdfService;

    public ExportController(TodoService todoService,
                            ExcelExportService excelService,
                            PdfExportService pdfService) {
        this.todoService = todoService;
        this.excelService = excelService;
        this.pdfService = pdfService;
    }

    /**
     * Excel形式でエクスポート
     */
    @GetMapping("/excel")
    public ResponseEntity<byte[]> exportExcel(
            @AuthenticationPrincipal UserDetails user)
            throws Exception {

        List<Todo> todos = todoService.findByUsername(
            user.getUsername());
        byte[] data = excelService.exportTodos(todos);

        String filename = "todos_"
            + LocalDate.now() + ".xlsx";

        return ResponseEntity.ok()
            .header(HttpHeaders.CONTENT_DISPOSITION,
                "attachment; filename=\"" + filename + "\"")
            .contentType(MediaType.parseMediaType(
                "application/vnd.openxmlformats-officedocument"
                + ".spreadsheetml.sheet"))
            .body(data);
    }

    /**
     * PDF形式でエクスポート
     */
    @GetMapping("/pdf")
    public ResponseEntity<byte[]> exportPdf(
            @AuthenticationPrincipal UserDetails user)
            throws Exception {

        List<Todo> todos = todoService.findByUsername(
            user.getUsername());
        byte[] data = pdfService.exportTodos(todos);

        String filename = "todos_"
            + LocalDate.now() + ".pdf";

        return ResponseEntity.ok()
            .header(HttpHeaders.CONTENT_DISPOSITION,
                "attachment; filename=\"" + filename + "\"")
            .contentType(MediaType.APPLICATION_PDF)
            .body(data);
    }
}
lightbulbContent-Dispositionヘッダー

attachmentを指定するとブラウザはファイルをダウンロードします。inlineにするとブラウザ内で表示（PDFの場合）できます。
ファイル名に日本語を含む場合はRFC 5987に従ってエンコードが必要です。

edit_note 基本プロンプト 
Spring Boot 3.xのToDoアプリにPDF/Excelエクスポート機能を追加してください。

【要件】
- Apache POIでタスク一覧をExcel(.xlsx)にエクスポート（ヘッダースタイル付き）
- iText 8でタスク一覧をPDFにエクスポート（日本語フォント対応）
- /export/excel、/export/pdfのエンドポイントを作成
- ファイル名に日付を含める（todos_2025-01-15.xlsx）
- タスク一覧画面にドロップダウン形式のダウンロードボタン

【技術スタック】
- Apache POI 5.x（Excel）
- iText 8.x（PDF）
- Spring Boot 3.x + MyBatis
- ResponseEntity<byte[]>でファイル返却
gps_fixed ステップ別プロンプト 
# Step 1: 依存関係追加
build.gradleにApache POIとiTextの依存関係を追加してください。

# Step 2: Excel出力サービス
Apache POIを使ってタスク一覧をExcelファイルに変換するサービスを作成してください。ヘッダー行のスタイル、列幅の自動調整を含めてください。

# Step 3: PDF出力サービス
iText 8を使ってタスク一覧をPDFに変換するサービスを作成してください。日本語フォント（NotoSansJP）に対応してください。

# Step 4: ダウンロードUI
タスク一覧画面にExcel/PDFのダウンロードボタンをドロップダウンメニュー形式で追加してください。
