schema.sql（favoriteテーブル）
CREATE TABLE favorite (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    todo_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_user_todo (user_id, todo_id),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (todo_id) REFERENCES todo(id) ON DELETE CASCADE
);
FavoriteMapper.java / FavoriteMapper.xml
// --- FavoriteMapper.java ---
@Mapper
public interface FavoriteMapper {
    void insert(@Param("userId") Integer userId,
                @Param("todoId") Integer todoId);
    void delete(@Param("userId") Integer userId,
                @Param("todoId") Integer todoId);
    boolean exists(@Param("userId") Integer userId,
                   @Param("todoId") Integer todoId);
    List<Integer> findTodoIdsByUserId(Integer userId);
}

// --- FavoriteMapper.xml ---
<insert id="insert">
    INSERT INTO favorite (user_id, todo_id)
    VALUES (#{userId}, #{todoId})
</insert>

<delete id="delete">
    DELETE FROM favorite
    WHERE user_id = #{userId} AND todo_id = #{todoId}
</delete>

<select id="exists" resultType="boolean">
    SELECT COUNT(*) > 0 FROM favorite
    WHERE user_id = #{userId} AND todo_id = #{todoId}
</select>

<select id="findTodoIdsByUserId" resultType="int">
    SELECT todo_id FROM favorite
    WHERE user_id = #{userId}
    ORDER BY created_at DESC
</select>
lightbulbON DELETE CASCADE

ユーザーやToDoが削除された場合に、関連するお気に入りレコードも自動削除されます。ON DELETE CASCADEで外部キー制約と連動させることで、データの整合性を保ちます。

Section 54-2

お気に入り追加/解除API

お気に入りの追加・解除を行うREST APIエンドポイントを実装します。トグル方式で1つのエンドポイントで両方を処理します。

toggle_on

トグルAPI

POSTでお気に入り状態をトグル（追加/解除）

security

認証ユーザー取得

@AuthenticationPrincipalでログインユーザーを特定

sync

レスポンス

現在のお気に入り状態をJSONで返却

FavoriteApiController.java
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

@RestController
@RequestMapping("/api/favorites")
public class FavoriteApiController {

    private final FavoriteMapper favoriteMapper;
    private final UserMapper userMapper;

    public FavoriteApiController(FavoriteMapper favoriteMapper,
                                 UserMapper userMapper) {
        this.favoriteMapper = favoriteMapper;
        this.userMapper = userMapper;
    }

    /**
     * お気に入りをトグル（追加/解除）
     */
    @PostMapping("/{todoId}")
    public Map<String, Object> toggle(
            @PathVariable Integer todoId,
            @AuthenticationPrincipal UserDetails userDetails) {

        User user = userMapper.findByUsername(userDetails.getUsername());
        boolean isFavorite = favoriteMapper.exists(user.getId(), todoId);

        if (isFavorite) {
            favoriteMapper.delete(user.getId(), todoId);
        } else {
            favoriteMapper.insert(user.getId(), todoId);
        }

        return Map.of("favorited", !isFavorite);
    }
}
lightbulbトグルAPIのメリット

追加と解除を別エンドポイントにする代わりに1つのPOSTエンドポイントでトグルすることで、フロントエンドの実装がシンプルになります。レスポンスで現在の状態を返せば、UIの更新も容易です。

<script>
function toggleFavorite(el) {
    var todoId = el.dataset.todoId;

    fetch('/api/favorites/' + todoId, { method: 'POST' })
        .then(function(res) { return res.json(); })
        .then(function(data) {
            var icon = el.querySelector('.material-icons');
            if (data.favorited) {
                el.classList.add('active');
                icon.textContent = 'star';
            } else {
                el.classList.remove('active');
                icon.textContent = 'star_border';
            }
            el.classList.add('animating');
            setTimeout(function() { el.classList.remove('animating'); }, 300);
        });
}
</script>

// --- TodoController.java に追加 ---
@GetMapping("/todo")
public String list(@RequestParam(required = false) Boolean favorites,
                   @AuthenticationPrincipal UserDetails userDetails,
                   Model model) {

    User user = userMapper.findByUsername(userDetails.getUsername());

    if (Boolean.TRUE.equals(favorites)) {
        model.addAttribute("todos",
            todoMapper.findFavoritesByUserId(user.getId()));
        model.addAttribute("favoritesOnly", true);
    } else {
        model.addAttribute("todos",
            todoMapper.findByUserId(user.getId()));
    }

    // お気に入りIDリスト（スター表示用）
    model.addAttribute("favoriteIds",
        favoriteMapper.findTodoIdsByUserId(user.getId()));

    return "todo/list";
}

// --- TodoMapper.xml ---
<select id="findFavoritesByUserId" resultType="Todo">
    SELECT t.* FROM todo t
    JOIN favorite f ON t.id = f.todo_id
    WHERE f.user_id = #{userId}
    ORDER BY f.created_at DESC
</select>

edit_note 基本プロンプト
Spring Boot 3.xのToDoアプリにお気に入り機能を追加してください。

【要件】
- favoriteテーブル（user_id, todo_id複合ユニーク）でお気に入り状態を管理
- REST APIでお気に入りのトグル（追加/解除）
- タスク一覧にスターアイコンを表示、クリックでfetch APIトグル
- お気に入りのみ表示するフィルタボタン
- スターのクリックにスケールアニメーション

【技術スタック】
- Spring Boot 3.x + MyBatis
- @AuthenticationPrincipal（ログインユーザー取得）
- Material Icons（star/star_border）
- fetch API + CSS animation
gps_fixed ステップ別プロンプト
# Step 1: favoriteテーブルとMapper
user_id, todo_idの複合ユニークキーでfavoriteテーブルを作成し、insert/delete/exists/findTodoIdsByUserIdのMapperを実装してください。

# Step 2: トグルREST API
POST /api/favorites/{todoId}でお気に入りをトグルし、現在の状態をJSONで返すRestControllerを作成してください。

# Step 3: スターアイコンUI
タスク一覧の各行にMaterial Iconsのstar/star_borderを表示し、クリックでfetch APIを呼び出してトグルするJavaScriptを実装してください。

# Step 4: お気に入りフィルタ
?favorites=trueでお気に入りのみ表示するフィルタボタンと、JOINで絞り込むSQLを実装してください。
lightbulbプロンプトのコツ

「スターのクリックでページ遷移させず、非同期でUI更新してください」のように、UXの要件を明示するとSPA風の快適な操作が実現できます。
